\textbf{Intersection Multiplicity.} If $Y, Z \subseteq \mathbf{A}^2$ are two distinct curves, given by equations $f = 0$, $g = 0$, and if $P \in Y \cap Z$, we define the \textbf{intersection multiplicity} $(Y \cdot Z)_P$ of $Y$ and $Z$ at $P$ to be the length of the $\mathcal{O}_P$-module $\mathcal{O}_P/(f,g)$.

\begin{enumerate}
\item[(a)] Show that $(Y \cdot Z)_P$ is finite, and $(Y \cdot Z)_P \geq \mu_P(Y) \cdot \mu_P(Z)$.
\item[(b)] If $P \in Y$, show that for almost all lines $L$ through $P$ (i.e., all but a finite number), $(Y \cdot L)_P = \mu_P(Y)$.
\end{enumerate}
