Let $Q$ be the maximal $p$-abelian $p$-ramified extension of $K_\infty$, and let $G = \operatorname{Gal}(Q/K_\infty)$. Then:

\begin{enumerate}
\item[(i)] $G$ is finitely generated over the Iwasawa algebra $\Lambda$, and
$$G/G^{\gamma-1} \cong \mathbb{Z}_p^\rho$$
where $\rho = [K_0 : \mathbb{Q}] - r_p - 1$ and $r_p$ is the $\mathbb{Z}_p$-rank of the closure of the global units in the local units.

\item[(ii)] If $K_0$ satisfies the Leopoldt conjecture, then $\rho = r_2$ (the number of complex places of $K_0$), and
$$G/G^{\gamma-1} \cong \mathbb{Z}_p^{r_2}.$$
\end{enumerate}
