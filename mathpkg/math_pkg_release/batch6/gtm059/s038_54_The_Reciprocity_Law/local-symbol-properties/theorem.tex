\begin{proposition}
The Kummer pairing $\langle \cdot, \cdot \rangle_{A,\pi,n} : A(\mathfrak{m}_0) \times K^{*} \to A[\pi^n]$ satisfies the following properties.

\begin{enumerate}
\item The pairing is biadditive: for $x_1, x_2 \in A(\mathfrak{m}_0)$ and $a \in K^{*}$,
$$\langle x_1 +_A x_2, a \rangle = \langle x_1, a \rangle +_A \langle x_2, a \rangle,$$
and for $x \in A(\mathfrak{m}_0)$ and $a_1, a_2 \in K^{*}$,
$$\langle x, a_1 a_2 \rangle = \langle x, a_1 \rangle +_A \langle x, a_2 \rangle.$$

\item The symbol induces a non-degenerate pairing
$$A(\mathfrak{m}_0) / [\pi^n]_A A(\mathfrak{m}_0) \times K^{*} / K^{* \pi^n} \to A[\pi^n],$$
and the kernel on the left is exactly $[\pi^n]_A A(\mathfrak{m}_0)$.

\item For $a \in K^{*}$ and $x \in A(\mathfrak{m}_0)$, the pairing satisfies the compatibility with the reciprocity map:
$$r_a(t) -_A t = \langle x, a \rangle$$
for any $t$ with $[\pi^n]_A(t) = x$.

\item The symbol is functorial in the formal group $A$ and compatible with base change.
\end{enumerate}
\end{proposition}
