Let $\chi$ be a character of $F_q^*$ of order $m > 1$. Let $\zeta$ be a primitive $(q-1)$-th root of unity, and let $\zeta_p$ be a primitive $p$-th root of unity.

\begin{enumerate}
\item $S(\chi)^m$ lies in $\mathbb{Q}(\zeta_m)$.
\item $S(\chi)$ lies in $\mathbb{Q}(\zeta_m, \zeta_p)$.
\end{enumerate}
