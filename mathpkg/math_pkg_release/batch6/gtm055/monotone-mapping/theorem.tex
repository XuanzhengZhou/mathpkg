Let $X$ and $Y$ be partially ordered sets. A mapping $f : X \to Y$ is:
\begin{itemize}
\item \textbf{monotone increasing} if $x \leq y$ implies $f(x) \leq f(y)$,
\item \textbf{monotone decreasing} if $x \leq y$ implies $f(x) \geq f(y)$,
\item \textbf{strictly monotone increasing} if $x < y$ implies $f(x) < f(y)$,
\item \textbf{strictly monotone decreasing} if $x < y$ implies $f(x) > f(y)$.
\end{itemize}
