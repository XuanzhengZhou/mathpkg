For a multigraph $\Gamma$ and edge $e \in E$:
\begin{enumerate}
\item[(a)] If $e$ is not an isthmus, then $\dim(\mathcal{Z}(\Gamma)) = \dim(\mathcal{Z}(\Gamma_{(e)})) + 1$ and $\dim(\mathcal{Z}^\perp(\Gamma)) = \dim(\mathcal{Z}^\perp(\Gamma_{(e)}))$.
\item[(b)] If $e$ is an isthmus, then $\dim(\mathcal{Z}(\Gamma)) = \dim(\mathcal{Z}(\Gamma_{(e)}))$ and $\dim(\mathcal{Z}^\perp(\Gamma)) = \dim(\mathcal{Z}^\perp(\Gamma_{(e)})) + 1$.
\item[(c)] $\nu_{-1}(\Gamma_{(e)}) - \nu_{-1}(\Gamma) = 0$ if $e$ is not an isthmus, and $=1$ if $e$ is an isthmus.
\end{enumerate}
