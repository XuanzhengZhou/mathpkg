Let $F \subseteq E$. The following six statements are equivalent:
\begin{enumerate}
\item[(a)] $|F| = \dim(\mathcal{Z}(\Gamma_{(E+F)})) = \dim(\mathcal{Z}(\Gamma))$;
\item[(b)] $\Gamma_{(E+F)}$ is a spanning coforest;
\item[(c)] $F$ is a maximal subset of $E$ which contains no nonempty cocycle of $\Gamma$;
\item[(d)] $F$ is a minimal subset of $E$ which meets every nonempty cycle of $\Gamma$;
\item[(e)] $\mathcal{Z}(\Gamma)$ admits a basis $\{Z_e : e \in F\}$ such that for each $e \in F$, $Z_e \cap F = \{e\}$;
\item[(f)] $\mathcal{Z}^\perp(\Gamma)$ admits a basis $\{C_e : e \in E + F\}$ such that for each $e \in E + F$, $C_e \cap (E + F) = \{e\}$.
\end{enumerate}
