$$
\text{Let }\mathcal{I} \subseteq \mathcal{P}(V)\text{ and let }r\colon \mathcal{P}(V)\to \mathbb{Z}_{\geq 0}.
$$
$$
\text{The following two statements are equivalent:}
$$

\begin{enumerate}
\item[(a)] $(V, \mathcal{I})$ is an independence system and
$$
r(U) = \max\{|J| : J \in \mathcal{I},\; J \subseteq U\}, \quad U \in \mathcal{P}(V);
$$

\item[(b)] $(V, r)$ is a rank structure and
$$
\mathcal{I} = \{J \in \mathcal{P}(V) : r(J) = |J|\}.
$$
\end{enumerate}
