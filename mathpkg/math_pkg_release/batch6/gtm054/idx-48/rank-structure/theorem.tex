$$
\text{Let }V\text{ be a set and let }r\colon \mathcal{P}(V)\to \mathbb{Z}_{\geq 0}\text{ be a function.}
$$
$$
\text{The pair }(V, r)\text{ is called a }\textit{rank structure}\text{ if for all }U \in \mathcal{P}(V)\text{ and all }x_1, x_2 \in V \setminus U:
$$
\begin{align}
\text{(C1)}\quad & r(\emptyset) = 0; \\
\text{(C2)}\quad & 0 \leq r(U \cup \{x_1\}) - r(U) \leq 1; \\
\text{(C3)}\quad & r(U) = r(U \cup \{x_1\}) = r(U \cup \{x_2\}) \;\Longrightarrow\; r(U) = r(U \cup \{x_1, x_2\}).
\end{align}
$$
