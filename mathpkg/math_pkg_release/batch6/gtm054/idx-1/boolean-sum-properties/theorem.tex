For $R, S, T \in \mathcal{P}(U)$:
\begin{align*}
\text{A1}&\quad S + T = T + S \\
\text{A2}&\quad (R + S) + T = R + (S + T) \\
\text{A3}&\quad S + T = S \Leftrightarrow T = \varnothing \\
\text{A4}&\quad S + T = \varnothing \Leftrightarrow S = T \\
\text{A5}&\quad S + T = (S \cup T) + (S \cap T) \\
\text{A6}&\quad R \cup (S + T) \supseteq (R \cup S) + (R \cup T) \\
\text{A7}&\quad R \cap (S + T) = (R \cap S) + (R \cap T) \\
\text{A8}&\quad R + (S \cap T) \supseteq (R + S) \cap (R + T) \\
\text{A9}&\quad (R + S) \cap (R + T) \subseteq R + (S \cup T) \subseteq (R + S) \cup (R + T)
\end{align*}
