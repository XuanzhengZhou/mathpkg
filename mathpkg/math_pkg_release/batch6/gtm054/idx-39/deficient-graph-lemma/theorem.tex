Let $\Gamma = (V, \mathcal{E})$ be a $d$-deficient $(q_1, q_2)$-graph and let $x \in V$. Let $S = N(x)$ and let $T = V \setminus (S \cup \{x\})$. Then

\begin{itemize}
\item[(a)] $\Gamma_S$ is a $(q_1 - 1, q_2)$-graph and $\Gamma_T$ is a $(q_1, q_2 - 1)$-graph.
\item[(b)] If $\Gamma_S$ has deficiency $d_S$ and $\Gamma_T$ has deficiency $d_T$, then $d \geq d_S + d_T + 1$.
\item[(c)] If $\Gamma$ is $0$-deficient, then $d_S = d_T = 0$ and $2 \leq \rho(x) \leq 3$ for all $x \in V$, where $\rho(x)$ denotes the valency of $x$.
\end{itemize}
