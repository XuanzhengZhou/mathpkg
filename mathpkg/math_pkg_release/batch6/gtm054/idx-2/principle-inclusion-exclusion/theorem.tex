Let $(V, f, E)$ be a system with selection $s$.

\begin{enumerate}
\item[(a)] For $r = 0, 1, \ldots, |E|$, the number of vertices belonging to precisely $r$ blocks is
$$\sum_{i=0}^{|E|-r} (-1)^i \binom{r+i}{i} \sum_{|A|=r+i} \bar{s}^*(A).$$

\item[(b)] For $k = 0, 1, \ldots, |V|$, the number of blocks of size $k$ is
$$\sum_{i=0}^{|V|-k} (-1)^i \binom{k+i}{i} \sum_{|S|=k+i} \bar{s}(S).$$
\end{enumerate}
