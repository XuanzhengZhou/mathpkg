\begin{proposition}[3.7.5]
\leavevmode
\begin{enumerate}
\item[(i)] On a ruled surface, generators are asymptotic curves. Consequently, $K \leq 0$.
\item[(ii)] A ruled surface $f$ is developable $\Leftrightarrow$ in $(s, t)$-parameters $f_{st}$ is a linear combination of $f_s$ and $f_t$ $\Leftrightarrow$ $K = 0$ on $f$.
\end{enumerate}
\end{proposition}
