Let $f: U \rightarrow \mathbb{R}^3$ be a surface.

\begin{enumerate}
\item[i)] Let $B: \mathbb{R}^3 \rightarrow \mathbb{R}^3$ be an isometry. Then $\tilde{f} := B \circ f$ is also a surface and
$$\tilde{I}_u(dB X, dB Y) = I_u(X, Y) \quad \text{for all } X, Y \in T_u f.$$

\item[ii)] Let $\phi: V \rightarrow U$ be a change of variables and let $\tilde{f} = f \circ \phi$. Then
$$\tilde{I}_v(X, Y) = I_{\phi(v)}(X, Y) \quad \text{for all } X, Y \in T_v \tilde{f} = T_{\phi(v)} f$$
and
$$\tilde{I}_v(\tilde{X}, \tilde{Y}) = I_{\phi(v)}(d\phi \tilde{X}, d\phi \tilde{Y}) \quad \text{for all } \tilde{X}, \tilde{Y} \in T_v \mathbb{R}^2.$$
\end{enumerate}
