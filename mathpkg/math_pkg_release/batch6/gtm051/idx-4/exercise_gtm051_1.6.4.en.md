\begin{enumerate}
  \item[(i)] Determine the curvature and torsion of the "elliptical helix" $c(t) = (a \cos t,\ b \sin t,\ ct)$, $t \in \mathbb{R}$, $abc \neq 0$.
  \item[(ii)] Determine the curvature and torsion of the curve given by $c(t) = (t, t^2, t^3)$.
\end{enumerate}
