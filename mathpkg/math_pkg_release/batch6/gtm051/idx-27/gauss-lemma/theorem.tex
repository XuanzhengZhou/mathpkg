\text{Polar coordinates are geodesic coordinates based on a geodesic circle.}

\text{Let }(U,g)\text{ be a coordinate system on }M\text{. Define }

\phi: V := ]-\rho,\rho[ \times ]-\pi,\pi[ \to B_\rho(u_0),

(v^1,v^2) \mapsto \exp_{u_0}(v^1\cos v^2\,e_1+v^1\sin v^2\,e_2)=:(u^1,u^2).

\text{When }v^1>0\text{, this is a geodesic coordinate system, i.e., }
\tilde{g}_{11}=1,\;\tilde{g}_{12}=0,\;\tilde{g}_{22}>0.
