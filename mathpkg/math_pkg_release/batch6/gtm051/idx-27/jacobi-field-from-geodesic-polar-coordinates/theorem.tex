\text{Let }c(t),\;0\leq t\leq a,\text{ be a geodesic with }|\dot{c}(t)|=1\text{ and call }c(0)=p.

\text{Define the map }\phi(t,\theta)=\exp_p(t\cos\theta\,e_1+t\sin\theta\,e_2)\text{ for }(t,\theta)\text{ near }[0,a]\times\{0\}.

\text{Then the vector field }Y(t)=\frac{\partial\phi}{\partial\theta}(t,0)=d\phi(e_2(t,0))\text{ is a Jacobi field along }c(t)\text{ with }Y(0)=0\text{ and }(\nabla Y/dt)(0)=e_2(p).
