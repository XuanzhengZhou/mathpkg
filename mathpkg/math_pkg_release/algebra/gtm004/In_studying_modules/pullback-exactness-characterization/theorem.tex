\text{Let $\varphi: A \to X$ and $\psi: B \to X$ be homomorphisms of $\Lambda$-modules. Then the square}
$$
\begin{CD}
Y @>{\alpha}>> A \\
@V{\beta}VV @VV{\varphi}V \\
B @>>{\psi}> X
\end{CD}
$$
\text{is a pull-back diagram if and only if the sequence}
$$
0 \to Y \xrightarrow{\{\alpha, \beta\}} A \oplus B \xrightarrow{\langle \varphi, -\psi \rangle} X
$$
\text{is exact, where $\{\alpha, \beta\}(y) = (\alpha y, \beta y)$ and $\langle \varphi, -\psi \rangle(a, b) = \varphi a - \psi b$.}
