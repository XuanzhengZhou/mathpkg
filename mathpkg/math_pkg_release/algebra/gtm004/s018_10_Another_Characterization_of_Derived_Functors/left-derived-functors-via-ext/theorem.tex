\begin{corollary}\label{cor:10.2}
Let $T \colon \mathfrak{M}_\Lambda \to \mathfrak{Ab}$ be a right exact functor and let $A$ be a $\Lambda$-module. Then there are natural isomorphisms
$$
\Gamma \colon [\operatorname{Ext}_\Lambda^q(A, -),\, T] \xrightarrow{\cong} L_q TA, \qquad q = 0, 1, \dots,
$$
where $L_q T$ denotes the $q$-th left derived functor of $T$.
\end{corollary}
