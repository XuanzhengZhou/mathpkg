\begin{theorem}[First Long Exact Sequence of Derived Functors]
Let $T: \mathfrak{M}_A \rightarrow \mathfrak{Ub}$ be an additive functor and let
\[
A' \xrightarrow{\mu} A \xrightarrow{\varepsilon} A''
\]
be a short exact sequence in $\mathfrak{M}_A$. Then there exist connecting homomorphisms
\[
\omega_n: L_n T A'' \rightarrow L_{n-1} T A', \quad n = 1, 2, \ldots
\]
such that the following sequence is exact:
\[
\cdots \rightarrow L_n T A' \xrightarrow{L_n T \mu} L_n T A \xrightarrow{L_n T \varepsilon} L_n T A'' \xrightarrow{\omega_n} L_{n-1} T A' \rightarrow \cdots
\]
\[
\cdots \rightarrow L_1 T A'' \xrightarrow{\omega_1} L_0 T A' \xrightarrow{L_0 T \mu} L_0 T A \xrightarrow{L_0 T \varepsilon} L_0 T A'' \rightarrow 0.
\]
\end{theorem}
