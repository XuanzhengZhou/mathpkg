\begin{theorem}[Theorem 4.5]
The ladder sequence
\[
D_m \xrightarrow{\beta_{m,n}} E_{m,n} \xrightarrow{\gamma_{m,n}} D_n \xrightarrow{\alpha_n} D_n
\tag{4.17}
\]
is exact. More precisely:
\begin{enumerate}
\item[(i)] $\ker\alpha_n = \operatorname{im}\gamma_{m,n}$;
\item[(ii)] $\ker\gamma_{m,n} = \operatorname{im}\beta_{m,n}$;
\item[(iii)] $\ker\beta_{m,n} = \operatorname{im}\alpha_m$.
\end{enumerate}
Moreover,
\[
E_{m,n} = \frac{\gamma^{-1}(\alpha^n D)}{\beta\alpha^{-m}(0)}.
\tag{4.18}
\]
In particular, $E_{n,n} = E_n$ and (4.17) for $m=n$ is exactly the $n^{\text{th}}$ derived couple.
\end{theorem}
