Let $H$ be a subgroup of $G$ and let $\rho: H \to \operatorname{GL}(W)$ be a representation of $H$. For $s \in G$, set $H_s = sHs^{-1} \cap H$ and define two representations of $H_s$:
\begin{itemize}
  \item $\operatorname{Res}_s(\rho)$: the restriction of $\rho$ to $H_s$,
  \item $\rho^s$: the representation of $H_s$ defined by $\rho^s(x) = \rho(s^{-1}xs)$ for $x \in H_s$.
\end{itemize}

\textbf{Proposition 23.} In order that the induced representation $V = \operatorname{Ind}_H^G(W)$ be irreducible, it is necessary and sufficient that the following two conditions be satisfied:
\begin{enumerate}
  \item[(a)] $W$ is irreducible.
  \item[(b)] For each $s \in G \setminus H$, the two representations $\operatorname{Res}_s(\rho)$ and $\rho^s$ of $H_s$ are disjoint (i.e., have no common irreducible component).
\end{enumerate}
