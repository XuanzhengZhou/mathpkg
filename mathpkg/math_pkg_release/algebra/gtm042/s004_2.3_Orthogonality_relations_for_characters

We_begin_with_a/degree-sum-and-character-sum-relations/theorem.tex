Let $G$ be a finite group of order $g$, and let $\chi_1, \ldots, \chi_h$ be the distinct irreducible characters of $G$ with degrees $n_i = \chi_i(1)$.

\begin{enumerate}
\item[(a)] The degrees $n_i$ satisfy the relation
$$\sum_{i=1}^{h} n_i^2 = g.$$
\item[(b)] If $s \in G$ is different from $1$, then
$$\sum_{i=1}^{h} n_i \chi_i(s) = 0.$$
\end{enumerate}
