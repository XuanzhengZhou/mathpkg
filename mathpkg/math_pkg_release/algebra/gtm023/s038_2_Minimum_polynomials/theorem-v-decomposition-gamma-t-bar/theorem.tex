\text{\textbf{Theorem V.}} \text{ Let $f = f_1^{k_1} \cdots f_r^{k_r}$ be the decomposition of $f$ into its prime factors and set $g = f_1 \cdots f_r$. Consider the factor algebra $\Gamma(\bar{t}) = \Gamma[t] / I_f$. Let $\bar{t} = \bar{t}_S + \bar{t}_N$ be the unique decomposition of Theorem IV, and let $A$ be the subalgebra of $\Gamma(\bar{t})$ generated by $\bar{1}$ and $\bar{t}_S$. Let $I_i$ be the ideal in $\Gamma[\bar{t}]$ generated by $\bar{f}_i$ ($i = 1, \dots, r$). Then:}
\begin{enumerate}
\item $A$ is the (unique) direct sum of irreducible ideals $I_i$ ($i = 1, \dots, r$),
  $$A = I_1 \oplus \cdots \oplus I_r,$$
  and each $I_i$ is isomorphic to the field $\Gamma[t] / I_{f_i}$.
\item The vector space $\Gamma(\bar{t})$ is the direct sum of the subalgebra $A$ and the radical $\operatorname{rad} \Gamma(\bar{t})$,
  $$\Gamma(\bar{t}) = A \oplus \operatorname{rad} \Gamma(\bar{t}).$$
\item The subalgebra $A$ consists precisely of the semisimple elements in $\Gamma(\bar{t})$.
\item $A$ is the only subalgebra of $\Gamma(\bar{t})$ which complements the radical.
\end{enumerate}
