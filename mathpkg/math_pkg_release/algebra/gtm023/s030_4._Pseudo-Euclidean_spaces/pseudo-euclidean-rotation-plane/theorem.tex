Let $E$ be a 2-dimensional pseudo-Euclidean space of index $1$, with light-cone generators $l_1, l_2$ satisfying $(l_1, l_2) = -1$. Let $\varphi$ be a proper pseudo-Euclidean rotation ($\det \varphi = +1$) that does not interchange $T^+$ and $T^-$. Then:

\begin{enumerate}
  \item $\varphi l_1 = \lambda l_1$ and $\varphi l_2 = \frac{1}{\lambda} l_2$ for some $\lambda > 0$.
  \item For any vector $x = \xi^1 l_1 + \xi^2 l_2$,
    $$(x, \varphi x) = \frac{1}{2}\left(\lambda + \frac{1}{\lambda}\right)(x, x).$$
  \item The pseudo-Euclidean angle $\theta$ between $x$ and $\varphi x$ satisfies
    $$\cosh \theta = \frac{1}{2}\left(\lambda + \frac{1}{\lambda}\right), \quad \sinh \theta = \frac{1}{2}\left(\lambda - \frac{1}{\lambda}\right).$$
\end{enumerate}
