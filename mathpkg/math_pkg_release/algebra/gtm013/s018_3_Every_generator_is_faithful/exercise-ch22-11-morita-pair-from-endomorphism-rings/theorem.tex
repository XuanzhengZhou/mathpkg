Let $M_K$ and $N_K$ be non-zero modules over a ring $K$. Let $R = \operatorname{End}(M_K)$ and $S = \operatorname{End}(N_K)$. Consider the bimodules
$${}_S P_R = \operatorname{Hom}_K({}_R M_K, {}_S N_K) \quad \text{and} \quad {}_R Q_S = \operatorname{Hom}_K({}_S N_K, {}_R M_K).$$
Prove that:
\begin{enumerate}
    \item[(1)] There is a Morita pair $(\theta, \phi)$ for $(P, Q)$ such that for each $f \in P$ and $g \in Q$,
    $$\theta(f \otimes g) = fg \quad \text{and} \quad \phi(g \otimes f) = gf.$$
    \item[(2)] Both $\theta$ and $\phi$ are epic iff $M_K$ and $N_K$ are similar.
    \item[(3)] If $M_K$ and $N_K$ are similar, then $\operatorname{End}(M_K) \approx \operatorname{End}(N_K)$.
\end{enumerate}
