\text{Let $\mathfrak{R}$ be an arbitrary ring with an identity element $1$,}
\text{and suppose $\mathfrak{R}$ contains a division subring $\Delta$ that contains $1$.}
\text{For $\rho \in \Delta$ and $x \in \mathfrak{R}$, take the product $\rho x$ to be}
\text{the ring product in $\mathfrak{R}$. Then:}

\begin{itemize}
\item \text{S1--S3 follow from the distributive and associative laws of the ring.}
\item \text{S4 holds because the identity element of $\Delta$ is the identity of $\mathfrak{R}$.}
\end{itemize}

\text{Hence the additive group $\mathfrak{R}$, the division ring $\Delta$,}
\text{and the multiplication $\rho x$ constitute a vector space over $\Delta$.}
\text{This space may or may not be finite dimensional.}

\textbf{Examples:}
\begin{enumerate}
\item \text{If $\mathfrak{R} = \mathbb{C}$ (complex numbers) and $\Delta = \mathbb{R}$ (real numbers),}
\text{then $\mathfrak{R}$ is finite dimensional over $\Delta$ with basis $\{1, \sqrt{-1}\}$.}
\item \text{If $\mathfrak{R} = \Delta[\lambda]$, the polynomial ring in an indeterminate $\lambda$}
\text{over $\Delta$, then $\mathfrak{R}$ is not finite dimensional over $\Delta$.}
\end{enumerate}
