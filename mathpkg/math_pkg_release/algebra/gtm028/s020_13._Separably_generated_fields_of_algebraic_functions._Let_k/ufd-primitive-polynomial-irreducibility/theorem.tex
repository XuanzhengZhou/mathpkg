Let $R$ be a unique factorization domain and let $K$ be the quotient field of $R$. Let $X$ be an indeterminate over $K$, and let $f(X) \in R[X]$ be a primitive polynomial.

\begin{enumerate}
\item[(1)] If $f(X)$ is irreducible in $R[X]$, then $f(X)$ is irreducible in $K[X]$.
\item[(2)] If $f(X)$ is irreducible in $K[X]$, then $f(X)$ is irreducible in $R[X]$.
\item[(3)] If $g(X) \in R[X]$ is a primitive polynomial and $g(X)$ divides $f(X)$ in $K[X]$, then $g(X)$ divides $f(X)$ in $R[X]$.
\end{enumerate}
