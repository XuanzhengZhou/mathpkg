Let $f : R \to S$ be a homomorphism of commutative rings. For ideals $\mathfrak{a}, \mathfrak{b}$ of $R$ and $\mathfrak{A}, \mathfrak{B}$ of $S$:

\begin{enumerate}
\item[(1)] $\mathfrak{A} \subset \mathfrak{B} \implies \mathfrak{A}^c \subset \mathfrak{B}^c$ and $\mathfrak{a} \subset \mathfrak{b} \implies \mathfrak{a}^e \subset \mathfrak{b}^e$.
\item[(2)] $\mathfrak{a}^{ec} \supset \mathfrak{a}$ and $\mathfrak{A}^{ce} \subset \mathfrak{A}$.
\item[(3)] $\mathfrak{a}^{ece} = \mathfrak{a}^e$ and $\mathfrak{A}^{cec} = \mathfrak{A}^c$.
\item[(4)] $(\mathfrak{a} + \mathfrak{b})^e = \mathfrak{a}^e + \mathfrak{b}^e$ and $(\mathfrak{A} + \mathfrak{B})^c \supset \mathfrak{A}^c + \mathfrak{B}^c$.
\item[(5)] $(\mathfrak{a} \cap \mathfrak{b})^e \subset \mathfrak{a}^e \cap \mathfrak{b}^e$ and $(\mathfrak{A} \cap \mathfrak{B})^c = \mathfrak{A}^c \cap \mathfrak{B}^c$.
\item[(6)] $(\mathfrak{a}\mathfrak{b})^e = \mathfrak{a}^e \mathfrak{b}^e$ and $(\mathfrak{A}\mathfrak{B})^c \supset \mathfrak{A}^c \mathfrak{B}^c$.
\item[(7)] $(\mathfrak{a} : \mathfrak{b})^e \subset \mathfrak{a}^e : \mathfrak{b}^e$ and $(\mathfrak{A} : \mathfrak{B})^c \subset \mathfrak{A}^c : \mathfrak{B}^c$.
\item[(8)] $(\sqrt{\mathfrak{a}})^e \subset \sqrt{\mathfrak{a}^e}$ and $(\sqrt{\mathfrak{A}})^c = \sqrt{\mathfrak{A}^c}$.
\end{enumerate}
