Let $A$ be a ring, let $B$ be an overring of $A$, and let $x \in B$. The following conditions are equivalent:

\begin{itemize}
\item[(c)] $x$ is integral over $A$, i.e., there exist $a_0, \dots, a_{n-1} \in A$ such that
  $$x^n + a_{n-1}x^{n-1} + \cdots + a_0 = 0.$$

\item[(c')] The ring $A[x]$ is a finite $A$-module.

\item[(c'')] The ring $A[x]$ is contained in a subring $R$ of $B$ which is a finite $A$-module.

\item[(c''')] There exists in $B$ a finite $A$-module $M$ such that
  \begin{enumerate}
    \item $xM \subset M$,
    \item zero is the only element $y$ of $A[x]$ with $yz = 0$ for all $z \in M$.
  \end{enumerate}
\end{itemize}
