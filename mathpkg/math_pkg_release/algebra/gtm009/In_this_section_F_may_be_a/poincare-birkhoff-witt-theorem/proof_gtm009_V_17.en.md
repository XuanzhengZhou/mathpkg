Fix an ordered basis $(x_{\lambda}; \lambda \in \Omega)$ of $L$. This choice identifies $\mathfrak{S}$ with the polynomial algebra in indeterminates $z_{\lambda}$ ($\lambda \in \Omega$). For each sequence $\Sigma = (\lambda_1, \ldots, \lambda_m)$ of indices, let $z_{\Sigma} = z_{\lambda_1} \cdots z_{\lambda_m} \in S^m$ and $x_{\Sigma} = x_{\lambda_1} \otimes \cdots \otimes x_{\lambda_m} \in T^m$. Call $\Sigma$ \textit{increasing} if $\lambda_1 \leq \lambda_2 \leq \cdots \leq \lambda_m$ in the given ordering of $\Omega$; by fiat, $\varnothing$ is increasing and $z_{\varnothing} = 1$. The set $\{z_{\Sigma} \mid \Sigma \text{ increasing}\}$ is a basis of $\mathfrak{S}$.

Associated with the grading $\mathfrak{S} = \bigoplus S^m$ is the filtration $S_m = S^0 \oplus \cdots \oplus S^m$. Write $\lambda \leq \Sigma$ if $\lambda \leq \mu$ for all $\mu \in \Sigma$.

\textbf{Lemma A.} For each $m \in \mathbb{Z}^+$, there exists a unique linear map $f_m: L \otimes S_m \to \mathfrak{S}$ satisfying:
\begin{align*}
(A_m) &\quad f_m(x_{\lambda} \otimes z_{\Sigma}) = z_{\lambda} z_{\Sigma} \quad \text{for } \lambda \leq \Sigma,\; z_{\Sigma} \in S_m;\\
(B_m) &\quad f_m(x_{\lambda} \otimes z_{\Sigma}) - z_{\lambda} z_{\Sigma} \in S_k \quad \text{for } k \leq m,\; z_{\Sigma} \in S_k;\\
(C_m) &\quad f_m(x_{\lambda} \otimes f_m(x_{\mu} \otimes z_T)) = f_m(x_{\mu} \otimes f_m(x_{\lambda} \otimes z_T)) + f_m([x_{\lambda} x_{\mu}] \otimes z_T) \quad \text{for } z_T \in S_{m-1}.
\end{align*}

The proof of Lemma A proceeds by induction on $m$, defining $f_m$ on elements $x_{\lambda} \otimes z_{\Sigma}$ with $\Sigma$ increasing. When $\lambda \leq \Sigma$, $(A_m)$ forces the definition. Otherwise, suppose $\lambda \not\leq \Sigma$. Let $\Sigma = (\mu, \Psi)$ where $\mu$ is the first index in $\Sigma$ and $\mu < \lambda$. Then set
$$f_m(x_{\lambda} \otimes z_{\Sigma}) = f_m(x_{\lambda} \otimes f_{m-1}(x_{\mu} \otimes z_{\Psi}))$$
and use $(C_m)$ to expand:
\begin{align*}
f_m(x_{\lambda} \otimes z_{\Sigma}) &= f_{m-1}(x_{\mu} \otimes f_{m-1}(x_{\lambda} \otimes z_{\Psi})) + f_{m-1}([x_{\lambda} x_{\mu}] \otimes z_{\Psi})\\
&\quad + \text{(terms already determined by induction)}.
\end{align*}
By induction on the well-ordering of indices, this defines $f_m$ uniquely.

\textbf{Lemma B.} The maps $f_m$ define an $L$-module structure on $\mathfrak{S}$ via $x \cdot z = f_m(x \otimes z)$ for $x \in L$, $z \in S_m$. Moreover, this action satisfies $x \cdot (y \cdot z) - y \cdot (x \cdot z) = [xy] \cdot z$ for all $x, y \in L$, $z \in \mathfrak{S}$.

This follows from $(C_m)$ and the definition of the maps $f_m$.

\textbf{Lemma C.} Let $\phi: \mathfrak{T}(L) \to \operatorname{End}_F(\mathfrak{S})$ be the algebra homomorphism extending the $L$-module structure (sending $1$ to $\operatorname{id}$). Then $\phi(J) = 0$, where $J$ is the ideal generated by $x \otimes y - y \otimes x - [xy]$. Hence $\phi$ factors through $\mathfrak{U}(L)$, making $\mathfrak{S}$ into a $\mathfrak{U}(L)$-module.

By Lemma B, for any generator $x \otimes y - y \otimes x - [xy]$ of $J$, we have $\phi(x \otimes y - y \otimes x - [xy]) = x \cdot (y \cdot -) - y \cdot (x \cdot -) - [xy] \cdot - = 0$, so $J \subset \ker \phi$.

Now observe that $\phi(\pi(T^m))(1) \subset S_m$ and that $\phi$ induces an $F$-linear map $\psi: \mathfrak{U}(L) \to \mathfrak{S}$ via $\psi(u) = \phi(u)(1)$. The composition $\mathfrak{S} \stackrel{\omega}{\longrightarrow} \mathfrak{G} \stackrel{\psi_*}{\longrightarrow} \mathfrak{S}$ is the identity (one checks on $S^m$ by induction), so $\omega$ is injective. Since $\omega$ is surjective by construction, it is an isomorphism.
