Let $\Phi$ be the set of roots of a semisimple Lie algebra $L$ relative to a maximal toral subalgebra $H$.
\begin{enumerate}
\item[(a)] $\Phi$ spans $H^*$, and if $\alpha \in \Phi$, the only multiples of $\alpha$ in $\Phi$ are $\pm \alpha$.
\item[(b)] For each $\alpha \in \Phi$, $\dim L_{\alpha} = 1$. In particular, $S_{\alpha} = L_{\alpha} + L_{-\alpha} + H_{\alpha}$ where $H_{\alpha} = [L_{\alpha} L_{-\alpha}]$, and for given nonzero $x_{\alpha} \in L_{\alpha}$, there exists a unique $y_{\alpha} \in L_{-\alpha}$ such that $[x_{\alpha} y_{\alpha}] = h_{\alpha}$.
\item[(c)] For $\alpha, \beta \in \Phi$, $\beta(h_{\alpha}) \in \mathbb{Z}$, and $\beta - \beta(h_{\alpha}) \alpha \in \Phi$ (whenever $\beta \neq \pm \alpha$).
\item[(d)] If $\alpha, \beta, \alpha+\beta \in \Phi$, then $[L_{\alpha} L_{\beta}] = L_{\alpha+\beta}$.
\item[(e)] Let $\alpha, \beta \in \Phi$, $\beta \neq \pm \alpha$. Let $r, q \in \mathbb{Z}^+$ be the largest integers for which $\beta - r\alpha \in \Phi$, $\beta + q\alpha \in \Phi$. Then all $\beta + i\alpha \in \Phi$ for $-r \leq i \leq q$, and $\beta(h_{\alpha}) = r - q$.
\item[(f)] The Cartan integers $2(\alpha, \beta)/(\beta, \beta) = \alpha(h_{\beta})$ are integers.
\end{enumerate}
