\text{Let} $\mathbf{T}$ \text{be an algebraic theory in} $\mathcal{L}$, \text{let} $\mathbf{S}$ \text{be an algebraic theory in} $\mathcal{K}$, \text{and let} $H: \mathcal{K} \longrightarrow \mathcal{L}$ \text{be a functor.}

\text{A functor} $\Gamma: \mathcal{K}^{\mathbf{S}} \longrightarrow \mathcal{L}^{\mathbf{T}}$ \text{is said to be} \emph{over} $H$ \text{if}
$$
\Gamma \circ U^{\mathbf{T}} = U^{\mathbf{S}} \circ H,
$$
\text{where} $U^{\mathbf{S}}: \mathcal{K}^{\mathbf{S}} \to \mathcal{K}$ \text{and} $U^{\mathbf{T}}: \mathcal{L}^{\mathbf{T}} \to \mathcal{L}$ \text{are the forgetful functors from the categories of algebras.}

\text{This generalizes the case} $H = \mathrm{id}_{\mathcal{K}}$, \text{in which a functor over the identity is an endofunctor on} $\mathcal{K}^{\mathbf{S}}$ \text{commuting with the forgetful functor to} $\mathcal{K}$.
