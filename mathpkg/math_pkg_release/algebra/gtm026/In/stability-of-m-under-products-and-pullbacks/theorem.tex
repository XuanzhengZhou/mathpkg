Let $(E, M)$ be an image factorization system in a category $\mathcal{K}$.

\begin{enumerate}
\item[(i)] If $\{m_i: C_i \longrightarrow D_i\}_{i \in I}$ is a family of morphisms in $M$ and the products $\prod C_i$ and $\prod D_i$ exist in $\mathcal{K}$, then the induced morphism
$$\prod m_i: \prod C_i \longrightarrow \prod D_i$$
(defined by $(\prod m_i) \cdot p_j = p_j \cdot m_j$ for all projection maps $p_j$) is also in $M$.

\item[(ii)] Given a pullback square
$$\xymatrix{
\cdot \ar[r] \ar[d] & \cdot \ar[d]^{m} \\
\cdot \ar[r] & \cdot
}$$
if $m \in M$, then the pullback of $m$ along any morphism is also in $M$.
\end{enumerate}
