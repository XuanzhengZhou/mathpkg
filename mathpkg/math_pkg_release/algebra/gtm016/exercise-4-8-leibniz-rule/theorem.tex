For $D \in \operatorname{Der}_k K$ and $x, y \in K$:
\begin{enumerate}
  \item[(a)] $D^n(xy) = \sum \binom{n}{m} D^m(x)D^{n-m}(y)$;
  \item[(b)] $(D - (a+b)I)^n(xy) = \sum \binom{n}{m} (D - aI)^m(x)(D - bI)^{n-m}(y)$ for $a, b \in k$;
  \item[(c)] $D^p \in \operatorname{Der}_k K$ for $D \in \operatorname{Der}_k K$;
  \item[(d)] letting $K_a(D) = \{x \in K \mid (D - aI)^m(x) = 0 \text{ for some } m\}$, show that $K_a(D)$ is a subring of $K$.
\end{enumerate}
