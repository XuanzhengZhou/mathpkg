$\Delta$ and $\varepsilon$ are $K$-linear mappings such that:

\begin{enumerate}
\item $\sum_i \Delta(x_{(1)i}) \otimes x_{(2)i} \otimes x_{(3)i} = \sum_i x_{(1)i} \otimes \Delta(x_{(2)i}) \otimes x_{(3)i}$ for $x \in H(K/k)$ \quad ($\Delta$ is coassociative);
\item $x = \sum_i \varepsilon(x_{(1)i}) x_{(2)i} = \sum_i \varepsilon(x_{(2)i}) x_{(1)i}$ for $x \in H(K/k)$ \quad ($\varepsilon$ is a coidentity);
\item $\Delta(I) = I \otimes I$ and $\Delta(xy) = \sum_{i,j} x_j y \otimes x_i y_j$ for $x, y \in H(K/k)$ \quad ($\Delta$ preserves products);
\item $\varepsilon(I) = I$ and $\varepsilon(xy) = \varepsilon(x)\varepsilon(y)$ for $x, y \in H(K/k)$ \quad ($\varepsilon$ preserves products).
\end{enumerate}
