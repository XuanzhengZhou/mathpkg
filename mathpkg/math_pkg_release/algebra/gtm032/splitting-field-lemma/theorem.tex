\begin{enumerate}
\item[(1)] If $P/\Phi$ is a splitting field of $f(x) \in \Phi[x]$ and $\Sigma/\Phi$ is a subfield of $P/\Phi$, then $P/\Sigma$ is a splitting field of $f(x)$.
\item[(2)] If $P/\Sigma$ is a splitting field for $f(x) \in \Phi[x]$ and $\Sigma = \Phi(\sigma_1, \ldots, \sigma_k)$ where the $\sigma_i$ are roots of $f(x)$ in $P$, then $P/\Phi$ is a splitting field for $f(x)$.
\end{enumerate}
