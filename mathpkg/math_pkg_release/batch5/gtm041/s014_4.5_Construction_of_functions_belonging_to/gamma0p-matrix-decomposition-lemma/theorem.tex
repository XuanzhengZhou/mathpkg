\begin{lemma}
\label{lem:gamma0p-coset-decomposition}
Let $V \in \Gamma_0(p)$ and let $0 \leq \lambda \leq p-1$. Define the translation operator
$$
T_\lambda \tau = \frac{\tau + \lambda}{p}.
$$
Then there exists an integer $\mu$ with $0 \leq \mu \leq p-1$ and a transformation $W_\mu \in \Gamma_0(p^2)$ such that
$$
T_\lambda V = W_\mu T_\mu.
$$
Moreover, as $\lambda$ runs through a complete residue system modulo $p$, so does $\mu$.
\end{lemma}
