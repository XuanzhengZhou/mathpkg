\begin{theorem}
\label{thm:4.5}
Let $f$ be automorphic under the full modular group $\Gamma$, and let $p$ be a prime. Define
$$
f_p(\tau) = \frac{1}{p} \sum_{\lambda=0}^{p-1} f\!\left(\frac{\tau + \lambda}{p}\right).
$$
Then $f_p$ is automorphic under the congruence subgroup $\Gamma_0(p)$.

Moreover, if $f$ has the Fourier expansion
$$
f(\tau) = \sum_{n=-m}^{\infty} a(n) e^{2\pi i n \tau},
$$
then $f_p$ has the Fourier expansion
$$
f_p(\tau) = \sum_{n=-[m/p]}^{\infty} a(np) e^{2\pi i n \tau}.
$$
\end{theorem}
