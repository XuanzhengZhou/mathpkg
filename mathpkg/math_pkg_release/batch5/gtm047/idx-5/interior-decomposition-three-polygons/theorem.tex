\textbf{Theorem 8.} Let $B_1, B_2, B_3$ be as in Theorem 7, with $\operatorname{Int} B_2$ in the interior $I_{13}$ of $B_1 \cup B_3$. Then
\begin{enumerate}
\item[(1)] The components of $I_{13} - \operatorname{Int} B_2$ are the interiors $I_{12}$ and $I_{23}$ of $B_1 \cup B_2$ and $B_2 \cup B_3$.
\item[(2)] $\overline{I}_{13} = \overline{I}_{12} \cup \overline{I}_{23}$.
\item[(3)] $\overline{I}_{13} - B_2 = I_{12} \cup I_{23}$.
\end{enumerate}
