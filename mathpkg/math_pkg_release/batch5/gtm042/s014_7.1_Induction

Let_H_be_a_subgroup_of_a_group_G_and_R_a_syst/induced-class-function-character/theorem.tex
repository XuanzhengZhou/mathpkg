\textbf{Proposition 20.}
\begin{enumerate}
  \item[(i)] The function $\operatorname{Ind}(f)$ is a class function on $G$.
  \item[(ii)] If $f$ is the character of a representation $W$ of $H$, then $\operatorname{Ind}(f)$ is the character of the induced representation $\operatorname{Ind}(W)$ of $G$.
\end{enumerate}
