Suppose $x, z \in M$ and $\delta(x, z) \neq 0$. With $O_z^3$ the group of spatial rotations about $z$, denote by $S$ the set
$$
S = \{ y \mid y = \psi x \text{ for some } \psi \in O_z^3 \}.
$$
\begin{enumerate}
  \item[(a)] $S$ is a spacelike $2$-submanifold, diffeomorphic to $S^2$, given explicitly by:
  $$
  y \in S \iff u^4 y = u^4 x \text{ and } \delta(x, z) = \delta(y, z).
  $$
  \item[(b)] The induced metric on $S$ is that of a sphere of radius $(u^4 x)^{2/3} \delta(x, z)$ in $\mathbb{R}^3$.
\end{enumerate}
(Hint: regard $S$ as a submanifold of the level surface $u^4 = u^4 x$.)
