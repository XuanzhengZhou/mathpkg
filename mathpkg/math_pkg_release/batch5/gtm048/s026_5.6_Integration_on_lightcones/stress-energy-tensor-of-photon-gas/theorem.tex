\textbf{Proposition 5.7.2.} Notation as in Sections 3.0.3 and 5.6. There exists a unique symmetric $(2, 0)$-tensor field $\hat{T}$ on $M$ such that
$$\hat{T}(\omega, \omega) = \int_{\mathcal{L}_x^+} \tilde{\omega}^2 F \Lambda_x \quad \forall \omega \in M_x^*,\; \forall x \in M.$$
Moreover, the following properties hold:
\begin{enumerate}
\item[(a)] $\hat{T}$ is a stress-energy tensor;
\item[(b)] $\hat{T}_x$ is normal unless $F \equiv 0$ on $\mathcal{L}_x^+$;
\item[(c)] $\operatorname{trace} \hat{T} = 0$; and
\item[(d)] $\forall$ instantaneous observer $(x, X)$, $T(X, X) = \int_{P_X} e(F \circ \pi_X) \pi_X = u$.
\end{enumerate}
