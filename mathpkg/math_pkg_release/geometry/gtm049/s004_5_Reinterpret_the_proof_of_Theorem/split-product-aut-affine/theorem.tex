Let $V$ be a vector space of dimension $\geq 2$ over a field $F$.

\begin{enumerate}
\item[(i)] $\operatorname{Aut} \mathcal{A}(V)$ is the split product of $(S, T)$ where $S$ is isomorphic to $S(V)$ by means of the isomorphism $f \mapsto \mathcal{A}(f)$ and $T = \operatorname{Tr} \mathcal{A}(V)$ is isomorphic to the additive group $V$, the isomorphism being $v \mapsto \mathcal{A}(t_v)$.

\item[(ii)] $\operatorname{Af} \mathcal{A}(V)$ is the split product of $(L, T)$ where $T$ is as before and $L$ is isomorphic to $\operatorname{Aut} V$, by means of the isomorphism $f \mapsto \mathcal{A}(f)$, restricted to $\operatorname{Aut} V$.
\end{enumerate}
