Let $c: I \rightarrow \mathbb{R}^n$ be a curve and $B: \mathbb{R}^n \rightarrow \mathbb{R}^n$ an isometry of $\mathbb{R}^n$. Let $\tilde{c} = B \circ c$. Then:
\begin{enumerate}
\item[(i)] The Frenet frames of $c$ and $\tilde{c}$ satisfy $\tilde{e}_i(t) = B(e_i(t))$, and the connection forms are invariant: $\tilde{\omega}_{ij}(t) = \omega_{ij}(t)$.
\item[(ii)] Under a parameter transformation $\phi: \tilde{I} \rightarrow I$, the connection forms transform by the chain rule: $\tilde{\omega}_{ij}(\tilde{t}) = \omega_{ij}(\phi(\tilde{t})) \cdot \phi'(\tilde{t})$.
\end{enumerate}
