\begin{enumerate}
\item[i)] The quadratic form
$$-dn_u \cdot df_u : T_u \mathbb{R}^2 \times T_u \mathbb{R}^2 \rightarrow \mathbb{R}$$
is called the \text{second fundamental form} of $f$ at $u$, and is denoted by $II$ (or $II_u$).

\item[ii)] The linear mapping $L_u := -dn_u \circ df_u^{-1} : T_u f \rightarrow T_u f$ is called the \text{Weingarten map}.
\end{enumerate}

\text{Remarks.}
\begin{enumerate}
\item[1.] $II_u : T_u f \times T_u f \rightarrow \mathbb{R}$ can be written as $L_u X \cdot Y$, i.e.,
$$II_u(X, Y) = L_u X \cdot Y \quad \text{for all } X, Y \in T_u f.$$

\item[2.] The matrix representation of $II_u$ with respect to the canonical basis $\{e_i\}$ of $T_u \mathbb{R}^2$ and the associated basis $\{f_{u^i}\}$ of $T_u f$ is
$$(h_{ik}) := (-n_{u^i} \cdot f_{u^k}) = (n \cdot f_{u^i u^k}).$$
In Gauss' notation:
$$\begin{pmatrix} L & M \\ M & N \end{pmatrix} := \begin{pmatrix} h_{11} & h_{12} \\ h_{21} & h_{22} \end{pmatrix}.$$
\end{enumerate}
