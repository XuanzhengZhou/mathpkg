Let $T = \operatorname{Spec} A$, where $A$ is a noetherian domain. Let $X \subseteq \mathbf{P}_T^n$ be a closed subscheme, flat over $T$, and let $\mathcal{F}$ be a coherent sheaf on $X$, flat over $T$. For each $t \in T$, let $\mathcal{F}_t$ be the induced sheaf on the fibre $X_t \subseteq \mathbf{P}_{k(t)}^n$. Then the following conditions are equivalent:

\begin{enumerate}
  \item[(i)] For each $m \gg 0$, $H^0(X, \mathcal{F}(m))$ is a free $A$-module of finite rank.
  \item[(ii)] The sheaf $\mathcal{F}$ is flat over $T$, and for each $m \gg 0$, the function
  $$t \mapsto \dim_{k(t)} H^0(X_t, \mathcal{F}_t(m))$$
  is constant on $T$.
  \item[(iii)] The Hilbert polynomial $P_t$ of $\mathcal{F}_t$ on $X_t = \mathbf{P}_{k(t)}^n$ is independent of $t$, for any $t \in T$.
\end{enumerate}
