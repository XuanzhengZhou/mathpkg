Let $Y$ be a noetherian scheme and let $\mathcal{E}$ be a locally free $\mathcal{O}_Y$-module of rank $n + 1$, $n \geqslant 1$. Let $X = \mathbf{P}(\mathcal{E})$ with the invertible sheaf $\mathcal{O}_X(1)$ and the projection morphism $\pi: X \to Y$.
\begin{enumerate}
\item[(a)] $\pi_*(\mathcal{O}(l)) \cong S^l(\mathcal{E})$ for $l \geqslant 0$, $\pi_*(\mathcal{O}(l)) = 0$ for $l < 0$; $R^i \pi_*(\mathcal{O}(l)) = 0$ for $0 < i < n$ and all $l \in \mathbf{Z}$; and $R^n \pi_*(\mathcal{O}(l)) = 0$ for $l > -n - 1$.
\item[(b)] There is a natural exact sequence
$$0 \to \Omega_{X/Y} \to (\pi^*\mathcal{E})(-1) \to \mathcal{O} \to 0,$$
and the relative canonical sheaf $\omega_{X/Y} = \wedge^n \Omega_{X/Y}$ is isomorphic to $(\pi^* \wedge^{n+1} \mathcal{E})(-n - 1)$.
\end{enumerate}
