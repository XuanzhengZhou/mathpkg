Let $X = \mathbf{P}_k^n$ over a field $k$, and let $\omega_X = \bigwedge^n \Omega_{X/k}$ be the canonical sheaf. Then:
\begin{enumerate}
\item[(a)] $H^n(X, \omega_X) \cong k$. Fix one such isomorphism.
\item[(b)] For any coherent sheaf $\mathcal{F}$ on $X$, the natural pairing
$$\operatorname{Hom}(\mathcal{F}, \omega) \times H^n(X, \mathcal{F}) \to H^n(X, \omega) \cong k$$
is a perfect pairing of finite-dimensional vector spaces over $k$.
\item[(c)] For every $i \geq 0$ there is a natural functorial isomorphism
$$\operatorname{Ext}^i(\mathcal{F}, \omega) \xrightarrow{\sim} H^{n-i}(X, \mathcal{F})^\vee$$
where $^\vee$ denotes the dual vector space, which for $i = 0$ is the one induced by the pairing of (b).
\end{enumerate}
