\textbf{Lemma 7.4.} Let $\mathcal{P}$ be a translation plane and let $\Pi = \operatorname{Aut} \mathcal{P}$. Then
\begin{enumerate}
\item[(i)] $\Pi_{[x_1, x_2]} = \bigcup_{X \in [\infty]} \Pi_{[X, x_2]}$.
\item[(ii)] $\Pi_{[A, x_2]} \cap \Pi_{[B, x_2]} = 1$ for all $A, B$ on $[\infty]$, $A \neq B$.
\item[(iii)] $\Pi_{[x_1, x_2]} = \Pi_{[A, x_2]} \cap \Pi_{[B, x_2]}$ for any pair $A, B$ on $[\infty]$, $A \neq B$.
\end{enumerate}
