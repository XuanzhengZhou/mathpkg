\text{An écart for a set } X \text{ is a non-negative real-valued function } e \text{ on } X \times X \text{ such that for all } x, y, z \in X \text{:}
\end{definition}

\begin{enumerate}
\item e(x, x) = 0
\item e(x, y) = e(y, x) \quad \text{(symmetry)}
\item e(x, z) \leq e(x, y) + e(y, z) \quad \text{(triangle inequality)}
\end{enumerate}

\text{An écart differs from a metric only in that } e(x,y) = 0 \text{ need not imply } x = y \text{. That is, an écart is a pseudometric.}
