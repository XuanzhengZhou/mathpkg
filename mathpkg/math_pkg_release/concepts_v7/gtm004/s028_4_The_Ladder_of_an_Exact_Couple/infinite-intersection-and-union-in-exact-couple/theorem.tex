\begin{definition}
Let $(D, E; \alpha, \beta, \gamma)$ be an exact couple in an abelian category with sufficiently well-behaved limits and colimits. Define
\[
\alpha^{\infty} D = \bigcap_{n \geq 0} \alpha^{n} D,
\qquad
\alpha^{-\infty}(0) = \bigcup_{m \geq 0} \alpha^{-m}(0),
\tag{4.24}
\]
where $\alpha^n D$ denotes the image of the $n$-fold iterate of $\alpha$, and $\alpha^{-m}(0)$ denotes the $m$-fold preimage of zero under $\alpha$, i.e.,
\[
\alpha^{-m}(0) = \ker(\alpha^m : D \to D).
\]
\end{definition}
