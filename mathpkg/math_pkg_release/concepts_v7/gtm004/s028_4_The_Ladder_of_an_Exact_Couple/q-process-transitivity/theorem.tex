\begin{proposition}[Proposition 4.3]
Consider the diagram
\[
\begin{CD}
{} &  & C_2 \\
{} &  & @VV{\varrho_1}V \\
{} &  & C_1 \\
{} &  & @VV{\varrho}V \\
A @>{\beta}>> B @>{\gamma}>> C
\end{CD}
\]
where $\gamma\beta$ factors through $\varrho\varrho_1$, i.e., $\gamma\beta = \varrho\varrho_1\delta$ for some $\delta : A \to C_2$. Let $(\beta', \gamma') = Q^{\varrho}(\beta, \gamma)$. Then $\gamma'\beta'$ factors through $\varrho_1$, and
\[
Q^{\varrho\varrho_1} = Q^{\varrho_1} Q^{\varrho}.
\tag{4.10}
\]
Dually,
\[
Q_{\sigma\sigma_1} = Q_{\sigma} Q_{\sigma_1}.
\tag{4.11}
\]
\end{proposition}
