\text{Let } E \text{ and } F \text{ be TVS over } \mathbb{K} \text{ and write } \mathcal{L}(E, F) \text{ for the space of continuous linear mappings.}

\begin{enumerate}
\item[(i)] \text{For each } x \in E \text{ and each continuous linear form } g \text{ on } F, \text{ the map } T \mapsto g(Tx) \text{ is a linear form on } \mathcal{L}(E, F). \text{ The initial topology } \tau_w \text{ for the family } \{T \mapsto g(Tx) : x \in E,\ g \in F'\} \text{ is called the weak operator topology; it is locally convex.}

\item[(ii)] \text{The strong operator topology } \tau_s \text{ is finer than } \tau_w.

\item[(iii)] \text{If } F \text{ is separated and locally convex, then } \tau_w \text{ is also separated.}

\item[(iv)] \text{If } F \text{ is finite-dimensional and separated, then } \tau_s = \tau_w.

\item[(v)] \text{Write } F' \text{ for the continuous dual of } F.
\end{enumerate}
