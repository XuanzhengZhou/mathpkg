\begin{theorem}
Let $\varphi: L^1(G) \to \mathcal{L}(H)$ be a nondegenerate $*$-representation on a Hilbert space $H$, and let $(u_j)_{j \in J}$ be any approximate identity in $L^1(G)$. Then:
\begin{enumerate}
\item[(1)] For each $t \in G$, the net $\varphi((u_j)_t)$ converges strongly to a unitary operator $U_t$ on $H$, independent of the approximate identity chosen.
\item[(2)] $\varphi(u_t) = U_t \varphi(u)$ for all $u \in L^1(G)$ and $t \in G$.
\item[(3)] $t \mapsto U_t$ is a continuous unitary representation of $G$ on $H$.
\item[(4)] $(\varphi(u)x \mid y) = \int f(t)(U_t x \mid y)\, dt$ for all $u = [f] \in L^1(G)$ and $x, y \in H$.
\end{enumerate}
\end{theorem}
