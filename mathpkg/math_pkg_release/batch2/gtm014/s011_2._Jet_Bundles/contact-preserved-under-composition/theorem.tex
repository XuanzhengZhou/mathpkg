Let $X$, $Y$, $Z$ be smooth manifolds, $p \in X$, and let $f, g: X \to Y$ be smooth maps with $f \sim_k g$ at $p$.

\begin{enumerate}
\item If $h: Y \to Z$ is smooth, then $h \circ f \sim_k h \circ g$ at $p$.
\item If $\phi: W \to X$ is a smooth diffeomorphism, then $f \circ \phi \sim_k g \circ \phi$ at $\phi^{-1}(p)$.
\end{enumerate}
