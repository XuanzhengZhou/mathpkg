Let $X$, $Y$, $Z$, $W$ be smooth manifolds.

\begin{enumerate}
\item Let $h: Y \to Z$ be smooth. Then $h$ induces a mapping $h_*: J^k(X, Y) \to J^k(X, Z)$ defined as follows: if $\sigma \in J^k(X, Y)_{p,q}$ and $f: X \to Y$ represents $\sigma$, then $h_*(\sigma) =$ the equivalence class of $h \circ f$ in $J^k(X, Z)_{p, h(q)}$.

\item Let $g: Z \to X$ be a smooth diffeomorphism. Then $g$ induces a mapping $g^*: J^k(X, Y) \to J^k(Z, Y)$ defined as follows: if $\tau \in J^k(X, Y)_{p,q}$ and $f: X \to Y$ represents $\tau$, then $g^*(\tau) =$ the equivalence class of $f \circ g$ in $J^k(Z, Y)_{g^{-1}(p), q}$.

\item Let $a: Z \to W$ be smooth. Then $a_* \circ h_* = (a \circ h)_*$ as mappings $J^k(X, Y) \to J^k(X, W)$ and $(\mathrm{id}_Y)_* = \mathrm{id}_{J^k(X, Y)}$. Thus if $h$ is a diffeomorphism, $h_*$ is a bijection.

\item Let $a: W \to Z$ be a smooth diffeomorphism. Then $a^* \circ g^* = (g \circ a)^*$ as mappings $J^k(X, Y) \to J^k(W, Y)$ and $(\mathrm{id}_X)^* = \mathrm{id}_{J^k(X, Y)}$, so that $g^*$ is a bijection.
\end{enumerate}
