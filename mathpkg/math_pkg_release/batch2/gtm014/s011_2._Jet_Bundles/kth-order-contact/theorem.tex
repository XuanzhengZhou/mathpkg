Let $X$ and $Y$ be smooth manifolds, and $p \in X$. Suppose $f, g: X \to Y$ are smooth maps with $f(p) = g(p) = q$.

\begin{enumerate}
\item $f$ has \textit{first order contact} with $g$ at $p$ if $(df)_p = (dg)_p$ as mappings of $T_p X \to T_q Y$.
\item $f$ has \textit{$k$-th order contact} with $g$ at $p$ if $(df)_p: TX \to TY$ has $(k-1)$-st order contact with $(dg)_p$ at every point in $T_p X$. This is written as $f \sim_k g$ at $p$, where $k$ is a positive integer.
\end{enumerate}
