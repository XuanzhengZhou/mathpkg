% Proposition 1.10: Open Neighborhood of Infinitesimally Stable Mappings
% From GTM014, Chapter V, §1
%
% STATEMENT ONLY — no proof text.

\begin{proposition}
Let $f: X \to Y$ be infinitesimally stable.
Then there exists a neighborhood $W$ of $f$ in $C^\infty(X,Y)$ such that every $g \in W$ is locally infinitesimally stable.
\end{proposition}
