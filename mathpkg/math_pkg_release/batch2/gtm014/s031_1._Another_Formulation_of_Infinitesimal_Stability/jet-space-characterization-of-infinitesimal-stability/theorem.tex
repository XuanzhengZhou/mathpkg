% Corollary 1.3: Jet Space Characterization of Local Infinitesimal Stability
% From GTM014, Chapter V, §1
%
% STATEMENT ONLY — no proof text.

\begin{corollary}
Let $f: X \to Y$ be smooth, let $p \in X$, let $q = f(p) \in Y$, and let $m = \dim Y$.
Then $f$ is locally infinitesimally stable at $p$ if and only if
\[
J^m(f^*TY)_p = (df)_p\, J^m(TX)_p + f^*\, J^m(TY)_q,
\]
where $(df)_p$ and $f^*$ are the maps induced on $m$-jets by the action of $df$ and $f^*$ on vector fields.
\end{corollary}
