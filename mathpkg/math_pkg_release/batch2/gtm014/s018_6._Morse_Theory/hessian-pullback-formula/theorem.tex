Let $\phi: X \to Y$ be a smooth map between smooth manifolds and $f: Y \to \mathbb{R}$ be smooth. Suppose that $q \in Y$ is a critical point for $f$ and $p \in \phi^{-1}(q)$. Then:
\begin{enumerate}
  \item $p$ is a critical point for the pullback $\phi^* f = f \circ \phi$;
  \item $(d^2 (\phi^* f))_p = (\phi^* (d^2 f))_p$, where
  $$(\phi^* (d^2 f))_p(v, w) = (d^2 f)_q((d\phi)_p v, (d\phi)_p w) \quad \forall v, w \in T_p X.$$
\end{enumerate}
