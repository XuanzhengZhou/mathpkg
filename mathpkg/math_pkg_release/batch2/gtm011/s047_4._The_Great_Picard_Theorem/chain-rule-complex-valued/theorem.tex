Let $f: [a, b] \to \mathbb{C}$ be a differentiable function.

\begin{enumerate}
\item[(a)] If $g: [c, d] \to [a, b]$ is differentiable, then $f \circ g$ is differentiable and
$$(f \circ g)'(t) = f'(g(t)) \cdot g'(t).$$

\item[(b)] If $G$ is an open subset of $\mathbb{C}$ such that $f([a, b]) \subset G$ and $h: G \to \mathbb{C}$ is analytic, then $h \circ f$ is differentiable and
$$(h \circ f)'(x) = h'(f(x)) \cdot f'(x).$$
\end{enumerate}
