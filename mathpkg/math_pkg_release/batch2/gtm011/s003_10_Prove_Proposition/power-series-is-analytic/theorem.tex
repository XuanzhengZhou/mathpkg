Let $f(z) = \sum_{n=0}^{\infty} a_n (z - a)^n$ have radius of convergence $R > 0$. Then:

\begin{enumerate}
    \item[(a)] For each $k \geq 1$ the series
    $$\sum_{n=k}^{\infty} n(n-1)\ldots(n-k+1) a_n (z-a)^{n-k}$$
    has radius of convergence $R$;
    \item[(b)] $f$ is infinitely differentiable on $B(a; R)$ and $f^{(k)}(z)$ is given by the series in (a);
    \item[(c)] $a_n = f^{(n)}(a)/n!$ for $n \geq 0$.
\end{enumerate}
