Let ${}_R M$ be a left $R$-module. Let $(K_\alpha)_{\alpha \in A}$ and $(I_\alpha)_{\alpha \in A}$ be families of subgroups of the additive groups of $M$ and $R$, respectively. Then:

\begin{enumerate}
\item $l_R\!\left(\sum_{\alpha \in A} K_\alpha\right) = \bigcap_{\alpha \in A} l_R(K_\alpha)$ and $r_M\!\left(\sum_{\alpha \in A} I_\alpha\right) = \bigcap_{\alpha \in A} r_M(I_\alpha)$;
\item $\sum_{\alpha \in A} l_R(K_\alpha) \subseteq l_R\!\left(\bigcap_{\alpha \in A} K_\alpha\right)$ and $\sum_{\alpha \in A} r_M(I_\alpha) \subseteq r_M\!\left(\bigcap_{\alpha \in A} I_\alpha\right)$.
\end{enumerate}
