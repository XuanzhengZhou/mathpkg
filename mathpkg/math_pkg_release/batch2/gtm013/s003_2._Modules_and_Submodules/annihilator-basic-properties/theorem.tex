Let ${}_R M_S$ be a bimodule, let $X, Y \subseteq M$, and let $A, B \subseteq R$. Then:

\begin{enumerate}
\item $X \subseteq Y$ implies $l_R(Y) \subseteq l_R(X)$ and $A \subseteq B$ implies $r_M(B) \subseteq r_M(A)$;
\item $X \subseteq r_M l_R(X)$ and $A \subseteq l_R r_M(A)$;
\item $l_R(X) = l_R r_M l_R(X)$ and $r_M(A) = r_M l_R r_M(A)$.
\end{enumerate}
