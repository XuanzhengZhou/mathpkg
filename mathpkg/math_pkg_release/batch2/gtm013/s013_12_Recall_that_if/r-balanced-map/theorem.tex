Let $M_R$ and ${}_R N$ be modules. A function $\beta: M \times N \to A$ (where $A$ is an abelian group) is \text{$R$-balanced} in case for all $m, m_1, m_2 \in M$, $n, n_1, n_2 \in N$, and $r \in R$:
\begin{enumerate}
\item[(1)] $\beta(m_1 + m_2, n) = \beta(m_1, n) + \beta(m_2, n)$;
\item[(2)] $\beta(m, n_1 + n_2) = \beta(m, n_1) + \beta(m, n_2)$;
\item[(3)] $\beta(mr, n) = \beta(m, rn)$.
\end{enumerate}
