For a left $R$-module $C$, the following are equivalent:
\begin{enumerate}
\item[(a)] $C$ is a cogenerator;
\item[(b)] For every homomorphism $f$ in ${}_R\mathbf{M}$, if $\operatorname{Hom}_R(f, C) = 0$, then $f = 0$;
\item[(c)] The functor $\operatorname{Hom}_R(-, C)$ is faithful;
\item[(d)] For every exact sequence $M \xrightarrow{f} M \xrightarrow{g} M$ in ${}_R\mathbf{M}$, if $\operatorname{Hom}_R(g, C)$ and $\operatorname{Hom}_R(f, C)$ are exact, then the original sequence is exact.
\end{enumerate}
