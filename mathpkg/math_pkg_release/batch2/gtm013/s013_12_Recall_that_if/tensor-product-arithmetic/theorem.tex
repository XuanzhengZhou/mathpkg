Each element of $M \otimes_R N$ can be expressed as a finite sum $\sum_i (m_i \otimes n_i)$ with $m_i \in M$, $n_i \in N$. Moreover, for all $m, m_i \in M$, $n, n_i \in N$, and $r \in R$:
\begin{enumerate}
\item[(1)] $(m_1 + m_2) \otimes n = (m_1 \otimes n) + (m_2 \otimes n)$,
\item[(2)] $m \otimes (n_1 + n_2) = (m \otimes n_1) + (m \otimes n_2)$,
\item[(3)] $mr \otimes n = m \otimes rn$.
\end{enumerate}
