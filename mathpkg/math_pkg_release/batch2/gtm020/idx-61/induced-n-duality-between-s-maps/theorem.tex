\text{Let } u: X \wedge X' \to S^n \text{ and } v: Y \wedge Y' \to S^n \text{ be two } n\text{-duality maps between finite } CW\text{-complexes. The induced } n\text{-duality between } S\text{-maps is the group isomorphism}

$$D_n(u, v) = u_{Y'}^{-1} \circ v^X: \{X, Y\} \to \{Y', X'\}$$

\text{where } v^X: \{X, Y\} \to \{X \wedge Y', S^n\} \text{ and } u_{Y'}: \{Y', X'\} \to \{X \wedge Y', S^n\}.

\text{The following commutative diagram holds:}

$$\begin{array}{ccc}
\{X, Y\} & \xrightarrow{D_n(u,v)} & \{Y', X'\} \\
\downarrow v^X & \searrow & \downarrow u_{Y'} \\
\{X \wedge Y', S^n\} & = & \{X \wedge Y', S^n\}
\end{array}$$

\text{If } w: Z \wedge Z' \to S^n \text{ is a third } n\text{-duality map, then}

$$D_n(u, w)(\{g\} \circ \{f\}) = (D_n(u, v)\{f\}) \circ (D_n(v, w)\{g\})$$

\text{and } D_n(u, u)\{1\} = \{1\}. \text{ Thus the duality maps are functorial.}
