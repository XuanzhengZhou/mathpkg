Let $\xi = (E, p, B)$ be a bundle, let $f: B_1 \to B$ be a map, and let $(f_\xi, f): f^*(\xi) \to \xi$ be the canonical morphism of the induced bundle.

\begin{enumerate}
\item If $s$ is a cross section of $\xi$, then $\sigma: B_1 \to E(f^*(\xi))$ defined by
$$\sigma(b_1) = (b_1, sf(b_1))$$
is a cross section of $f^*(\xi)$ satisfying $f_\xi \sigma = sf$.

\item If $f$ is an identification map and $\sigma$ is a cross section of $f^*(\xi)$ such that $f_\xi \sigma$ is constant on all sets $f^{-1}(b)$ for $b \in B$, then there exists a cross section $s$ of $\xi$ such that $sf = f_\xi \sigma$.
\end{enumerate}
