For each complex vector bundle $\xi$ over a space $B$ there is a total Chern class
$$c(\xi) \in H^*(B, \mathbb{Z})$$
with the following properties:

\begin{itemize}
\item[(C$_0$)] $c(\xi) = 1 + c_1(\xi) + \cdots + c_n(\xi)$, where $c_i(\xi) \in H^{2i}(B, \mathbb{Z})$ and $c_i(\xi) = 0$ for $i > \dim \xi$.
\item[(C$_1$)] If $\xi$ and $\eta$ are $B$-isomorphic, then $c(\xi) = c(\eta)$. If $f: B_1 \to B$ is a map, then $f^*(c(\xi)) = c(f^*(\xi))$.
\item[(C$_2$)] For two vector bundles $\xi$ and $\eta$ over $B$, the Whitney sum formula holds:
$$c(\xi \oplus \eta) = c(\xi) \cup c(\eta).$$
\item[(C$_3$)] For the canonical line bundle $\lambda$ over $S^2 = \mathbb{C}P^1$, the element $c_1(\lambda)$ is the given generator of $H^2(S^2, \mathbb{Z})$.
\item[(C$'_3$)] For the canonical line bundle $\gamma_1$ over $\mathbb{C}P^\infty$, the element $c_1(\gamma_1)$ is the generator $z$ of the polynomial ring $H^*(\mathbb{C}P^\infty, \mathbb{Z}) \cong \mathbb{Z}[z]$.
\end{itemize}
