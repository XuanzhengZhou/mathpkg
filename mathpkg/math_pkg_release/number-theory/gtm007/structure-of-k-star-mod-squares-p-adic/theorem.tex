\textbf{Lemma.}
\begin{enumerate}
\item[(a)] The $\mathbb{F}_2$-vector space $k^*/k^{*2}$ has $2^r$ elements with $r = 2$ if $p \neq 2$ and $r = 3$ if $p = 2$.
\item[(b)] If $a \in k^*/k^{*2}$ and $\varepsilon = \pm 1$, let $H_a^\varepsilon = \{x \in k^*/k^{*2} : (x, a) = \varepsilon\}$. If $a = 1$, then $H_1^1$ has $2^r$ elements and $H_1^{-1} = \varnothing$. If $a \neq 1$, then $H_a^\varepsilon$ has $2^{r-1}$ elements.
\item[(c)] For $a, a' \in k^*/k^{*2}$ and $\varepsilon, \varepsilon' = \pm 1$, if $H_a^\varepsilon \cap H_{a'}^{\varepsilon'} = \varnothing$, then $a = a'$ and $\varepsilon = -\varepsilon'$.
\end{enumerate}
