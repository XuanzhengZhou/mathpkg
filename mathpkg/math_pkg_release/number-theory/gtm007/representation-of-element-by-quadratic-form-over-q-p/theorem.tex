\textbf{Corollary.} Let $a \in \mathbb{Q}_p^*/\mathbb{Q}_p^{*2}$. A quadratic form $f$ of rank $n$ with discriminant $d$ and invariant $\varepsilon$ represents $a$ if and only if:
\begin{enumerate}
\item[(i)] $n = 1$ and $a = d$,
\item[(ii)] $n = 2$ and $(a, -d) = \varepsilon$,
\item[(iii)] $n = 3$ and either $a \neq -d$ or $a = -d$ and $(-1, -d) = \varepsilon$,
\item[(iv)] $n \geq 4$.
\end{enumerate}
