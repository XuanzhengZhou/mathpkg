\textbf{Theorem 1.} Let $D = \{z \in H : |z| \geq 1, |\operatorname{Re}(z)| \leq 1/2\}$.
\begin{enumerate}
\item[(1)] For every $z \in H$, there exists $g \in G$ such that $gz \in D$.
\item[(2)] If $z, z' \in D$ are distinct and congruent modulo $G$, then $\operatorname{Re}(z) = \pm 1/2$ and $z = z' \pm 1$, or $|z| = 1$ and $z' = -1/z$.
\item[(3)] The stabilizer $I(z) = \{g \in G : gz = z\}$ is $\{1\}$ except at $z = i$ (stabilizer of order 2), $z = \rho = e^{2\pi i/3}$ (order 3), and $z = -\bar{\rho}$ (order 3).
\end{enumerate}
