\textbf{Proposition 6.} Let $n \geq 1$, $d \in \mathbb{Q}_p^*/\mathbb{Q}_p^{*2}$ and $\varepsilon = \pm 1$. There exists a quadratic form $f$ of rank $n$ with $d(f) = d$ and $\varepsilon(f) = \varepsilon$ if and only if:
\begin{itemize}
\item $n = 1$, $\varepsilon = 1$; or
\item $n = 2$, $d \neq -1$; or $n = 2$, $\varepsilon = 1$; or
\item $n \geq 3$.
\end{itemize}
