\textbf{Theorem 3.6.}
\begin{enumerate}
\item[(a)] If $h' \equiv \pm h \pmod{k}$, then $s(h', k) = \pm s(h, k)$, with the same sign as in the congruence.
\item[(b)] If $h\bar{h} \equiv \pm 1 \pmod{k}$ then $s(\bar{h}, k) = \pm s(h, k)$.
\item[(c)] If $h^2 + 1 \equiv 0 \pmod{k}$, then $s(h, k) = 0$.
\end{enumerate}
