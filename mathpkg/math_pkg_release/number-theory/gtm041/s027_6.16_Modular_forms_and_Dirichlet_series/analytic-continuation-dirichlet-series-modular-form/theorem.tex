Let $\varphi(s)$ be the function defined for $\sigma = \operatorname{Re}(s) > k$ by the Dirichlet series
$$\varphi(s) = \sum_{n=1}^{\infty} \frac{c(n)}{n^s}$$
associated with a modular form $f(\tau)$ in $M_k$ having the Fourier series
$$f(\tau) = c(0) + \sum_{n=1}^{\infty} c(n)e^{2\pi i n\tau},$$
where $k$ is an even integer $\geq 4$. Then $\varphi(s)$ can be continued analytically beyond the line $\sigma = k$ with the following properties:

\begin{enumerate}
\item[(a)] If $c(0) = 0$, $\varphi(s)$ is an entire function of $s$.
\item[(b)] If $c(0) \neq 0$, $\varphi(s)$ is analytic for all $s$ except for a simple pole at $s = k$ with residue
$$\frac{(-1)^{k/2} c(0) (2\pi)^k}{\Gamma(k)}.$$
\item[(c)] The function $\varphi$ satisfies the functional equation
$$(2\pi)^{-s} \Gamma(s) \varphi(s) = (-1)^{k/2} (2\pi)^{s-k} \Gamma(k-s) \varphi(k-s).$$
\end{enumerate}
