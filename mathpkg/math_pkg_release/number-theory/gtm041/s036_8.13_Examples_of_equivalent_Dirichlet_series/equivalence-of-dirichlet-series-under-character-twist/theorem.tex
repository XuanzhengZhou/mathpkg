\begin{theorem}[Theorem 8.17]
Let $k \geq 1$ be a given integer, and let $\chi$ be any Dirichlet character modulo $k$.
Let $\sum_{n=1}^{\infty} a(n)n^{-s}$ be any Dirichlet series whose coefficients have the following property:
$$a(n) \neq 0 \text{ implies } (n, k) = 1.$$
Then
$$\sum_{n=1}^{\infty} \frac{a(n)}{n^s} \sim \sum_{n=1}^{\infty} \frac{a(n)\chi(n)}{n^s}.$$
\end{theorem}
