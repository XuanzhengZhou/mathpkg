\text{Let } k \geq m \in \mathbb{N}, \, x_n \in K, \text{ and let } x = (x_0, \dots, x_{n-1}) \text{ be a norm-coherent sequence.}
\text{Assume } x_n = N_{k,n} x_k. \text{ Then}
$$
[x_n, x_n] \equiv [x_n] \pmod{n^{1+k}},
$$
\text{where } [\cdot, \cdot] \text{ is the local pairing and } [\cdot] \text{ denotes the value independent of } m \text{ (for } m \text{ sufficiently large).}
