\begin{enumerate}
\item Obvious from the definition of $j$ and the order on $P$.
\item $j''[p]_{P_\beta} \subseteq [j(p)]_{P_\alpha}$ is obvious from (1). Now take $q \in [j(p)]_{P_\alpha}$. Then $q \cup p \in \Gamma_\beta$. It is easy to see that $(q \cup p) \in [p]_{P_\beta}$ and $j(q \cup p) = q$. Hence $[j(p)]_{P_\alpha} \subseteq j''[p]_{P_\beta}$.
\end{enumerate}
