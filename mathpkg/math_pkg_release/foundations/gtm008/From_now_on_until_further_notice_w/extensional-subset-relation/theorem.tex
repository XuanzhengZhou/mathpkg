\textbf{Theorem 16.22.} Let $v \in V^{(B)}$ be extensional, $u \in V^{(B)}$, and $b = \llbracket v \subseteq u \rrbracket$. If $v' = b \cdot v$, then
\begin{enumerate}
\item $v'$ is extensional,
\item $\llbracket v' \subseteq u \rrbracket = 1$,
\item $\llbracket v \subseteq u \rrbracket = \llbracket v' = v \rrbracket$.
\end{enumerate}

\textbf{Theorem 16.23.} Let $u, v \in V^{(B)}$ be extensional and $\llbracket v \subseteq u \rrbracket = 1$. Define $v'$ by $\mathcal{D}(v') = \mathcal{D}(u)$ and $(\forall x \in \mathcal{D}(u))[v'(x) = \llbracket x \in v \rrbracket]$. Then $\llbracket v = v' \rrbracket = 1$.
