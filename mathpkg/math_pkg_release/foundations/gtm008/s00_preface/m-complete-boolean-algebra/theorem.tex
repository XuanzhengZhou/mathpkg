\begin{enumerate}
\item A Boolean algebra $B$ is \text{$M$-complete} iff $(\forall A \subseteq |B|) \left[ A \in M \rightarrow \sum_{b \in A} b \in |B| \right]$.
\item A homomorphism $f$ on $B$ is \text{$M$-complete} iff $(\forall A \subseteq |B|) \left[ A \in M \rightarrow f\left(\sum_{b \in A} b\right) = \sum_{b \in A} f(b) \land f\left(\prod_{b \in A} b\right) = \prod_{b \in A} f(b) \right]$.
\item An ideal $I$ in $B$ is \text{$M$-complete} iff $(\forall A \subseteq I) \left[ A \in M \rightarrow \sum_{b \in A} b \in I \right]$.
\end{enumerate}
