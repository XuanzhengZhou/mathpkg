\textbf{MacMillan's Theorem:} Let $X_n(w)$, $n = 0, \pm 1, \ldots$, be a stationary ergodic process taking finitely many values. Let
$$p_n(X_0, \ldots, X_{n-1}) = P[X_0 = X_0(w), \ldots, X_{n-1} = X_{n-1}(w)].$$
Then the entropy per unit time
$$H = \lim_{n \to \infty} \frac{H_n}{n}$$
exists, where $H_n$ is the entropy of $(X_1, \ldots, X_n)$. Moreover,
$$-\frac{1}{n} \log p_n(X_0, \ldots, X_{n-1}) \to H$$
in mean as $n \to \infty$.

Thus, with probability nearly one, the probability of a message of length $n$ is approximately $e^{-nH}$.
\end{enumerate}
