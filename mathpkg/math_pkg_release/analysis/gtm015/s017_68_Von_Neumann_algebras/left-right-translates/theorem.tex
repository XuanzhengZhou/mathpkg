\begin{definition}
For $f: G \to \mathbb{C}$ and $s \in G$, the \text{left translate} $f_s$ and \text{right translate} $f^s$ are defined by
$$f_s(t) = f(s^{-1}t), \qquad f^s(t) = f(ts).$$
If $f \in \mathcal{L}^1(G)$, then $f_s, f^s \in \mathcal{L}^1(G)$ and
$$\int f_s(t)\, dt = \int f(t)\, dt, \qquad \int f^s(t)\, dt = \Delta(s) \int f(t)\, dt.$$
For $u = [f] \in L^1(G)$, define $u_s = [f_s]$ and $u^s = [f^s]$.
\end{definition}
