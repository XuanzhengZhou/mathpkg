\begin{definition}
Let $u \mapsto T_u$ be the faithful left regular representation of $L^1(G)$ on $L^2(G)$. The set
$$\{T_u + \lambda I : u \in L^1(G),\; \lambda \in \mathbb{C}\}$$
is a $*$-subalgebra of $\mathcal{L}(L^2(G))$ containing the identity operator $I$. Its norm closure $\mathcal{A}$ in $\mathcal{L}(L^2(G))$ is a $C^*$-algebra with unit $I$.
\end{definition}
