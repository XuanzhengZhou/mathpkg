Let $D$ be a nonvoid set. For $f \in c_0(D)$, define $\|f\|_u = \sup\{|f(x)| : x \in D\}$.
\begin{itemize}
\item[(a)] $c_0(D)$ is a Banach space.
\item[(b)] $c_0(D)$ is separable iff $D$ is countable.
\item[(c)] If $\varphi$ is a bounded linear functional on $c_0(D)$, there exists $g \in \ell_1(D)$ such that $\varphi(f) = \sum_{x \in D} f(x) g(x)$ for all $f \in c_0(D)$.
\item[(d)] The mapping $\varphi \mapsto g$ is a norm-preserving isomorphism from $c_0(D)^*$ onto $\ell_1(D)$.
\end{itemize}
