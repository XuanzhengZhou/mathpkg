Let $G$ denote any group containing at least two elements. For $g \in G$, let $u_g \in \mathcal{L}(\ell^2(G))$ be defined by $u_g(f)(h) = f(g^{-1}h)$. The bicommutant of $\{u_g : g \in G\}$ in $\mathcal{L}(\ell^2(G))$ is called the $W^*$-group algebra $W(G)$.

\begin{enumerate}
\item[(a)] Denote by $e$ the neutral element of $G$ and let $\delta_e(f) := f(e)$, $f \in \ell^2(G)$. Then $\tau(x) := [x\delta_e, \delta_e]$ defines a normal trace that is faithful (i.e., a strictly positive linear form). Conclude that $W(G)$ is finite. (By use of the Cauchy--Schwarz inequality show that $\tau(xu_g) = 0$ for all $g \in G$ if $\tau(x) = 0$ and $x \geq 0$. Conclude that $\tau$ is strictly positive.)

\item[(b)] $W(G)$ is a factor iff for each $g \in G$, $g \neq e$ the coset $\{hgh^{-1} : h \in G\}$ is infinite; if so then $W(G)$ is of type $\mathrm{II}_1$.
\end{enumerate}
