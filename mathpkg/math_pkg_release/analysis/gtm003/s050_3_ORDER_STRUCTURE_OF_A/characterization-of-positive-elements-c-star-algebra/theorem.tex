\begin{lemma}\label{lem:positive-square-root}
Let $A$ be a $C^*$-algebra and $A_{sa} = \{x \in A : x = x^*\}$ its self-adjoint part.
\begin{enumerate}
\item[(a)] Let $x \in A_{sa}$ satisfy $\sigma(x) \subset \mathbf{R}_+$. There exists a unique element $y \in A_{sa}$ satisfying $\sigma(y) \subset \mathbf{R}_+$ and $y^2 = x$.
\item[(b)] Let $A$ be unital and $x \in A_{sa}$. The following are equivalent:
\begin{enumerate}
\item[(i)] There exists $t \in \mathbf{R}_+$ satisfying $\|x\| \leq t$ and $\|te - x\| \leq t$.
\item[(ii)] For all $t \in \mathbf{R}_+$, $\|x\| \leq t$ implies $\|te - x\| \leq t$.
\end{enumerate}
\end{enumerate}
\end{lemma}
