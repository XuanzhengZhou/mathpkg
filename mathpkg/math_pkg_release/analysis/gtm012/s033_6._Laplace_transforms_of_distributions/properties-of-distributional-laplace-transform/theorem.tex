Suppose $F, G \in \mathcal{L}'$ and suppose each satisfies a bound of the form
$$|F(u)| \leq K |u|_{k,a,M}, \quad u \in \mathcal{L}.$$
Then the Laplace transforms of $F$ and $G$ are defined for $\operatorname{Re} z > a$, and the following identities hold.

\begin{itemize}
  \item[(5)] $L(F+G) = LF + LG$.
  \item[(6)] $L(cF) = c\,LF$ for any constant $c \in \mathbb{C}$.
  \item[(7)] $L(T_s F)(z) = e^{-zs}\,LF(z)$, where $T_s$ is the translation operator.
  \item[(8)--(10)] Analogous identities for the operators $D^k$, $e_w$, and multiplication by $t^n$.
  \item[(11)] The distribution $e_w F$ is defined by
    $$e_w F(u) = F(e_w u), \quad u \in \mathcal{L},$$
    where $e_w(t) = e^{-wt}$.
  \item[(12)] If $F$ is the regular distribution defined by a function $f$, i.e.\ $F = F_f$, then
    $$LF = Lf,$$
    where $Lf$ is the classical Laplace transform of $f$.
\end{itemize}
