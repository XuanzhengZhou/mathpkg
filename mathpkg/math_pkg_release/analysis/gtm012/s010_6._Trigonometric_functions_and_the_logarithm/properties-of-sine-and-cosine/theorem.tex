The functions $S$ and $C$ of Theorem 6.1 have the following properties.

\begin{enumerate}
\item[(a)] $S$ and $C$ are of class $C^\infty$ and real-valued. In fact,
  $$
  S(x) = \frac{1}{2i}(e^{ix} - e^{-ix}), \qquad
  C(x) = \frac{1}{2}(e^{ix} + e^{-ix}),
  $$
  so $e^{ix} = C(x) + iS(x)$ for all $x \in \mathbb{R}$.

\item[(b)] $S' = C$, $C' = -S$, and $S(x)^2 + C(x)^2 = 1$ for all $x \in \mathbb{R}$.

\item[(c)] There exists a smallest positive number $p$ such that $C(p) = 0$.

\item[(d)] If $p$ is the number in (c), then
  $$S(x + 4p) = S(x), \qquad C(x + 4p) = C(x), \qquad \text{all } x \in \mathbb{R}.$$
\end{enumerate}
