The function $E(z) = \sum_{n=0}^{\infty} (n!)^{-1} z^n$ converges for all $z \in \mathbb{C}$ and satisfies:
\begin{itemize}
\item[(a)] $E'(x) = E(x)$, all $x \in \mathbb{R}$, and $E(0) = 1$,
\item[(b)] $E: \mathbb{R} \to (0, \infty)$ is strictly increasing and onto,
\item[(c)] $E(x + y) = E(x)E(y)$, all $x, y \in \mathbb{R}$,
\item[(d)] $E(n) = e^n$, $e = E(1)$, for all integers $n$,
\item[(e)] $E(z)E(-z) = 1$, and $E(z^*) = E(z)^*$.
\end{itemize}
Define $e^z = \exp(z) = E(z)$ for all $z \in \mathbb{C}$.
