\begin{theorem}[Existence, Uniqueness, and Weak$^*$ Continuity of Adjoints; One-to-One Transformations]
\label{thm:21.5}
Let $E$ and $F$ be locally convex linear topological spaces, and let $T$ be a linear transformation of $E$ into $F$.

\begin{itemize}
\item[(i)] If $T$ is continuous, then $T$ is weakly continuous. Moreover, $T$ has an adjoint $T^*$ if and only if $T$ is weakly continuous. The adjoint, if it exists, is unique and is necessarily weak$^*$ continuous.

\item[(ii)] A linear transformation $S$ of $F^*$ into $E^*$ is the adjoint of some linear transformation $T$ of $E$ into $F$ if and only if $S$ is weak$^*$ continuous. The transformation $T$, if it exists, is weakly continuous, and is uniquely determined by $S$.
\end{itemize}
\end{theorem}
