\textbf{Definition 28.1.}
Let $\mathfrak{A}$ be an $\mathcal{L}$-structure.
\begin{itemize}
\item $\mathfrak{A}$ is \textit{$\mathfrak{m}$-saturated} iff for every $X \subseteq A$ with $|X| < \mathfrak{m}$, if $\mathcal{L}'$ is an expansion of $\mathcal{L}$ obtained by adding a new individual constant $c_x$ for each $x \in X$, and if $\Delta \subseteq \mathrm{Fmla}_{\mathcal{L}'}^1$ is such that for each finite subset $\Delta'$ of $\Delta$ there is an $a \in A$ with $(\mathfrak{A}, x)_{x \in X} \models \bigwedge \Delta'[a]$, then there is an $a \in A$ such that $(\mathfrak{A}, x)_{x \in X} \models \varphi[a]$ for all $\varphi \in \Delta$.
\item $\mathfrak{A}$ is \textit{saturated} iff it is $|A|$-saturated.
\end{itemize}
