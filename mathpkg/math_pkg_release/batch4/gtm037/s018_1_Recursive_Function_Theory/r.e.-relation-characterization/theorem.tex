For $R \subseteq {}^m \omega$ the following are equivalent:

\begin{enumerate}
\item[(i)] $R = \emptyset$ or there exist elementary functions $f_0, \ldots, f_{m-1}$ with $R = \{(f_0 x, \ldots, f_{m-1} x) : x \in \omega\}$;
\item[(ii)] like (i) with ``elementary'' replaced by ``primitive recursive'';
\item[(iii)] $R$ is recursively enumerable;
\item[(iv)] there exist partial recursive functions $f_0, \ldots, f_{m-1}$ with $R = \{(f_0 x, \ldots, f_{m-1} x) : x \in \operatorname{Dmn} f_0 \cap \cdots \cap \operatorname{Dmn} f_{m-1}\}$;
\item[(v)] $R = \emptyset$ or there is an elementary function $f$ with $R = \{((f x)_0, \ldots, (f x)_{m-1}) : x \in \omega\}$;
\item[(vi)] like (v), with ``elementary'' replaced by ``primitive recursive'';
\item[(vii)] like (v), with ``elementary'' replaced by ``recursive'';
\item[(viii)] there is a partial recursive function $f$ such that $R = \{((f x)_0, \ldots, (f x)_{m-1}) : x \in \operatorname{Dmn} f\}$;
\item[(ix)] there is an $m$-ary partial recursive function $f$ such that $R = \operatorname{Dmn} f$;
\item[(x)] there is an $(m+1)$-ary recursive relation $S$ such that $R = \{(x_1, \ldots, x_m) : \exists y \, S(x_1, \ldots, x_m, y)\}$.
\end{enumerate}
