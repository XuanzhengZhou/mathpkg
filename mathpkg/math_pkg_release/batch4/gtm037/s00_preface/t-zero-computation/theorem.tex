For any tape description $F$ and any $e \in \mathbb{Z}$,
\begin{enumerate}
\item[(i)] if $Fe = 0$, then $\langle (F, 0, e) \rangle$ is a computation of $T_0$;
\item[(ii)] if $Fe = 1$, then $\langle (F, 0, e), (F_0^e, 0, e) \rangle$ is a computation of $T_0$, where $F_0^e$ is the tape description that is identical to $F$ except that $F_0^e(e) = 0$.
\end{enumerate}
