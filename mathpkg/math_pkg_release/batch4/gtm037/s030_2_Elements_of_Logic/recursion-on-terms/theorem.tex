Let $f$ map $\operatorname{Rng} v$ into a set $A$, and for each $O \in \operatorname{Dmn} \mathcal{O}$ (say with $\mathcal{O}O = m$), let $g_O$ map ${}^m A \times {}^m \operatorname{Trm}$ into $A$. Then there is an $h$: $\operatorname{Trm} \to A$ such that:
\begin{enumerate}
\item[(i)] $h\langle v_n \rangle = f(v_n)$ for all $n \in \omega$;
\item[(ii)] $h\langle O \rangle \sigma_0 \cdots \sigma_{m-1} = g_O(h\sigma_0, \ldots, h\sigma_{m-1}, \sigma_0, \ldots, \sigma_{m-1})$ whenever $O \in \operatorname{Dmn} \mathcal{O}$, $\mathcal{O}O = m$, and $\sigma \in {}^m \operatorname{Trm}$.
\end{enumerate}
