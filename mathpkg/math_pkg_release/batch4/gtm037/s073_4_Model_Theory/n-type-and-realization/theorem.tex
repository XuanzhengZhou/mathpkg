\begin{enumerate}
\item[(i)] For any $n \in \omega$, $\mathrm{Fmla}^n_{\mathcal{L}}$ is the set of all formulas $\varphi$ of $\mathcal{L}$ such that $\mathrm{Fv}\,\varphi \subseteq \{v_0, \ldots, v_{n-1}\}$.
\item[(ii)] If $\Delta \subseteq \mathrm{Fmla}^n_{\mathcal{L}}$, $\mathfrak{A}$ is an $\mathcal{L}$-structure, and $a \in {}^n A$, then we say that $a$ \emph{realizes} $\Delta$ in $\mathfrak{A}$ provided that $\mathfrak{A} \models \varphi[a]$ for all $\varphi \in \Delta$.
\item[(iii)] If $\Delta \subseteq \mathrm{Fmla}^n_{\mathcal{L}}$ and $\mathfrak{A}$ is an $\mathcal{L}$-structure, then we say that $\mathfrak{A}$ \emph{realizes} $\Delta$ provided that some $a \in {}^n A$ realizes $\Delta$ in $\mathfrak{A}$. If no $a \in {}^n A$ realizes $\Delta$, then we say that $\mathfrak{A}$ \emph{omits} $\Delta$.
\item[(iv)] If $\mathfrak{A}$ is an $\mathcal{L}$-structure and $a \in {}^n A$, then the \emph{$n$-type} of $a$ in $\mathfrak{A}$ is the set $\{\varphi \in \mathrm{Fmla}^n_{\mathcal{L}} : \mathfrak{A} \models \varphi[a]\}$.
\end{enumerate}
