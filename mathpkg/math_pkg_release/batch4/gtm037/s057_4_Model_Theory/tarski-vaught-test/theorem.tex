\text{Proposition 19.16 (Tarski--Vaught Test).}\quad
\text{Let } \mathfrak{A} \text{ and } \mathfrak{B} \text{ be } \mathcal{L}\text{-structures, and assume that } \mathfrak{A} \subseteq \mathfrak{B}.
\text{Then the following two conditions are equivalent:}

\text{(i) } \mathfrak{A} \preceq \mathfrak{B};

\text{(ii) for every formula } \varphi, \text{ every } k \in \omega, \text{ and every } x \in {}^\omega A, \text{ if } \mathfrak{B} \models \exists v_k \, \varphi[x], \text{ then there is an } a \in A \text{ such that } \mathfrak{B} \models \varphi[x_a^k].
