\textbf{Proposition 8.33.}

\begin{itemize}
  \item[(i)] $\vdash (\varphi \wedge \psi)^{d} \leftrightarrow \varphi^{d} \vee \psi^{d}$;
  \item[(ii)] $\vdash (\varphi \vee \psi)^{d} \leftrightarrow \varphi^{d} \wedge \psi^{d}$;
  \item[(iii)] $\vdash \varphi \leftrightarrow \varphi^{dd}$;
  \item[(iv)] if $f$ and $g$ are models of $\mathcal{P}$ and $fs = g^{+} \neg \langle s \rangle$ for all $s \in P$, then $f^{+} \varphi = g^{+} (\neg \varphi^{d})$ for every sentence $\varphi$;
  \item[(v)] $\Gamma \vdash \varphi$ iff $\{\neg \psi^{d} : \psi \in \Gamma\} \vdash \neg \varphi^{d}$;
  \item[(vi)] $\vdash \varphi$ iff $\vdash \neg \varphi^{d}$;
  \item[(vii)] $\vdash \varphi \rightarrow \psi$ iff $\vdash \psi^{d} \rightarrow \varphi^{d}$;
  \item[(viii)] $\vdash \varphi \leftrightarrow \psi$ iff $\vdash \varphi^{d} \leftrightarrow \psi^{d}$.
\end{itemize}
