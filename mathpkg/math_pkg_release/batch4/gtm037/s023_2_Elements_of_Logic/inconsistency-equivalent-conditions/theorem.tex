\textbf{Theorem 8.25.} The following conditions are equivalent:

\begin{itemize}
  \item[(i)] $\Gamma$ is inconsistent.
  \item[(ii)] $\Gamma \vdash \neg(\varphi \rightarrow \varphi)$ for every sentence $\varphi$.
  \item[(iii)] $\Gamma \vdash \neg(\varphi \rightarrow \varphi)$ for some sentence $\varphi$.
\end{itemize}
