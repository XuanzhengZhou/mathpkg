\begin{theorem}[28.6]
\label{thm:existence-of-m-saturated}
Let $\mathcal{L}, \mathcal{L}', \mathcal{L}''$ be first-order languages, $m$ and $n$ cardinals, and $\Gamma$ and $\Gamma'$ theories in $\mathcal{L}$ and $\mathcal{L}'$ respectively, subject to the following conditions:
\begin{enumerate}
\item[(i)] $\aleph_0 \leq m, n$;
\item[(ii)] $\mathcal{L}'$ is an expansion of $\mathcal{L}$, and $|\mathrm{Fmla}_{\mathcal{L}'}| = n$;
\item[(iii)] $\mathcal{L}''$ is an expansion of $\mathcal{L}'$ rich by $C$;
\item[(iv)] If $p$ is any nonzero cardinal $< m$, then $n^p \leq n$;
\item[(v)] $\Gamma \subseteq \Gamma'$, and $\Gamma'$ is consistent;
\item[(vi)] For each $B \subseteq C$ with $|B| < m$, let $\mathcal{L}_B$ be the reduct of $\mathcal{L}''$ to the symbols of $\mathcal{L}$ and $B$; then there are at most $n$ $1$-types over $\Gamma$ in $\mathcal{L}_B$.
\end{enumerate}
Then $\Gamma$ has an $m$-saturated model of power $\leq n$.
\end{theorem}
