\begin{definition}[28.15]
\label{def:universal-homogeneous-structures}
Let $K$ be a class of $\mathcal{L}$-structures and let $m$ be a cardinal.
\begin{enumerate}
\item An $\mathcal{L}$-structure $\mathfrak{A}$ is \textbf{$m$-universal over $K$} provided that every member of $K$ of power $<m$ can be embedded in $\mathfrak{A}$.
\item An $\mathcal{L}$-structure $\mathfrak{A}$ is \textbf{$m$-universal} provided that every model $\mathfrak{B} \equiv_{\mathrm{ee}} \mathfrak{A}$ of power $<m$ can be elementarily embedded in $\mathfrak{A}$. $\mathfrak{A}$ is \textbf{universal} iff it is $|A|$-universal.
\item An $\mathcal{L}$-structure $\mathfrak{A}$ is \textbf{algebraically $m$-homogeneous over $K$} provided that for every substructure $\mathfrak{B}$ of $\mathfrak{A}$ and for every isomorphism $f$ of $\mathfrak{B}$ into $\mathfrak{A}$, if $|B| < m$ and $\mathfrak{B} \in K$, then $f$ can be extended to an automorphism of $\mathfrak{A}$.
\item An $\mathcal{L}$-structure $\mathfrak{A}$ is \textbf{$m$-homogeneous} iff for all $\alpha < m$ and all $x \in {}^\alpha A$ and $y \in {}^\alpha A$, if $(\mathfrak{A}, x_\xi)_{\xi < \alpha} \equiv_{\mathrm{ee}} (\mathfrak{A}, y_\xi)_{\xi < \alpha}$, then for any $c \in A$ there is a $d \in A$ such that $(\mathfrak{A}, x_\xi, c)_{\xi < \alpha} \equiv_{\mathrm{ee}} (\mathfrak{A}, y_\xi, d)_{\xi < \alpha}$. $\mathfrak{A}$ is \textbf{homogeneous} if it is $|A|$-homogeneous.
\end{enumerate}
\end{definition}
