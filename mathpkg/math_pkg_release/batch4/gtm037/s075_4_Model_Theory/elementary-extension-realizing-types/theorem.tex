\begin{lemma}[(*)]
\label{lem:elementary-extension-realizing-types}
Suppose $|\mathrm{Fmla}_{\mathcal{L}}| \leq m$, and $\mathfrak{A}$ is an $\mathcal{L}$-structure with $\aleph_0 \leq |A| \leq 2^m$. Then there is an elementary extension $\mathfrak{B}$ of $\mathfrak{A}$ such that $|B| = 2^m$, and if $X$ is a subset of $A$ of power $\leq m$, $\mathcal{L}_X$ is an $X$-expansion of $\mathcal{L}$, and $\Delta$ is a $1$-type over $\mathrm{Th}((\mathfrak{A}, x)_{x \in X})$ in $\mathcal{L}_X$, then $\Delta$ is realized in $(\mathfrak{B}, x)_{x \in X}$.
\end{lemma}
