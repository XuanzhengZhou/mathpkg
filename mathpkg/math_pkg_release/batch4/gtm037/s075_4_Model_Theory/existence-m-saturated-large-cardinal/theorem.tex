\begin{theorem}[28.11]
\label{thm:existence-m-saturated-large-cardinal}
Let $\mathcal{L}, \mathcal{L}'$ be first-order languages, $m$ and $n$ cardinals, and $\Gamma'$ a theory in $\mathcal{L}'$, subject to the following conditions:
\begin{enumerate}
\item[(i)] $|\mathrm{Fmla}_{\mathcal{L}}| < m$;
\item[(ii)] $\mathcal{L}'$ is an expansion of $\mathcal{L}$, and $|\mathrm{Fmla}_{\mathcal{L}'}| = n$;
\item[(iii)] $\Gamma'$ is consistent.
\end{enumerate}
Then $\Gamma'$ has an $m$-saturated model.
\end{theorem}
