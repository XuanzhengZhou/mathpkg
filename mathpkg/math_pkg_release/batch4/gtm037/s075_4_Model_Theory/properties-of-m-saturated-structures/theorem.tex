\begin{proposition}[28.5]
\label{prop:properties-of-m-saturated}
Let $\mathfrak{A}$ be an $L$-structure and let $m$ be a cardinal.
\begin{enumerate}
\item[(i)] $\mathfrak{A}$ is $m$-saturated iff $\mathfrak{A}$ is $n$-saturated for every $n \leq m$;
\item[(ii)] If $m$ is a limit cardinal, then $\mathfrak{A}$ is $m$-saturated iff $\mathfrak{A}$ is $n$-saturated for every $n < m$;
\item[(iii)] $\mathfrak{A}$ is $m$-saturated iff for every $X \subseteq A$ with $|X| < m$, the structure $(\mathfrak{A}, x)_{x \in X}$ is $0$-saturated;
\item[(iv)] If $\mathfrak{A}$ is $m$-saturated and $X \subseteq A$ with $|X| < m$, then $(\mathfrak{A}, x)_{x \in X}$ is $m$-saturated.
\end{enumerate}
\end{proposition}
