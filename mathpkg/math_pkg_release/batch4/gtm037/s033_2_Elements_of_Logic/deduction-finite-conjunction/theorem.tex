Assume $\Gamma \cup \Delta \cup \{\varphi\} \subseteq \text{Sent}$. Then the following conditions are equivalent:

\begin{enumerate}
\item[(i)] $\Gamma \cup \Delta \vdash \varphi$;
\item[(ii)] $\Gamma \vdash \psi_0 \wedge \cdots \wedge \psi_{m-1} \rightarrow \varphi$ for some $m \in \omega$ and some $\psi \in {}^m\Delta$.
\end{enumerate}
