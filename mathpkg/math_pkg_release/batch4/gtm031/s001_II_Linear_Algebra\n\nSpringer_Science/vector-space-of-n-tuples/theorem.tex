\text{Let $\Delta$ be a division ring and $n$ a fixed positive integer.}
\text{Let $\Delta^{(n)}$ denote the totality of ordered $n$-tuples}
$(\xi_1, \xi_2, \dots, \xi_n)$ \text{with $\xi_i \in \Delta$. Two $n$-tuples}
$x = (\xi_1, \dots, \xi_n)$ \text{and} $y = (\eta_1, \dots, \eta_n)$
\text{are regarded as equal if and only if $\xi_i = \eta_i$ for $i = 1, 2, \dots, n$.}

\text{Define \textbf{addition} of vectors by}
$$x + y = (\xi_1 + \eta_1, \xi_2 + \eta_2, \dots, \xi_n + \eta_n)$$

\text{and \textbf{left scalar multiplication} by elements $\alpha \in \Delta$ via}
$$\alpha x = (\alpha \xi_1, \alpha \xi_2, \dots, \alpha \xi_n)$$

\text{The system $\Delta^{(n)}$ with these two compositions is called the}
\text{\textbf{vector space of $n$-tuples over $\Delta$}.}
