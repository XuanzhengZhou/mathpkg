Let $Q$ be a transient kernel on $S \times S$. A non-negative function $f(x)$ on $S$ is called

\textbf{D2}
\begin{itemize}
\item[(a)] $Q$-\textbf{regular} if
$$Qf(x) = \sum_{y \in S} Q(x,y)f(y) = f(x), \quad x \in S,$$

\item[(b)] $Q$-\textbf{excessive} if
$$Qf(x) = \sum_{y \in S} Q(x,y)f(y) \leq f(x), \quad x \in S,$$

\item[(c)] a $Q$-\textbf{potential} if
$$f(x) = \sum_{y \in S} g(x,y)\psi(y) < \infty, \quad x \in S$$
where $\psi(x) \geq 0$ for $x \in S$. In this case $f$ is called the \textbf{potential of the charge} $\psi$.
\end{itemize}
