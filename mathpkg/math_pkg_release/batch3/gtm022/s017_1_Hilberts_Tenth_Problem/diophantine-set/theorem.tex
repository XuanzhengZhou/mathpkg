\textbf{Definition 1.5.}
\begin{enumerate}
  \item[(i)] $S \subseteq \mathbb{N}^n$ is \textit{diophantine} if there is a polynomial $P(x_1, \ldots, x_n, y_1, \ldots, y_m)$ in $m + n \geq n$ variables with coefficients in $\mathbb{N}$ such that $(x_1, \ldots, x_n) \in S$ if and only if there exist values $y_1, \ldots, y_m$ for which $P(x_1, \ldots, x_n, y_1, \ldots, y_m) = 0$.

  \item[(ii)] A relation $\rho$ on $\mathbb{N}^n$ is \textit{diophantine} if the set $\{(x_1, \ldots, x_n) \mid \rho(x_1, \ldots, x_n) \text{ is true}\}$ is diophantine. In particular, a function $f: \mathbb{N}^n \to \mathbb{N}$ is \textit{diophantine} if $\{(x_1, \ldots, x_n, y) \mid y = f(x_1, \ldots, x_n)\}$ is diophantine.
\end{enumerate}

Equivalently,
$$
(x_1, \ldots, x_n) \in S \text{ iff } (\exists y_1, \ldots, y_m)\bigl(P(x_1, \ldots, x_n, y_1, \ldots, y_m) = 0\bigr).
$$
