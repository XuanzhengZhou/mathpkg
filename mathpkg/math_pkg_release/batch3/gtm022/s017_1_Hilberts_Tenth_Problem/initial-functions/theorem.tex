The set $I$ of \textit{initial functions} consists of:

\begin{enumerate}
  \item[(i)] the \textit{zero function} $c: \mathbb{N} \to \mathbb{N}$ given by $c(x) = 0$,
  \item[(ii)] the \textit{successor function} $s: \mathbb{N} \to \mathbb{N}$ given by $s(x) = x + 1$,
  \item[(iii)] the \textit{projection functions} $U_i^n: \mathbb{N}^n \to \mathbb{N}$ ($n \geq 1$, $1 \leq i \leq n$) given by $U_i^n(x_1, \ldots, x_n) = x_i$.
\end{enumerate}
