For $w \in \tilde{P}(V, \mathcal{R})$, the set $V(w)$ of variables occurring in $w$ is defined inductively by:
\begin{itemize}
\item[(i)] $V(F) = \emptyset$.
\item[(ii)] If $r \in \mathcal{R}$, $\text{ar}(r) = n$, and $x_1, \ldots, x_n \in V$, then $V(r(x_1, \ldots, x_n)) = \{x_1, \ldots, x_n\}$.
\item[(iii)] If $w_1, w_2 \in \tilde{P}$, then $V(w_1 \Rightarrow w_2) = V(w_1) \cup V(w_2)$.
\item[(iv)] If $x \in V$ and $w \in \tilde{P}$, then $V((\forall x)w) = \{x\} \cup V(w)$.
\end{itemize}
