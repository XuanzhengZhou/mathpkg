\text{A set } R \text{ with two binary operations } + \text{ (addition) and } \cdot \text{ (multiplication) is called a }
\text{RING if the following ring axioms are satisfied:}

\begin{enumerate}
\item[$R_1$.] $R$ \text{ is an abelian group with respect to addition}.
\item[$R_2$.] \text{If } a, b, c \in R, \text{ then } a(bc) = (ab)c.
\item[$R_3$.] \text{If } a, b, c \in R, \text{ then } a(b + c) = ab + ac \text{ and } (b + c)a = ba + ca
\text{ (distributive laws)}.
\end{enumerate}

\text{The identity element of } R \text{ (as an additive group) is denoted by } 0, \text{ and the additive inverse of } a
\text{ is denoted by } -a. \text{ Thus:}

$$0 + a = a + 0 = a,$$
$$a + (-a) = (-a) + a = 0,$$
$$-(-a) = a,$$
$$a + (b + c) = (a + b) + c,$$
$$a + b = b + a.$$
