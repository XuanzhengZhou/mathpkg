Let $\mathcal{K}$ be a category with an image factorization system $(E, M)$.

\begin{enumerate}
\item If $\{m_i: C_i \longrightarrow D_i\}_{i \in I}$ is a family of morphisms in $M$ and the products $\prod C_i$ and $\prod D_i$ exist in $\mathcal{K}$, then the induced morphism
$$
\prod m_i: \prod C_i \longrightarrow \prod D_i
$$
(defined by $(\prod m_i) \circ p_j = p_j \circ m_j$ for each projection $p_j$) belongs to $M$.

\item Given a pullback square
$$
\begin{array}{ccc}
P & \xrightarrow{t} & C \\
{\scriptstyle g}\downarrow & & \downarrow{\scriptstyle m} \\
B & \xrightarrow{f} & D
\end{array}
$$
with $m \in M$, the pullback morphism $t$ also belongs to $M$.
\end{enumerate}
