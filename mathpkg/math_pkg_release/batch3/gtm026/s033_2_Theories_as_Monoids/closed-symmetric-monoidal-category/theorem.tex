\text{A symmetric monoidal category} $(\mathcal{K}, \otimes, I, a, b, c)$ \text{is} \emph{closed} \text{if the functor}
$$
- \otimes A: \mathcal{K} \longrightarrow \mathcal{K}
$$
\text{has a right adjoint for every object} $A \in \mathcal{K}$.

\text{The cofree object over} $B$ \text{with respect to} $- \otimes A$ \text{is denoted} $B^A$, \text{so that}
$$
\frac{C \otimes A \longrightarrow B}{C \longrightarrow B^A}
$$
\text{and, setting} $C = I$, \text{“elements”} $I \longrightarrow B^A$ \text{are in bijective correspondence with morphisms} $A \longrightarrow B$.
