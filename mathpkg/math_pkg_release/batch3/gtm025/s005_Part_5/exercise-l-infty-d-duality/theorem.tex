Let $D$ be a nonvoid set and $\ell_\infty(D)$ the set of all bounded complex-valued functions on $D$ with $\|f\|_u = \sup\{|f(x)| : x \in D\}$.
\begin{itemize}
\item[(a)] If $\varphi \in \ell_1(D)^*$, there exists $g \in \ell_\infty(D)$ such that $\varphi(f) = \sum_{x \in D} f(x) g(x)$ for all $f \in \ell_1(D)$.
\item[(b)] The mapping $\varphi \mapsto g$ is a norm-preserving isomorphism from $\ell_1(D)^*$ onto $\ell_\infty(D)$.
\item[(c)] If $D$ is infinite, then $c_0(D)$ is not reflexive.
\end{itemize}
