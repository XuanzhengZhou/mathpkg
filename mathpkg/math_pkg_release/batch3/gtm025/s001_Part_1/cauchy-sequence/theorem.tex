Let $F$ be an ordered field. A sequence $(a_n) = (a_1, a_2, \ldots)$ of elements of $F$ is called:
\begin{itemize}
\item \textbf{bounded} if there is an element $b \in F$ such that $|a_n| \leq b$ for each positive integer $n$;
\item \textbf{Cauchy} if for every $e \in F$ such that $e > 0$, there is a positive integer $N(e)$ such that $|a_p - a_q| < e$ for all $p, q \geq N(e)$;
\item \textbf{null} if for every $e \in F$ such that $e > 0$, there is a positive integer $N(e)$ such that $|a_n| < e$ for all $n \geq N(e)$.
\end{itemize}
