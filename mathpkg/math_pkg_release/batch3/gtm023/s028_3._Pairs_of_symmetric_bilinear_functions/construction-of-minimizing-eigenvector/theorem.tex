Let $E$ be a real vector space of dimension $n \geq 3$, and let $\Phi$ and $\Psi$ be symmetric bilinear functions on $E$ with $\Phi(x)^2 + \Psi(x)^2 \neq 0$ for $x \neq 0$. Assume $\Psi$ is non-degenerate, and let $F$ be the homogeneous functional defined via the path-independent integral $J$. Introduce a positive definite inner product $|\cdot|$ on $E$. Then:
\begin{enumerate}
\item The continuous function $F$ attains a minimum on the unit sphere $S = \{x \in E : |x| = 1\}$.
\item If $e_1 \in S$ is a point where $F$ attains its minimum, then for all $y \in E$,
$$\Phi(e_1)\Psi(e_1, y) - \Phi(e_1, y)\Psi(e_1) = 0.$$
\item Moreover, $\Psi(e_1) \neq 0$, and setting
$$\lambda_1 = \frac{\Phi(e_1)}{\Psi(e_1)},$$
we have
$$\Phi(e_1, y) = \lambda_1 \Psi(e_1, y) \quad \text{for all } y \in E.$$
\end{enumerate}
