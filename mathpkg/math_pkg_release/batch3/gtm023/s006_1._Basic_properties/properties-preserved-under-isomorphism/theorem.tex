Let $\varphi: E \xrightarrow{\cong} F$ be a linear isomorphism.

\textbf{Property I.} The image under $\varphi$ of a generating set (resp., linearly independent set, basis) in $E$ is a generating set (resp., linearly independent set, basis) in $F$.

\textbf{Property II.} If $E_1$ is any subspace of $E$, then $\varphi$ determines linear isomorphisms
$$
E_1 \xrightarrow{\cong} \varphi(E_1), \qquad E/E_1 \xrightarrow{\cong} \varphi(E)/\varphi(E_1).
$$

\textbf{Property III.} If $G$ is a third vector space, then the mappings
$$
\psi \mapsto \psi \circ \varphi^{-1} \quad (\psi \in L(E; G)), \qquad \psi \mapsto \varphi \circ \psi \quad (\psi \in L(G; E))
$$
are linear isomorphisms
$$
L(E; G) \xrightarrow{\cong} L(F; G), \qquad L(G; E) \xrightarrow{\cong} L(G; F).
$$
