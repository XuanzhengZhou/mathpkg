\text{\textbf{Lemma III.}} \text{ Let $g \in \Gamma[t]$ be a polynomial such that $g$ and $g'$ are relatively prime. Then for each integer $k \geq 1$ there exist polynomials $u, v \in \Gamma[t]$ with the following properties:}
\begin{itemize}
\item[(i)] $g^{k} \mid g(u)$,
\item[(ii)] $u + g v = t$,
\item[(iii)] \text{If $k \geq 2$, then $v$ is relatively prime to $g$.}
\end{itemize}
