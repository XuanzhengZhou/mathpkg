\text{\textbf{Lemma IV.}} \text{ Let $g \in \Gamma[t]$ be a polynomial such that $g$ and $g'$ are relatively prime. Then for each $k \geq 1$ there exist polynomials $u, \omega \in \Gamma[t]$ with the following properties:}
\begin{itemize}
\item[(i)] $g^{k} \mid g(u)$,
\item[(ii)] $u + \omega = t$,
\item[(iii)] $g \mid \omega$.
\end{itemize}
