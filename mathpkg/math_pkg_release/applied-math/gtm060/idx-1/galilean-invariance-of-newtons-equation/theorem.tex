Galileo's principle of relativity imposes the following constraints on Newton's equation $\ddot{x} = F(x, \dot{x}, t)$ in an inertial coordinate system:

\begin{enumerate}
\item \textbf{Time translation invariance:} The right-hand side does not depend on time: $\ddot{x} = \Phi(x, \dot{x})$.

\item \textbf{Space translation invariance:} The right-hand side depends only on relative coordinates $x_j - x_k$.

\item \textbf{Galilean boost invariance:} The right-hand side depends only on relative velocities $\dot{x}_j - \dot{x}_k$:
$$\ddot{x}_i = f_i(\{x_j - x_k, \dot{x}_j - \dot{x}_k\}), \quad i, j, k = 1, \ldots, n.$$

\item \textbf{Rotational invariance:} For any orthogonal transformation $G: \mathbb{R}^3 \rightarrow \mathbb{R}^3$,
$$\mathbf{F}(G\mathbf{x}, G\dot{\mathbf{x}}) = G\mathbf{F}(\mathbf{x}, \dot{\mathbf{x}}).$$
\end{enumerate}
