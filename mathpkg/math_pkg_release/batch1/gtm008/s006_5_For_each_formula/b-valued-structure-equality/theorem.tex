For all $t_1, t_2, t_3 \in T_{\alpha}$,
\begin{enumerate}
\item $\llbracket t_1 = t_2 \rrbracket \wedge \llbracket t_2 \in t_3 \rrbracket \leq \llbracket t_1 \in t_3 \rrbracket$.
\item $\llbracket t_1 = t_2 \rrbracket \wedge \llbracket t_3 \in t_1 \rrbracket \leq \llbracket t_3 \in t_2 \rrbracket$.
\item $\llbracket t_1 = t_2 \rrbracket \wedge \llbracket t_3 = t_1 \rrbracket \leq \llbracket t_3 = t_2 \rrbracket$.
\item $\llbracket t_1 = t_2 \rrbracket \wedge \llbracket t_2 = t_3 \rrbracket \leq \llbracket t_1 = t_3 \rrbracket$.
\end{enumerate}
