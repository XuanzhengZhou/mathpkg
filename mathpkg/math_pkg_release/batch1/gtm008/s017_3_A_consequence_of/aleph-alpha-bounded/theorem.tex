Let $P = \langle P, \leq, 1 \rangle$ be strongly coatomic. $P$ is said to be \emph{$\aleph_\alpha$-bounded} iff
\begin{enumerate}
\item $\overline{CA(p)} < \aleph_\alpha$,
\item Define $IC(p) = \{q \mid q \text{ is coatomic } \land p \text{ and } q \text{ are incompatible}\}$. Then $\overline{IC(p)} \leq \aleph_\alpha$,
\item If $p$ and $q$ are incompatible, then there exists $q_0 \in IC(p)$ such that $q \leq q_0$,
\item Let $A$ be a set of coatomic elements with $\bar{A} \leq \aleph_\alpha$. Then $\overline{\{p \mid CA(p) \subseteq A\}} \leq \aleph_\alpha$.
\end{enumerate}
