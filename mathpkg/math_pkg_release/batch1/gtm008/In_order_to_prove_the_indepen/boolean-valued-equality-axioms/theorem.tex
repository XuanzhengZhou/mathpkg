\begin{enumerate}
\item $[\![\operatorname{rank}(u) < \alpha]\!] \wedge [\![\operatorname{rank}(u') < \alpha]\!] \wedge [\![\operatorname{rank}(v) \leq \alpha]\!] \rightarrow [\![u = u']\!] \cdot [\![u \in v]\!] \leq [\![u' \in v]\!]$.

\end{enumerate}
