\begin{enumerate}
\item Let $d \subseteq V^{(\mathbf{B})}$. A function $g: d \rightarrow V^{(\mathbf{B})}$ is called \textbf{extensional} iff
$$(\forall x, x' \in d)[\llbracket x = x' \rrbracket \leq \llbracket g(x) = g(x') \rrbracket].$$
\item Let $u \in V^{(\mathbf{B})}$. Then $u$ is \textbf{definite} iff
$$(\forall x \in \mathcal{D}(u))[u(x) = 1].$$
\end{enumerate}
