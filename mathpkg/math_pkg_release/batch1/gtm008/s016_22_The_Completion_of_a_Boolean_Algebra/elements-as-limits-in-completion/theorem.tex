Let $\mathbf{B}$ be a Boolean algebra and $\tilde{B}$ its completion. For any $b \in \tilde{B}$, define $b_{\alpha} = \sup (B \cap R(\alpha) \cap \{x \mid x \leq b\})$ for $\alpha \in \mathrm{On}$. Then:

\begin{enumerate}
\item $b_{\alpha} \leq b$.
\item $\alpha < \beta \rightarrow b_{\alpha} \leq b_{\beta}$ and $b_{\alpha} = \prod \{b_{\beta} \mid \beta \geq \alpha\}$.
\item $\sum_{\alpha \in \mathrm{On}} b_{\alpha} = b$.
\end{enumerate}
