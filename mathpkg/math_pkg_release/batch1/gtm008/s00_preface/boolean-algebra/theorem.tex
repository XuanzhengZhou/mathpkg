A structure $\langle B, +, \cdot, ^-, 0, 1 \rangle$ is a \text{Boolean algebra} with universe $B$ iff $0$ and $1$ are two (distinct) elements of $B$; $+$ and $\cdot$ are binary operations on $B$; $^-$ is a unary operation on $B$; and $\forall a, b, c \in B$:

\begin{enumerate}
\item $a + b = b + a$, \quad $ab = ba$ \qquad \text{Commutative Laws.}
\item $a + (b + c) = (a + b) + c$, \quad $a(bc) = (ab)c$ \qquad \text{Associative Laws.}
\item $a + bc = (a + b)(a + c)$, \quad $a(b + c) = ab + ac$ \qquad \text{Distributive Laws.}
\item $0 + a = a$, \quad $1a = a$ \qquad \text{Identity Laws.}
\item $a + a^- = 1$, \quad $a(a^-) = 0$ \qquad \text{Complementation Laws.}
\end{enumerate}
