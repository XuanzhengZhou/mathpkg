Let $B_0, B_1$ be complete Boolean algebras and let $i: B_0 \rightarrow B_1$ be a complete monomorphism. For $b_1 \in B_1$, define
$$\#(b_1) = \inf \{b_0 \in B_0 \mid b_1 \leq i(b_0)\} \quad \text{in } B_0.$$
Then:

\begin{enumerate}
\item $b_1 \leq i(\#(b_1))$.
\item $b_1 = 0 \rightarrow \#(b_1) = 0$.
\item $\#(i(b_0)) = b_0$.
\item $b_1 \leq i(b_0) \leftrightarrow \#(b_1) \leq b_0$.
\item $b_0 \cdot \#(b_1) = \#(i(b_0) \cdot b_1)$.
\item $i(b_0) \cdot b_1 = 0 \rightarrow b_0 \cdot \#(b_1) = 0$.
\end{enumerate}
