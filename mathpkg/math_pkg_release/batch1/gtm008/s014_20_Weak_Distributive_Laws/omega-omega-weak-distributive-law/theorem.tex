Let $B$ be a complete Boolean algebra.

\begin{definition}
$B$ satisfies the $(\omega, \omega)$-weak distributive law ($(\omega, \omega)$-WDL) iff for every family $\{b_{nm} \mid n, m \in \omega\} \subseteq B$,
$$
\prod_{m < \omega} \sum_{n < \omega} b_{nm} = \sum_{f \in \omega^{\omega}} \prod_{n \in \omega} \sum_{m \leq f(n)} b_{nm}.
$$

If $\omega_{\alpha}$ is not cofinal with $\omega$, $B$ satisfies the $(\omega, \omega_{\alpha})$-weak distributive law iff for every family $\{b_{n\xi} \mid n \in \omega \land \xi < \omega_{\alpha}\} \subseteq B$,
$$
\prod_{n < \omega} \sum_{\xi < \omega_{\alpha}} b_{n\xi} = \sum_{f \in \omega_{\alpha}^{\omega}} \prod_{n \in \omega} \sum_{\xi \leq f(n)} b_{n\xi}.
$$
\end{definition}

\begin{remark}
If $\operatorname{cf}(\omega_{\alpha}) > \omega$, the right-hand side is equal to
$$
\sum_{\eta < \omega_{\alpha}} \prod_{n < \omega} \sum_{\xi \leq \eta} b_{n\xi}.
$$
\end{remark}
