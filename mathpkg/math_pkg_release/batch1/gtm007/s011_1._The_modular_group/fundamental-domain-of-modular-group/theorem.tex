Let $D$ be the subset of the upper half-plane $H$ formed of all points $z$ such that $|z| \geq 1$ and $|\operatorname{Re}(z)| \leq 1/2$.

\begin{enumerate}
\item[(1)] For every $z \in H$, there exists $g \in G$ such that $gz \in D$.
\item[(2)] Suppose that two distinct points $z, z'$ of $D$ are congruent modulo $G$. Then $\operatorname{Re}(z) = \pm \frac{1}{2}$ and $z = z' \pm 1$, or $|z| = 1$ and $z' = -1/z$.
\item[(3)] Let $z \in D$ and let $I(z) = \{g \mid g \in G,\; gz = z\}$ be the stabilizer of $z$ in $G$. One has $I(z) = \{1\}$ except in the following three cases:
\begin{itemize}
\item $z = i$, in which case $I(z)$ is the group of order 2 generated by $S$;
\item $z = \rho = e^{2\pi i/3}$, in which case $I(z)$ is the group of order 3 generated by $ST$;
\item $z = -\bar{\rho} = e^{\pi i/3}$, in which case $I(z)$ is the group of order 3 generated by $TS$.
\end{itemize}
\end{enumerate}
