Let $M_k$ be the space of modular forms of weight $2k$, and $M_k^0$ the subspace of cusp forms. Let $\Delta = g_2^3 - 27 g_3^2$, where $g_2 = 60 G_2$ and $g_3 = 140 G_3$.

\begin{enumerate}
\item[(i)] $M_k = 0$ for $k < 0$ and for $k = 1$.
\item[(ii)] For $k = 0, 2, 3, 4, 5$, $M_k$ is a vector space of dimension $1$ with basis $1, G_2, G_3, G_4, G_5$ respectively; moreover $M_k^0 = 0$.
\item[(iii)] Multiplication by $\Delta$ defines an isomorphism of $M_{k-6}$ onto $M_k^0$.
\end{enumerate}
