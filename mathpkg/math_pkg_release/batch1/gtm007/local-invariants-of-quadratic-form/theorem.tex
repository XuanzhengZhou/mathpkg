Let $f \sim a_1 X_1^2 + \cdots + a_n X_n^2$ be a quadratic form of rank $n$ over $\mathbb{Q}$. One associates to it:
\begin{enumerate}
\item[(a)] The discriminant $d(f) = a_1 \cdots a_n \in \mathbb{Q}^*/\mathbb{Q}^{*2}$.
\item[(b)] For each $v \in V$, the local invariants $d_v(f)$ (image of $d(f)$ in $\mathbb{Q}_v^*/\mathbb{Q}_v^{*2}$) and $\varepsilon_v(f) = \prod_{i < j} (a_i, a_j)_v$ in $\{\pm 1\}$, satisfying $\prod_{v \in V} \varepsilon_v(f) = 1$.
\item[(c)] The signature $(r, s)$ of the real quadratic form $f_\infty$.
\end{enumerate}
