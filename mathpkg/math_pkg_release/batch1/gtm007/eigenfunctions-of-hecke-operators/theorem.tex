\textbf{Theorem 7.} Let $f(z) = \sum_{n=0}^{\infty} c(n) q^n$ be a nonzero modular form of weight $2k$, $k > 0$, which is an eigenfunction of all $T(n)$: $T(n)f = \lambda(n)f$.
\begin{enumerate}
\item[(a)] $c(1) \neq 0$.
\item[(b)] If $f$ is normalized by $c(1) = 1$, then $c(n) = \lambda(n)$ for all $n$, and the $c(n)$ satisfy
$$c(m)c(n) = \sum_{a \mid \gcd(m,n)} a^{2k-1} c(mn/a^2).$$
In particular, $c(mn) = c(m)c(n)$ if $\gcd(m,n) = 1$, and for $p$ prime,
$$c(p^{n+1}) = c(p)c(p^n) - p^{2k-1}c(p^{n-1}).$$
\end{enumerate}
