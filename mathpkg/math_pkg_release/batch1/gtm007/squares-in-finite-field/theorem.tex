\textbf{Theorem 4.}
\begin{enumerate}
\item[(a)] If $p = 2$, then all elements of $\mathbb{F}_q$ are squares.
\item[(b)] If $p \neq 2$, then the squares of $\mathbb{F}_q^*$ form a subgroup of index $2$ in $\mathbb{F}_q^*$; this subgroup is the kernel of the homomorphism $x \mapsto x^{(q-1)/2}$ with values in $\{\pm 1\}$. In other terms, one has an exact sequence:
$$1 \to \mathbb{F}_q^{*2} \to \mathbb{F}_q^* \to \{\pm 1\} \to 1.$$
\end{enumerate}
