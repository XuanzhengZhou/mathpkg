\begin{theorem}[Theorem 3.21]
For any measurable set $A \subset \mathbb{R}$, let $\phi(A)$ denote the set of points of $\mathbb{R}$ where $A$ has density $1$. Write $A \sim B$ when $m(A \triangle B) = 0$. Then $\phi$ has the following properties:
\begin{enumerate}
\item[(1)] $\phi(A) \sim A$,
\item[(2)] $A \sim B$ implies $\phi(A) = \phi(B)$,
\item[(3)] $\phi(\emptyset) = \emptyset$ and $\phi(\mathbb{R}) = \mathbb{R}$,
\item[(4)] $\phi(A \cap B) = \phi(A) \cap \phi(B)$,
\item[(5)] $A \subset B$ implies $\phi(A) \subset \phi(B)$.
\end{enumerate}
\end{theorem}
