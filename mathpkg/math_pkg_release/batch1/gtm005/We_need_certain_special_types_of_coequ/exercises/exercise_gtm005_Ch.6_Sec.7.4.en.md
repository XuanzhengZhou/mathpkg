\begin{enumerate}
\item[(a)] Show that the canonical presentation of a $T$-algebra $\langle x, h \rangle$ is
$$\langle T^2 x, \mu_{Tx} \rangle \xrightarrow{T h} \langle T x, \mu_x \rangle \xrightarrow{h} \langle x, h \rangle.$$
\item[(b)] Show that the comparison functor $M : X^T \to A$ in Beck's theorem appears as the coequalizer diagram
$$F G F x \xrightarrow{\varepsilon_{F x}} F x \to M(x, h).$$
\end{enumerate}
