\text{In any square pullback diagram}
$$
\begin{array}{ccc}
p & \xrightarrow{f'} & t \\[4pt]
{\scriptstyle g'}\downarrow && \downarrow{\scriptstyle g} \\[4pt]
s & \xrightarrow{f} & a
\end{array}
$$
\text{$f$ monic implies $f'$ monic (and $g$ monic implies $g'$ monic).}

\text{Briefly, pullbacks of monics are monic.}
