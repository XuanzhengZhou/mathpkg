For the root space decomposition $L = H \oplus \bigoplus_{\alpha \in \Phi} L_{\alpha}$:
\begin{enumerate}
\item[(a)] $\kappa(L_{\alpha}, L_{\beta}) = 0$ unless $\alpha + \beta = 0$.
\item[(b)] The restriction of $\kappa$ to $L_{\alpha} \times L_{-\alpha}$ is nondegenerate.
\item[(c)] $[L_{\alpha} L_{-\alpha}]$ is one-dimensional, with basis $t_{\alpha}$ characterized by $\kappa(t_{\alpha}, h) = \alpha(h)$ for all $h \in H$.
\item[(d)] $\kappa(t_{\alpha}, t_{\alpha}) \neq 0$; equivalently, $\alpha(t_{\alpha}) \neq 0$.
\item[(e)] $[x_{\alpha} x_{-\alpha}] = \kappa(x_{\alpha}, x_{-\alpha}) t_{\alpha}$ for all $x_{\alpha} \in L_{\alpha}, x_{-\alpha} \in L_{-\alpha}$.
\item[(f)] Set $h_{\alpha} = \frac{2 t_{\alpha}}{\kappa(t_{\alpha}, t_{\alpha})}$; then $\alpha(h_{\alpha}) = 2$.
\item[(g)] $h_{-\alpha} = -h_{\alpha}$.
\end{enumerate}
