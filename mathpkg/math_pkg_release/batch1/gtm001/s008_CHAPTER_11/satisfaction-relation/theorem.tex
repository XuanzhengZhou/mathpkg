\begin{align*}
(1)&\quad [A, R] \models a \in b \leftrightarrow a \in A \wedge b \in A \wedge a R b,\\
(2)&\quad [A, R] \models \neg \psi \leftrightarrow \neg [[A, R] \models \psi],\\
(3)&\quad [A, R] \models \psi \wedge \eta \leftrightarrow [[A, R] \models \psi] \wedge [[A, R] \models \eta],\\
(4)&\quad [A, R] \models (\forall x) \psi(x) \leftrightarrow (\forall x \in A) [[A, R] \models \psi(x)].
\end{align*}
