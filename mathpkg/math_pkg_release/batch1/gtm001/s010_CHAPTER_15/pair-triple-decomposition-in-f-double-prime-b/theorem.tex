If $9 \subseteq b \subseteq \mathrm{On}$ and $b$ is closed w.r.t. $C$ and $J$, then
\begin{enumerate}
\item $(\forall x, y)[\{x, y\} \in F''b \rightarrow x, y \in F''b],$
\item $(\forall x, y)[\langle x, y \rangle \in F''b \rightarrow x, y \in F''b],$
\item $(\forall x, y, z)[\langle x, y, z \rangle \in F''b \rightarrow x, y, z \in F''b].$
\end{enumerate}
