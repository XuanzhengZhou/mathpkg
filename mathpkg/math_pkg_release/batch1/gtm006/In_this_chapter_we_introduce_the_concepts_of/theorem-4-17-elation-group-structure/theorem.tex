Let $\mathcal{P}$ be a projective plane and $\Gamma$ a collineation group of $\mathcal{P}$.
\begin{enumerate}
\item[(i)] If $\ell$ is any line of $\mathcal{P}$, then for any line $m \neq \ell$, $(\Gamma_{(\ell,\ell)})_m = \Gamma_{(A,\ell)}$ where $A = \ell m$.
\item[(ii)] If $\mathcal{P}$ is finite of order $n$ and $|\Gamma_{(\ell,\ell)}| > n$, then $|\Gamma_{(B,\ell)}| > 1$ for all $B$ on $\ell$.
\item[(iii)] If $\mathcal{P}$ is finite of order $n$, then $|\Gamma_{(\ell,\ell)}| \mid n$ for any line $m$.
\end{enumerate}
