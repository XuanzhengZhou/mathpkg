Let $w_1, w_2 \in \tilde{P}(V, \mathcal{R})$. The relation $w_1 \approx w_2$ is defined inductively by:
\begin{itemize}
\item[(a)] $d(w_1) = d(w_2) = 0$ and $w_1 = w_2$, or
\item[(b)] $d(w_1) = d(w_2) > 0$, $w_1 = a_1 \Rightarrow b_1$, $w_2 = a_2 \Rightarrow b_2$, $a_1 \approx a_2$ and $b_1 \approx b_2$, or
\item[(c)] $w_1 = (\forall x)a$, $w_2 = (\forall x)b$ and $a \approx b$ where $x \in V$.
\end{itemize}
