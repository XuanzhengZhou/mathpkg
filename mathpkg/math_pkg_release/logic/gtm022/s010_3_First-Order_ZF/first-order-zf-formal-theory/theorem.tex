$$
\mathcal{S} \text{ is the first-order theory with relation symbols } \mathcal{I} \text{ (binary, for equality) and } \in \text{ (binary, for membership).}
$$
$$
\text{No individual constants are used. The language uses an infinite set } V \text{ of variables.}
$$
$$
\text{For readability, write } x \in y \text{ for } {\in}(x, y) \text{ and } x = y \text{ for } \mathcal{I}(x, y).
$$
$$
\text{Negations: } x \notin y \text{ and } x \neq y.
$$
