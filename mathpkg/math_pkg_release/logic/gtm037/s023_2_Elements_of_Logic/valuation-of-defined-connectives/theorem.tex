\textbf{Proposition 8.31.} If $f$ is a model of $\mathcal{P}$, then for any sentences $\varphi, \psi$:
\begin{align*}
f^+(\varphi \lor \psi) &= 1 \quad \text{iff } f^+\varphi = 1 \text{ or } f^+\psi = 1,\\
f^+(\varphi \land \psi) &= 1 \quad \text{iff } f^+\varphi = 1 \text{ and } f^+\psi = 1,\\
f^+(\varphi \leftrightarrow \psi) &= 1 \quad \text{iff } f^+\varphi = f^+\psi.
\end{align*}
