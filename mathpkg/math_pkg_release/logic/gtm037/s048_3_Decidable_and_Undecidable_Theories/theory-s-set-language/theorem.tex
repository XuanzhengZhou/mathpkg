Let $\mathcal{L}_{\text{set}}$ be a fixed first-order language with a single non-logical constant, a binary relation symbol $\in$. Let $S$ be the theory in $\mathcal{L}_{\text{set}}$ with the following axioms:

\begin{enumerate}
\item[(1)] $\exists v_0 \forall v_1 \neg (v_1 \in v_0)$,
\item[(2)] $\forall v_0 \forall v_1 [\forall v_2(v_2 \in v_0 \leftrightarrow v_2 \in v_1) \rightarrow v_0 = v_1]$,
\item[(3)] $\forall v_0 \forall v_1 \exists v_2 \forall v_3(v_3 \in v_2 \leftrightarrow v_3 \in v_0 \lor v_3 = v_1)$.
\end{enumerate}
