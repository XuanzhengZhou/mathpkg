Let $\Gamma \subseteq \text{Fmla}$ and $\Delta \subseteq \text{Fmla}$. The following conditions are equivalent:
\begin{enumerate}
\item[(i)] $\Gamma \cup \Delta$ is inconsistent;
\item[(ii)] there exist $m \in \omega \setminus 1$ and $\varphi \in {}^m\Delta$ such that
  $$\Gamma \vdash \neg\varphi_0 \lor \dots \lor \neg\varphi_{m-1}.$$
\end{enumerate}
