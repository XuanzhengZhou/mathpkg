If $\Gamma \subseteq \text{Sent}$, then the following conditions are equivalent:
\begin{enumerate}
\item[(i)] $\Gamma$ is inconsistent;
\item[(ii)] there exist $m \in \omega \setminus 1$ and $\varphi \in {}^m\Gamma$ such that
  $$\vdash \neg\varphi_0 \lor \dots \lor \neg\varphi_{m-1}.$$
\end{enumerate}
