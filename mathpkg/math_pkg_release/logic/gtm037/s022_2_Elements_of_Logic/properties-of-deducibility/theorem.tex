Let $\Gamma, \Delta \subseteq \text{Sent}_{\mathcal{P}}$ and $\varphi, \psi \in \text{Sent}_{\mathcal{P}}$. Then:
\begin{enumerate}
\item[(i)] if $\Delta \subseteq \Gamma$ and $\Delta \vdash \varphi$, then $\Gamma \vdash \varphi$;
\item[(ii)] if $\Gamma \vdash \varphi$, then $\Theta \vdash \varphi$ for some finite subset $\Theta$ of $\Gamma$;
\item[(iii)] if $\Gamma \vdash \chi$ for each $\chi \in \Delta$ and $\Delta \vdash \varphi$, then $\Gamma \vdash \varphi$;
\item[(iv)] if $\Gamma \vdash \varphi$ and $\Gamma \vdash \varphi \to \psi$, then $\Gamma \vdash \psi$.
\end{enumerate}
