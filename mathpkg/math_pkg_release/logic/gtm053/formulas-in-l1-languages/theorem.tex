\begin{enumerate}
\item[(a)] If $p$ is a relation of degree $r$ and $t_1, \ldots, t_r$ are terms, then $p(t_1, \ldots, t_r)$ is an (atomic) formula.
\item[(b)] If $P$ and $Q$ are formulas and $x$ is a variable, then the expressions
$$(P) \Leftrightarrow (Q),\; (P) \Rightarrow (Q),\; (P) \vee (Q),\; (P) \wedge (Q),$$
$$\neg(P),\; \forall x(P),\; \exists x(P)$$
are formulas.
\end{enumerate}
