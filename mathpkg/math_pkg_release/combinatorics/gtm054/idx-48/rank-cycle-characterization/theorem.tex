$$
\text{Let }\Lambda = (V, \mathcal{M})\text{ be a matroid with rank function }r.
$$
$$
\text{Let }U \in \mathcal{P}(V)\text{ and }x \in V \setminus U.
$$
$$
\text{There exists a cycle }M \in \mathcal{M}\text{ such that }x \in M \subseteq U \cup \{x\}
\;\Longleftrightarrow\;
r(U) = r(U \cup \{x\}).
$$
