\text{For any graph } $\Gamma$ \text{ with girth } $\gamma$ \text{ and imbedding index } $\iota$,
$$
\rho < H(\gamma, \iota),
$$
\text{where } $H(\gamma, \iota)$ \text{ is a function of the girth } $\gamma$ \text{ and imbedding index } $\iota$ \text{ that satisfies:}
\begin{aligned}
H(\gamma, 0) = H(\gamma, 1) &= \{G(\gamma)\}, \\
H(\gamma, 2) &= [G(\gamma) + 1], \\
H(\gamma, \iota) &= \text{the least } H_0 \in \mathbb{N} \text{ such that } G(\gamma)\left(1 + \frac{\iota - 2}{H_0 + 1}\right) < H_0 \text{ for } \iota > 2.
\end{aligned}
