\text{Define } g: V \rightarrow \mathbb{Q} \text{ by}

$$
g(x) = \begin{cases}
  -1 & \text{if } \rho(x) = 5, \\
   0 & \text{if } \rho(x) = 6, \\
   \displaystyle\sum_{u \in U(x)} \frac{1}{n(u)} & \text{if } \rho(x) \geq 7.
\end{cases}
$$

\text{where } $U(x)$ \text{ denotes the set of 5-valent vertices adjacent to } $x$ \text{, and } $n(u)$ \text{ denotes the number of vertices incident with } $u$ \text{ having valence at least 7.}
