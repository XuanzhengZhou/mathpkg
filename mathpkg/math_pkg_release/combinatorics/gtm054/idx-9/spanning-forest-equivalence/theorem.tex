Let $F \subseteq E$. The following statements are equivalent:
\begin{enumerate}
\item[(a)] $|F| = \dim(\mathcal{Z}(\Gamma))$ and $\dim(\mathcal{Z}(\Gamma_{(F)})) = 0$;
\item[(b)] $\Gamma_{(F)}$ is a spanning forest;
\item[(c)] $F$ is a minimal subset of $E$ which meets every nonempty cocycle;
\item[(d)] $F$ is a maximal subset of $E$ which contains no nonempty cycle;
\item[(e)] $\mathcal{Z}^\perp(\Gamma)$ admits a basis $\{C_e : e \in F\}$ such that $C_e \cap F = \{e\}$ for each $e \in F$.
\end{enumerate}
