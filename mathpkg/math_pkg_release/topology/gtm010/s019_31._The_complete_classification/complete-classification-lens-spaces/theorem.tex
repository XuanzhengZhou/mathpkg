For lens spaces $L = L(p; q_1, \dots, q_n)$ and $L' = L(p; q'_1, \dots, q'_n)$:
\begin{enumerate}
\item $L \simeq L'$ (homotopy equivalent) $\iff$ $q_1 \dots q_n \equiv \pm a^n q'_1 \dots q'_n \pmod{p}$ for some $a$ with $(a, p) = 1$.
\item $L$ and $L'$ are simple-homotopy equivalent $\iff$ $(q_1, \dots, q_n)$ is a permutation of $(\pm a q'_1, \dots, \pm a q'_n)$ modulo $p$ for some $a$ with $(a, p) = 1$.
\item The torsion invariant $\tau(L)$ completely determines the simple-homotopy type.
\end{enumerate}
