% Proposition 1.5: Simultaneous Local Infinitesimal Stability
% From GTM014, Chapter V, §1
%
% STATEMENT ONLY — no proof text.

\begin{proposition}
Let $f: X \to Y$ be smooth, let $q \in Y$, and let $S = \{p_1,\ldots,p_k\} \subset f^{-1}(q)$.
Set $m = \dim Y$.
Then $f$ is simultaneously locally infinitesimally stable at $p_1,\ldots,p_k$ if and only if
\[
J^m(f^*TY)_S = (df)\, J^m(TX)_S + f^*\, J^m(TY)_q,
\]
where $J^m(f^*TY)_S = \bigoplus_{i=1}^{k} J^m(f^*TY)_{p_i}$,
$J^m(TX)_S = \bigoplus_{i=1}^{k} J^m(TX)_{p_i}$,
and $(df)$ and $f^*$ act coordinatewise.
\end{proposition}
