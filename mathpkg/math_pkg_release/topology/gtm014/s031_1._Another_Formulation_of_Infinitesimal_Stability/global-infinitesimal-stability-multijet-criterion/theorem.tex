% Theorem 1.6: Global Infinitesimal Stability via Multijet Criterion
% From GTM014, Chapter V, §1
%
% STATEMENT ONLY — no proof text.

\begin{theorem}
Let $f: X \to Y$ be a smooth map with $X$ compact, and let $m = \dim Y$.
Then $f$ is infinitesimally stable if and only if the following condition $(\dagger)$ holds:
\begin{quote}
For every $q \in Y$ and every finite subset $S \subset f^{-1}(q)$ with $|S| \leq m+1$,
\[
J^m(f^*TY)_S = (df)\bigl(J^m(TX)_S\bigr) + f^*\bigl(J^m(TY)_q\bigr),
\]
where $J^m(f^*TY)_S = \bigoplus_{p \in S} J^m(f^*TY)_p$ and similarly for $J^m(TX)_S$.
\end{quote}
\end{theorem}
