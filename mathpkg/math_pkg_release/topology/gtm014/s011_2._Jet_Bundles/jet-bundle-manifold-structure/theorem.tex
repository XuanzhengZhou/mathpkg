Let $X$ and $Y$ be smooth manifolds of dimensions $n$ and $m$ respectively, and let $k \geq 0$ be an integer.

\begin{enumerate}
\item $J^k(X, Y)$ is a smooth manifold. With the smooth structure given by the charts $\tau_{U,V}$ (Lemma 2.6), the source map $\alpha: J^k(X, Y) \to X$ and the target map $\beta: J^k(X, Y) \to Y$ are smooth submersions.

\item $J^k(X, Y)$ is a fiber bundle over $X \times Y$ with projection $\alpha \times \beta: J^k(X, Y) \to X \times Y$. The fiber over $(p, q)$ is $J^k(X, Y)_{p,q}$, which is diffeomorphic to $B_{n,m}^k = \bigoplus_{i=1}^m A_n^k$, where $A_n^k$ is the vector space of polynomials in $n$ variables of degree $\leq k$ with zero constant term.
\end{enumerate}
