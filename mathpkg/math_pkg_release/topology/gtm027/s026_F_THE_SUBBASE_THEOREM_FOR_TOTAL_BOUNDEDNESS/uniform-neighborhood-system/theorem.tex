\text{A uniform neighborhood system for a set } X \text{ is a pair } (V, \geq) \text{ where } (A, \geq) \text{ is a directed set and } V \text{ assigns to each } a \in A \text{ a function } V_a : X \to \mathcal{P}(X) \text{ satisfying:}
\end{definition}

\begin{enumerate}
\item x \in V_a(x) \text{ for all } a \in A, x \in X
\item \text{For each } a \in A \text{ there is } b \in A \text{ with } y \in V_a(x) \text{ whenever } x \in V_b(y)
\item \text{For each } a \in A \text{ there exist } b, c \in A \text{ such that } z \in V_a(x) \text{ whenever } y \in V_b(x) \text{ and } z \in V_c(y)
\item \text{For each } a \in A \text{ there is } b \in A \text{ such that } z \in V_a(x) \text{ whenever } y \in V_b(x) \text{ and } z \in V_b(y)
\end{enumerate}
