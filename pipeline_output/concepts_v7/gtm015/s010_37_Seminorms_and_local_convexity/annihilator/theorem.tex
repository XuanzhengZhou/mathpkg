Let $B: E \times F \to \mathbb{K}$ be a duality.
\begin{enumerate}
\item[(i)] If $S \subset E$, the annihilator of $S$ (in $F$), denoted $S^\perp$, is the set of all $y \in F$ such that $B(s, y) = 0$ for all $s \in S$.
\item[(ii)] If $T \subset F$, the annihilator of $T$ (in $E$), denoted $^\perp T$, is the set of all $x \in E$ such that $B(x, t) = 0$ for all $t \in T$.
\end{enumerate}
