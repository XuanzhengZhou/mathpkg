Let $B: E \times F \to \mathbb{K}$ be a duality, and let $S, S_1, S_2 \subset E$ and $T, T_1, T_2 \subset F$. Then:
\begin{enumerate}
\item[(i)] $S_1 \subset S_2 \implies S_2^\perp \subset S_1^\perp$ and $T_1 \subset T_2 \implies {}^\perp T_2 \subset {}^\perp T_1$;
\item[(ii)] $S \subset {}^\perp(S^\perp)$ and $T \subset ({}^\perp T)^\perp$;
\item[(iii)] $S^\perp = ({}^\perp(S^\perp))^\perp$ and ${}^\perp T = {}^\perp(({}^\perp T)^\perp)$;
\item[(iv)] $S^\perp$ is a $\sigma(F,E)$-closed linear subspace of $F$, and ${}^\perp T$ is a $\sigma(E,F)$-closed linear subspace of $E$;
\item[(v)] If $M$ is a $\sigma(E,F)$-closed linear subspace of $E$, then $M = {}^\perp(M^\perp)$; dually, if $N$ is a $\sigma(F,E)$-closed linear subspace of $F$, then $N = ({}^\perp N)^\perp$;
\item[(vi)] ${}^\perp(S^\perp)$ is the $\sigma(E,F)$-closed linear subspace generated by $S$; $(^\perp T)^\perp$ is the $\sigma(F,E)$-closed linear subspace generated by $T$.
\end{enumerate}
