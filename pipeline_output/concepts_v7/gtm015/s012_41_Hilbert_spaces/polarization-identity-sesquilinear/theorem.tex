Let $\varphi$ be a sesquilinear form on a complex vector space $E$ such that $\varphi(x, x) \geq 0$ for all $x \in E$. Then:

\begin{enumerate}
\item[(i)] \textbf{Polarization identity (real part).} For all $x, y \in E$,
$$\varphi(x, y) + \varphi(y, x) = \frac{1}{2}\Bigl[\varphi(x + y, x + y) - \varphi(x - y, x - y)\Bigr].$$

\item[(ii)] \textbf{Polarization identity (imaginary part).} For all $x, y \in E$,
$$\varphi(x, y) - \varphi(y, x) = \frac{i}{2}\Bigl[\varphi(x - iy, x - iy) - \varphi(x + iy, x + iy)\Bigr].$$

\item[(iii)] \textbf{Cauchy--Schwarz inequality.} For all $x, y \in E$,
$$|\varphi(x, y)|^2 \leq \varphi(x, x)\,\varphi(y, y).$$

\item[(iv)] \textbf{Hermiticity criterion.} $\varphi$ is Hermitian if and only if $\varphi(x, x)$ is real for all $x \in E$.

\item[(v)] \textbf{Positive Hermitian case.} If $\varphi$ is positive (i.e., $\varphi(x, x) \geq 0$ for all $x$) and Hermitian, then the Cauchy--Schwarz inequality $|\varphi(x, y)|^2 \leq \varphi(x, x)\,\varphi(y, y)$ holds.
\end{enumerate}
