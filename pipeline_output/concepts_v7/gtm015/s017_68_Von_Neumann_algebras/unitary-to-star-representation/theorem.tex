\begin{theorem}
Let $t \mapsto U_t$ be a continuous unitary representation of $G$ on a Hilbert space $H$. Then:
\begin{enumerate}
\item[(1)] There exists a unique $*$-representation $\varphi: L^1(G) \to \mathcal{L}(H)$ such that
$$(\varphi(u)x \mid y) = \int f(t)(U_t x \mid y)\, dt$$
for all $u = [f] \in L^1(G)$ and $x, y \in H$.
\item[(2)] $\varphi$ is nondegenerate.
\item[(3)] $\varphi(u_s) = U_s \varphi(u)$ for all $u \in L^1(G)$ and $s \in G$.
\item[(4)] If $(u_j)$ is any approximate identity in $L^1(G)$, then $\varphi((u_j)_t) \to U_t$ strongly for each $t \in G$.
\end{enumerate}
\end{theorem}
