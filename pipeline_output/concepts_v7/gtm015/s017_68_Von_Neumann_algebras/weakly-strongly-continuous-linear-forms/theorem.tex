\begin{proposition}
Let $M$ be a linear subspace of $\mathcal{L}(H)$ and let $f$ be a linear form on $M$. The following conditions on $f$ are equivalent:
\begin{enumerate}
\item[(a)] $f$ is weakly continuous;
\item[(b)] $f$ is strongly continuous;
\item[(c)] there exist vectors $x_1, \ldots, x_n, y_1, \ldots, y_n$ in $H$ such that
$$f(T) = \sum_{i=1}^{n} (Tx_i \mid y_i)$$
for all $T \in M$.
\end{enumerate}
\end{proposition}
