\begin{lemma}
Let $K = H \oplus H \oplus \cdots$ be the orthogonal sum of $\aleph_0$ copies of $H$, and let $U_j : H_j \to K$ be the natural injections. For $P \in \mathcal{L}(K)$, define $P_{ij} \in \mathcal{L}(H)$ by $P_{ij} x_j = U_i^* P U_j x_j$ for $x_j \in H_j$. Then for $T \in \mathcal{L}(H)$ and $P \in \mathcal{L}(K)$,
$$\tilde{T}P = P\tilde{T} \quad \text{if and only if} \quad T P_{ij} = P_{ij} T \quad \text{for all } i,j.$$
\end{lemma}
