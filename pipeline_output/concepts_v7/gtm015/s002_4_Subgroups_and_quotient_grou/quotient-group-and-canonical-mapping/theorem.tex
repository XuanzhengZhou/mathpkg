\text{Let } H \text{ be a subgroup of a group } G.
\begin{enumerate}
\item G/H \text{ denotes the set of all left cosets } xH \text{ (i.e., the set of all equivalence classes for the equivalence relation } x R y \text{ defined by } x^{-1}y \in H\text{).}
\item \text{The canonical mapping } \pi: G \to G/H \text{ is defined by } \pi(x) = xH.
\item H\backslash G \text{ denotes the set of all right cosets } Hx.
\item \text{When } H \text{ is normal in } G, \text{ i.e., when } xH = Hx \text{ for all } x \in G, \text{ we need not distinguish between } G/H \text{ and } H\backslash G; \text{ for definiteness, we use the notation } G/H.
\end{enumerate}
