\text{If } E \text{ and } F \text{ are normed spaces over } \mathbb{K} \text{ and } T: E \to F \text{ is a linear mapping, then the following conditions are equivalent:}

\begin{enumerate}
\item[(a)] T \text{ is continuous;}
\item[(b)] \text{there exists a constant } K \geq 0 \text{ such that } \|Tx\| \leq K\|x\| \text{ for all } x \in E;
\item[(c)] \{\|Tx\| : \|x\| = 1\} \text{ is bounded;}
\item[(d)] \{\|Tx\| : \|x\| < 1\} \text{ is bounded;}
\item[(e)] \{\|Tx\| : \|x\| \leq 1\} \text{ is bounded.}
\end{enumerate}

\text{In this case, if } M_1, M_2, M_3 \text{ are the suprema of the sets in (c), (d), (e), then } M_1 = M_2 = M_3. \text{ Writing } M \text{ for their common value, one has } \|Tx\| \leq M\|x\| \text{ for all } x \in E, \text{ and } M \text{ is minimal among all such constants.}
