\text{Let } E \text{ and } F \text{ be normed spaces over } \mathbb{K}, \text{ and let } S, T \in \mathcal{L}(E, F) \text{ and } \lambda \in \mathbb{K}. \text{ Then:}

\begin{enumerate}
\item[(i)] (S + T)' = S' + T'.
\item[(ii)] (\lambda T)' = \lambda T'.
\item[(iii)] T''x'' = (Tx)'' \text{ for all } x \in E.
\item[(iv)] \|T'\| = \|T\|.
\item[(v)] \text{If } T \text{ is isometric and surjective, then so is } T'.
\end{enumerate}
