\text{Let } E \text{ and } F \text{ be normed spaces over } \mathbb{K}.

\begin{enumerate}
\item[(iv)] \text{The strong operator topology on } \mathcal{L}(E, F) \text{ is locally convex and separated; a basic neighborhood of } 0 \text{ is given by}
$$\{T \in \mathcal{L}(E, F) : \|Tx_i\| \leq 1 \ (i = 1, \ldots, n)\},$$
\text{where } \{x_1, \ldots, x_n\} \text{ is any finite subset of } E.

\item[(v)] \text{The uniform operator topology } \tau_n \text{ on } \mathcal{L}(E, F) \text{ is finer than the strong operator topology } \tau_s.

\item[(vi)] \text{Assuming } F \neq \{\theta\}, \text{ one has } \tau_n = \tau_s \text{ if and only if } E \text{ is finite-dimensional.}
\end{enumerate}
