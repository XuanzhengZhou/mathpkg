Suppose $u_n \in [\mathcal{M}]$ ($n = 1, 2, 3, \ldots$) and $u \in [\mathcal{M}]$; say $u_n = [f_n]$ and $u = [f]$. Then
\begin{enumerate}
\item[(i)] $u_n \to u$ for the metric $d(u, v) = \int \frac{|f - g|}{1 + |f - g|} \, d\mu$ if and only if $f_n \to f$ in measure;
\item[(ii)] $(u_n)$ is Cauchy for $d$ if and only if $(f_n)$ is fundamental in measure.
\end{enumerate}
