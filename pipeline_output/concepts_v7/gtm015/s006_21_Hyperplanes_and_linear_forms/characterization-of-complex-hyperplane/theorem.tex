Let $E$ be a vector space over $\mathbb{C}$ and let $H \subset E$. The following conditions are equivalent:
\begin{enumerate}
\item[(a)] $H$ is a $\mathbb{C}$-hyperplane in $E$, i.e., $H = \{x \in E : f(x) = \lambda\}$ where $f$ is a nonzero $\mathbb{C}$-form and $\lambda \in \mathbb{C}$;
\item[(b)] there exist a maximal $\mathbb{R}$-linear subspace $M_0$ of $E$ and vectors $x_0, y_0 \in E$ such that
$$H = (x_0 + M_0) \cap (y_0 + iM_0);$$
\item[(c)] there exist a maximal $\mathbb{R}$-linear subspace $M_0$ of $E$ and a vector $z_0 \in E$ such that
$$H = z_0 + (M_0 \cap iM_0).$$
\end{enumerate}
