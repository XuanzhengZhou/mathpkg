Let $E$ be a vector space over $\mathbb{C}$. If $f: E \to \mathbb{C}$ is a $\mathbb{C}$-form, define $g, h: E \to \mathbb{R}$ by
$$g(x) = \operatorname{Re} f(x), \quad h(x) = \operatorname{Im} f(x) \qquad (x \in E).$$
Then:
\begin{enumerate}
\item[(i)] $g$ and $h$ are $\mathbb{R}$-forms on $E$, and $f = g + i h$;
\item[(ii)] $g(ix) = -h(x)$ and $h(ix) = g(x)$ for all $x \in E$;
\item[(iii)] $f$ is determined by its real part $g$ via $f(x) = g(x) - i g(ix)$ for all $x \in E$.
\end{enumerate}
