\textbf{Theorem 7.} Let $f$ be a function, let $A$ and $B$ be sets, and let $\{X_c : c \in C\}$ be a family of sets for a non-void index set $C$. Then:
\begin{enumerate}
\item[(a)] $f^{-1}[A \sim B] = f^{-1}[A] \sim f^{-1}[B]$.
\item[(b)] $f^{-1}[A \cup B] = f^{-1}[A] \cup f^{-1}[B]$.
\item[(c)] $f^{-1}[A \cap B] = f^{-1}[A] \cap f^{-1}[B]$.
\item[(d)] $f^{-1}\left[\bigcup\{X_c : c \in C\}\right] = \bigcup\{f^{-1}[X_c] : c \in C\}$.
\item[(e)] $f^{-1}\left[\bigcap\{X_c : c \in C\}\right] = \bigcap\{f^{-1}[X_c] : c \in C\}$.
\end{enumerate}
