Suppose $(z_n)_{n=1}^{\infty}$ and $(w_n)_{n=1}^{\infty}$ are sequences in $\mathbb{C}$ with $z_n \to z$, $w_n \to w$. Then:
\begin{itemize}
\item[(a)] $(z_n + w_n) \to z + w$.
\item[(b)] $a z_n \to a z$ for any $a \in \mathbb{C}$.
\item[(c)] $(z_n)_{n=1}^{\infty}$ is bounded: there exists $M \geq 0$ such that $|z_n| \leq M$ for all $n$.
\item[(d)] $(z_n w_n) \to z w$.
\item[(e)] If $z \neq 0$, then $z_n^{-1} \to z^{-1}$.
\end{itemize}
