\textbf{Axioms of addition:}
\begin{itemize}
\item[A1.] $(x + y) + z = x + (y + z)$ for any $x, y, z \in \mathbb{R}$.
\item[A2.] $x + y = y + x$ for any $x, y \in \mathbb{R}$.
\item[A3.] There is an element $0 \in \mathbb{R}$ such that $x + 0 = x$ for every $x \in \mathbb{R}$.
\item[A4.] For each $x \in \mathbb{R}$ there is an element $-x \in \mathbb{R}$ such that $x + (-x) = 0$.
\end{itemize}

\textbf{Axioms of multiplication:}
\begin{itemize}
\item[M1.] $(xy)z = x(yz)$ for any $x, y, z \in \mathbb{R}$.
\item[M2.] $xy = yx$ for any $x, y \in \mathbb{R}$.
\item[M3.] There is an element $1 \in \mathbb{R}$, $1 \neq 0$, such that $x \cdot 1 = x$ for every $x \in \mathbb{R}$.
\item[M4.] For each $x \in \mathbb{R}$, $x \neq 0$, there is an element $x^{-1} \in \mathbb{R}$ such that $x \cdot x^{-1} = 1$.
\end{itemize}

\textbf{Distributive law:}
\begin{itemize}
\item[DL.] $x(y + z) = xy + xz$ for any $x, y, z \in \mathbb{R}$.
\end{itemize}
