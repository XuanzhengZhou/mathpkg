Let $E: \mathbb{R} \to \mathbb{R}$ be defined by
$$E(x) = \sum_{n=0}^{\infty} \frac{x^n}{n!}.$$
Then $E$ satisfies the following properties:
\begin{enumerate}
\item[(a)] $E(x) > 0$ for all $x \in \mathbb{R}$.
\item[(b)] $E$ maps $\mathbb{R}$ bijectively onto $(0, \infty)$.
\item[(c)] $E(x+y) = E(x)E(y)$ for all $x, y \in \mathbb{R}$.
\item[(d)] $\displaystyle\lim_{x \to +\infty} E(x) = +\infty$ and $\displaystyle\lim_{x \to -\infty} E(x) = 0$.
\end{enumerate}
Moreover, $E(nx) = E(x)^n$ for all $x \in \mathbb{R}$ and $n = 0, \pm 1, \pm 2, \dots$, and consequently
$$e = E(1), \quad e^{1/n} = E(1/n), \quad e^{m/n} = E(m/n).$$
