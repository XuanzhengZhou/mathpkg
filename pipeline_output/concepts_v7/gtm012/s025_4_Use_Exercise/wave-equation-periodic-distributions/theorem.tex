\begin{theorem}
For each $G, H \in \mathcal{P}'$ there is a unique family $(F_t)_{t > 0} \subset \mathcal{P}'$ with the following properties:
\begin{enumerate}
  \item $\displaystyle \frac{d}{dt} F_t\Big|_s$ exists for each $s > 0$,
  \item $\displaystyle \left(\frac{d}{dt}\right)^2 F_t\Big|_s = D^2 F_s$ for all $s > 0$,
  \item $F_t \to G$ in $\mathcal{P}'$ as $t \to 0$,
  \item $\displaystyle \frac{d}{dt} F_t\Big|_s \to H$ in $\mathcal{P}'$ as $s \to 0$.
\end{enumerate}
\end{theorem}
