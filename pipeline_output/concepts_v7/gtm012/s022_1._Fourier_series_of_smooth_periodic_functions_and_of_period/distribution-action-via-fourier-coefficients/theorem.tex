\begin{theorem}
Suppose $F \in \mathcal{P}'$ has $(a_n)_{-\infty}^{\infty}$ as its sequence of Fourier coefficients, and suppose $u \in \mathcal{P}$ has $(b_n)_{-\infty}^{\infty}$ as its sequence of Fourier coefficients. Then
$$F(u) = \sum_{-\infty}^{\infty} a_n b_{-n} = \sum_{-\infty}^{\infty} a_{-n} b_n.$$
\end{theorem}
