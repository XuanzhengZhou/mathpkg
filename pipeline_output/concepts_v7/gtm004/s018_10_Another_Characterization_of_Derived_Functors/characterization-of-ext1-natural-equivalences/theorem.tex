\begin{theorem}\label{thm:10.4}
For a homomorphism $\alpha \colon A' \to A$ of $\Lambda$-modules, the induced natural transformation
$$
\alpha^* \colon \operatorname{Ext}_\Lambda^1(A, -) \longrightarrow \operatorname{Ext}_\Lambda^1(A', -)
$$
is a natural equivalence if and only if $\alpha$ can be factored as
$$
A' \xrightarrow{\iota} A \oplus P \xrightarrow{\pi_A} A,
$$
where $P$ is a projective $\Lambda$-module, $\iota$ is an isomorphism, and $\pi_A$ is the canonical projection onto the summand $A$.
\end{theorem}
