\begin{theorem}\label{thm:10.1}
Let $T \colon \mathfrak{M}_\Lambda \to \mathfrak{Ab}$ be an additive covariant functor and let $A$ be a $\Lambda$-module. Then there are natural isomorphisms
$$
\Gamma \colon [\operatorname{Ext}_\Lambda^q(A, -),\, T] \xrightarrow{\cong} \tilde{L}_q TA, \qquad q = 0, 1, \dots,
$$
where $[\operatorname{Ext}_\Lambda^q(A, -),\, T]$ denotes the set (or class) of natural transformations from $\operatorname{Ext}_\Lambda^q(A, -)$ to $T$, and $\tilde{L}_q T$ denotes the $q$-th left satellite of $T$.
\end{theorem}
