\text{Let } A \text{ be an abelian category with enough projectives. Then for any objects } A_i, B, A, B_j \text{ in } A:

\begin{enumerate}
\item[(i)] $$\text{Ext}_A\left(\bigoplus_i A_i, B\right) \cong \prod_i \text{Ext}_A(A_i, B)$$
\item[(ii)] $$\text{Ext}_A\left(A, \prod_j B_j\right) \cong \prod_j \text{Ext}_A(A, B_j)$$
\end{enumerate}
