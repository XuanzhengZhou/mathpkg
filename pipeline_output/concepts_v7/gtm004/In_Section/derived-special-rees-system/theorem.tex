\textbf{Proposition 6.5.} If the Rees system (6.9)
\begin{align*}
D &\xrightarrow{\alpha} D \xrightarrow{\beta} E \xrightarrow{\gamma} D, \\
\bar{D} &\xrightarrow{\bar{\alpha}} \bar{D} \xrightarrow{\bar{\beta}} E \xrightarrow{\bar{\gamma}} \bar{D}, \\
\bar{D} &\xrightarrow{\xi} D \xrightarrow{\varphi} \Gamma \xrightarrow{\bar{\varphi}} \bar{D} \xrightarrow{\xi} D
\end{align*}
is special with automorphism $\theta : \Gamma \to \Gamma$ satisfying
\begin{align*}
\theta \varphi &= \varphi \alpha, \\
\bar{\varphi} \theta &= \bar{\alpha} \bar{\varphi}, \\
\bar{\varphi} \theta^{-1} \varphi &= \bar{\gamma} \beta,
\end{align*}
then the derived Rees system (6.16)
\begin{align*}
D_1 &\xrightarrow{\alpha_1} D_1 \xrightarrow{\beta_1} E_1 \xrightarrow{\gamma_1} D_1, \\
\bar{D}_1 &\xrightarrow{\bar{\alpha}_1} \bar{D}_1 \xrightarrow{\bar{\beta}_1} E_1 \xrightarrow{\bar{\gamma}_1} \bar{D}_1, \\
\bar{D}_1 &\xrightarrow{\xi_1} D_1 \xrightarrow{\varphi_1} \Gamma \xrightarrow{\bar{\varphi}_1} \bar{D}_1 \xrightarrow{\xi_1} D_1
\end{align*}
is special with the same automorphism $\theta : \Gamma \to \Gamma$.
