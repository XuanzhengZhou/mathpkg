\text{Let}~C : C^0 \rightarrow C^1 \rightarrow \cdots \rightarrow C^n \rightarrow C^{n+1} \rightarrow \cdots~\text{be a positive acyclic cochain complex, and let}~D : D^0 \rightarrow D^1 \rightarrow \cdots \rightarrow D^n \rightarrow D^{n+1} \rightarrow \cdots~\text{be a positive cochain complex with each}~D^n~\text{injective. Then:}

\begin{enumerate}
\item \text{To every homomorphism}~\varphi : H^0(C) \rightarrow H^0(D)~\text{there exists a cochain map}~\varphi : C \rightarrow D~\text{inducing}~\varphi~\text{in cohomology.}
\item \text{Any two cochain maps inducing the same}~\varphi~\text{are homotopic.}
\end{enumerate}
