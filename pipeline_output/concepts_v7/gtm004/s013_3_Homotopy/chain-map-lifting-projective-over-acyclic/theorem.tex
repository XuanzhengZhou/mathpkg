\text{Let}~C : \cdots \rightarrow C_n \rightarrow C_{n-1} \rightarrow \cdots \rightarrow C_0~\text{be a positive chain complex with each}~C_n~\text{projective, and let}~D : \cdots \rightarrow D_n \rightarrow D_{n-1} \rightarrow \cdots \rightarrow D_0~\text{be a positive acyclic chain complex. Then:}

\begin{enumerate}
\item \text{To every homomorphism}~\varphi : H_0(C) \rightarrow H_0(D)~\text{there exists a chain map}~\varphi : C \rightarrow D~\text{inducing}~\varphi~\text{in homology.}
\item \text{Any two chain maps inducing the same}~\varphi~\text{are homotopic.}
\end{enumerate}
