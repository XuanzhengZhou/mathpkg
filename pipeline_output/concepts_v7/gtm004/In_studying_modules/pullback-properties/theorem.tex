\text{If the square}
$$
\begin{CD}
Y @>{\alpha}>> A \\
@V{\beta}VV @VV{\varphi}V \\
B @>>{\psi}> X
\end{CD}
$$
\text{is a pull-back diagram, then:}
\begin{enumerate}
\item[(i)] \text{$\beta$ induces a homomorphism $\ker \alpha \to \ker \psi$;}
\item[(ii)] \text{if $\psi$ is an epimorphism, then $\alpha$ is also an epimorphism.}
\end{enumerate}
