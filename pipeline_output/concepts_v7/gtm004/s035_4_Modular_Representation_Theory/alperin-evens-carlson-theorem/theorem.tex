\begin{theorem}[Alperin--Evens; Carlson]
\label{thm:alperin-evens-carlson}
Let $G$ be a finite group, $k$ a field of characteristic $p$, and $M$ an indecomposable $kG$-module. Then
$$
c_G(M) = \max\left\{ c_E(M_E) \mid E \text{ is an elementary abelian $p$-subgroup of } G \right\},
$$
where $M_E$ denotes the restriction of $M$ to $kE$.
\end{theorem}
