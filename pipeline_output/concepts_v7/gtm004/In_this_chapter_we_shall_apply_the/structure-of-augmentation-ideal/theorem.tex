\begin{lemma}[1.2]
\leavevmode
\begin{enumerate}
  \item[(i)] As an abelian group, $IG$ is free on the set
    $$W = \{\,x - 1 \mid 1 \neq x \in G\,\}.$$
  \item[(ii)] Let $S$ be a generating set for $G$. Then, as a $G$-module, $IG$ is generated by $S - 1 = \{\,s - 1 \mid s \in S\,\}$.
\end{enumerate}
\end{lemma}
