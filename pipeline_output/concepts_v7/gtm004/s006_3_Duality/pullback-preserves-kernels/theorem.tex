Let the square
$$
\begin{array}{ccc}
Y & \xrightarrow{\alpha} & A \\
{\scriptstyle\beta}\downarrow & & \downarrow{\scriptstyle\varphi} \\
B & \xrightarrow{\psi} & X
\end{array}
$$
be a pull-back diagram in a category $\mathfrak{C}$ with zero object. Then:
\begin{enumerate}
\item[(i)] If $(J, \mu)$ is the kernel of $\beta$, then $(J, \alpha \mu)$ is the kernel of $\varphi$.
\item[(ii)] If $(J, \nu)$ is the kernel of $\varphi$, then $\nu = \alpha \mu$ where $(J, \mu)$ is the kernel of $\beta$.
\end{enumerate}
