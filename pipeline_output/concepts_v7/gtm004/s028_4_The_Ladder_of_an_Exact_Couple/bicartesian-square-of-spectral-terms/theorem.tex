\begin{theorem}[Theorem 4.6]
Applying the $Q^{\alpha_n}$- and $Q_{\sigma_m}$-processes to the ladder (4.17), the square
\[
\begin{CD}
E_{m,n} @>{\sigma_{m,n}}>> E_{m,n+1} \\
@V{\varrho_{m,n}}VV @VV{\varrho_{m+1,n}}V \\
E_{m+1,n} @>{\sigma_{m+1,n}}>> E_{m+1,n+1}
\end{CD}
\]
is bicartesian (i.e., simultaneously a pullback and a pushout). In particular, this square encodes how the $E_{m,n}$ terms are related under the vertical and horizontal structure maps $\varrho_{m,n}$ and $\sigma_{m,n}$ arising from the ladder construction.
\end{theorem}
