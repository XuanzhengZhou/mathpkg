\begin{definition}
For an exact couple $(D, E; \alpha, \beta, \gamma)$ in an abelian category admitting suitable limits, define the following limit terms:
\[
E_{m,\infty} = \lim_{n \to \infty} E_{m,n} = \frac{\gamma^{-1}(\alpha^{\infty} D)}{\beta\alpha^{-m}(0)},
\tag{4.22}
\]
\[
E_{\infty} = \lim_{n \to \infty} \lim_{m \to \infty} E_{m,n} = \frac{\gamma^{-1}(\alpha^{\infty} D)}{\beta\alpha^{-\infty}(0)},
\tag{4.23}
\]
where $\alpha^{\infty}D = \bigcap_n \alpha^n D$ and $\alpha^{-\infty}(0) = \bigcup_m \alpha^{-m}(0)$ are defined as in (4.24), and $\alpha^{-m}(0) = \ker(\alpha^m)$. More generally,
\[
E_{\infty} = \lim_{n} \lim_{m} (E_{m,n}; \varrho_{m,n}, \sigma_{m,n}).
\tag{4.21}
\]
In a concrete category (e.g., modules), $E_{m,\infty} = \bigcap_n E_{m,n}$.
\end{definition}
