\begin{proposition}[Proposition 4.2]
In the ladder construction, the square
\[
\begin{CD}
B_{0,1} @>{\sigma_{0,1}}>> B_1 \\
@V{\varrho_{0,1}}VV @VV{\sigma_{1,0}}V \\
B @>{\sigma_{1,0}}>> B_{1,0}
\end{CD}
\]
is bicartesian; that is, it is simultaneously a pullback and a pushout. Equivalently, this square may be written as
\[
\begin{CD}
\gamma^{-1}(C_1) @>>> \gamma^{-1}(C_1)/\beta\sigma^{-1}(0) \\
@VVV @VVV \\
B @>>> B/\beta\sigma^{-1}(0)
\end{CD}
\]
with the obvious morphisms, which is plainly bicartesian.
\end{proposition}
