Let $\xi_n$, $n \geq 1$, be random variables.
\begin{enumerate}
\item[(a)] If $\xi_n \geq \eta$ for all $n \geq 1$ and $E\eta > -\infty$, then
  $$E \liminf \xi_n \leq \liminf E \xi_n.$$
\item[(b)] If $\xi_n \leq \zeta$ for all $n \geq 1$ and $E\zeta < \infty$, then
  $$\limsup E \xi_n \leq E \limsup \xi_n.$$
\item[(c)] If $|\xi_n| \leq \eta$ for all $n \geq 1$ and $E\eta < \infty$, then
  $$E \liminf \xi_n \leq \liminf E \xi_n \leq \limsup E \xi_n \leq E \limsup \xi_n.$$
\end{enumerate}
