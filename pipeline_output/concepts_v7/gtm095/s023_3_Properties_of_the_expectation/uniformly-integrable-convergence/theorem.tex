Let $\{\xi_n\}_{n \geq 1}$ be a uniformly integrable family of random variables. Then
\begin{enumerate}
\item[(a)] $E \liminf \xi_n \leq \liminf E\xi_n \leq \limsup E\xi_n \leq E \limsup \xi_n$.
\item[(b)] If in addition $\xi_n \to \xi$ (a.s.) then $\xi$ is integrable and
  $$E\xi_n \to E\xi, \quad E|\xi_n - \xi| \to 0 \quad \text{as } n \to \infty.$$
\end{enumerate}
