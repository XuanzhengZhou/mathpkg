Let $\{F_n\}$ be a sequence of distribution functions $F_n = F_n(x)$, $x \in \mathbb{R}$, and let $\{\varphi_n\}$ be the corresponding sequence of characteristic functions,
$$\varphi_n(t) = \int_{-\infty}^{\infty} e^{itx} \, dF_n(x), \quad t \in \mathbb{R}.$$

\begin{enumerate}
\item[(1)] If $F_n \xrightarrow{w} F$, where $F = F(x)$ is a distribution function, then $\varphi_n(t) \to \varphi(t)$ for all $t \in \mathbb{R}$, where $\varphi(t)$ is the characteristic function of $F$.

\item[(2)] If $\lim_n \varphi_n(t)$ exists for each $t \in \mathbb{R}$ and $\varphi(t) = \lim_n \varphi_n(t)$ is continuous at $t = 0$, then $\varphi(t)$ is the characteristic function of a probability distribution $F = F(x)$, and
$$F_n \xrightarrow{w} F.$$
\end{enumerate}
