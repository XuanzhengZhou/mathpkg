Let $A$ have $n$ elements and $B$ have $m$ elements. Then the number of:
\begin{itemize}
\item \text{Functions} $F: A \to B$ is $N(F) = m^n$.
\item \text{Injections} $I: A \to B$ is $N(I) = (m)_n = m(m-1)\cdots(m-n+1)$ \quad $(n \leq m)$.
\item \text{Surjections} $S: A \to B$ is $N(S) = \sum_{i=0}^{m} (-1)^i C_m^i (m-i)^n$ \quad $(n \geq m)$.
\item \text{Bijections} $B: A \to B$ is $N(B) = n!$ \quad $(n = m)$.
\end{itemize}
