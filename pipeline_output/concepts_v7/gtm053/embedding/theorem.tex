Let $A$ and $B$ be $L$-structures with domains $A$ and $B$ respectively. An \text{embedding} $h : A \rightarrow B$ is a map $A \rightarrow B$ such that:

\begin{enumerate}
\item For any $n$-ary relation symbol $p \in L$ and $a_1, \ldots, a_n \in A$,
$$A \vDash p(a_1, \ldots, a_n) \text{ iff } B \vDash p(h(a_1), \ldots, h(a_n)).$$

\item For any $n$-ary operation symbol $f \in L$ and $a_1, \ldots, a_n, a \in A$,
$$A \vDash f(a_1, \ldots, a_n) = a \text{ iff } B \vDash f(h(a_1), \ldots, h(a_n)) = h(a).$$

\item For any constant symbol $c \in L$ and $a \in A$,
$$A \vDash c^A = a \text{ iff } B \vDash c^B = h(a),$$
where $c^A$ denotes the interpretation of $c$ in $A$.
\end{enumerate}
