\textbf{Definition 1.1.} Let $k$ be any field.

\begin{enumerate}
\item[(1.1.1)] The set $\{(x_1, \ldots, x_n) \mid x_i \in k\}$ is called \textit{affine $n$-space over $k$}; we denote it by $k^n$, or by $k_{X_1, \ldots, X_n}$. Each $n$-tuple of $k^n$ is called a \textit{point}.

\item[(1.1.2)] Let $k[X_1, \ldots, X_n] = k[\mathbf{X}]$ be the ring of polynomials in $n$ indeterminants $X_1, \ldots, X_n$, with coefficients in $k$. Let $p(\mathbf{X}) \in k[\mathbf{X}] \setminus k$. The set
$$V(p) = \{(\mathbf{x}) \in k^n \mid p(\mathbf{x}) = 0\}$$
is called a \textit{hypersurface} of $k^n$, or an \textit{affine hypersurface}.

\item[(1.1.3)] If $\{p_\alpha(\mathbf{X})\}$ is any collection of polynomials in $k[\mathbf{X}]$, the set
$$V(\{p_\alpha\}) = \{(\mathbf{x}) \in k^n \mid \text{each } p_\alpha(\mathbf{x}) = 0\}$$
is called an \textit{algebraic variety} in $k^n$, an \textit{affine algebraic variety}, or, if the context is clear, just a \textit{variety}. If we wish to make explicit reference to the field $k$, we say affine variety over $k$, $k$-variety, etc.; $k$ is called the \textit{ground field}.

\item[(1.1.4)] $k^2$ is called the \textit{affine plane}. If $p \in k[X_1, X_2] \setminus k$, $V(p)$ is called a \textit{plane affine curve} (or plane curve, affine curve, curve, etc., if the meaning is clear from context).
\end{enumerate}
