Let $R \subset R^*$ be coordinate rings, and let $\pi : \mathcal{V}_{R^*} \to \mathcal{V}_R$ be the natural projection $\mathfrak{m}^* \mapsto \mathfrak{m}^* \cap R$.

\begin{enumerate}
\item[(10.8.1)] Let $\mathfrak{a} \subset R$ and $\mathfrak{a}^* \subset R^*$ be ideals. If $\mathfrak{a}$ defines $W \subset \mathcal{V}_R$ and $\mathfrak{a}^*$ defines $W^* \subset \mathcal{V}_{R^*}$, then
$$(\mathfrak{a}^* \text{ lies over } \mathfrak{a}) \implies (W^* \text{ lies over } W).$$

\item[(10.8.2)] Let $W \subset \mathcal{V}_R$ and $W^* \subset \mathcal{V}_{R^*}$ be varieties. If $W$ defines $\mathfrak{c} \subset R$ and $W^*$ defines $\mathfrak{c}^* \subset R^*$, then
$$(W^* \text{ lies over } W) \implies (\mathfrak{c}^* \text{ lies over } \mathfrak{c}).$$
\end{enumerate}
