The following conditions on modules over a ring $R$ are equivalent:
\begin{enumerate}
\item[(i)] $0 \to A \xrightarrow{\varphi} B \xrightarrow{\psi} C \to 0$ is a split exact sequence;
\item[(ii)] $0 \to \operatorname{Hom}_R(D,A) \xrightarrow{\bar{\varphi}} \operatorname{Hom}_R(D,B) \xrightarrow{\bar{\psi}} \operatorname{Hom}_R(D,C) \to 0$ is a split exact sequence for every $D$;
\item[(iii)] $0 \to \operatorname{Hom}_R(C,D) \xrightarrow{\bar{\psi}} \operatorname{Hom}_R(B,D) \xrightarrow{\bar{\varphi}} \operatorname{Hom}_R(A,D) \to 0$ is a split exact sequence for every $D$.
\end{enumerate}
