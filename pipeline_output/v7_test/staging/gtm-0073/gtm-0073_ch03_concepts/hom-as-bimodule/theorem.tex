Let $R$ and $S$ be rings and let ${}_R A_S$, ${}_R B$, ${}_R C_S$, ${}_R D$ be (bi)modules.
\begin{enumerate}
\item[(i)] $\operatorname{Hom}_R(A,B)$ is a right $S$-module, with $(fs)(a) = (f(a))s$.
\item[(ii)] If $\varphi : A \to A'$ is an $R$-module homomorphism, then $\bar{\varphi} : \operatorname{Hom}_R(A',B) \to \operatorname{Hom}_R(A,B)$ is a homomorphism of right $S$-modules.
\item[(iii)] $\operatorname{Hom}_R(C,D)$ is a left $S$-module, with $(sg)(c) = g(cs)$.
\item[(iv)] If $\psi : D \to D'$ is an $R$-module homomorphism, then $\bar{\psi} : \operatorname{Hom}_R(C,D) \to \operatorname{Hom}_R(C,D')$ is a homomorphism of left $S$-modules.
\end{enumerate}
