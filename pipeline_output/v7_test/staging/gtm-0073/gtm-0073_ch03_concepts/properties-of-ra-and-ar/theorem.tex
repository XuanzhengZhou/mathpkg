Let $r \in R$ and $p \in R$ prime.
\begin{enumerate}
\item[(i)] $rA$ and $A[r]$ are submodules of $A$.
\item[(ii)] $R/(p)$ is a field and $A[p]$ is a vector space over $R/(p)$.
\item[(iii)] $(R/(p^n))[p] \cong R/(p)$ and $p^m(R/(p^n)) \cong R/(p^{n-m})$ ($0 \leq m < n$).
\item[(iv)] If $A \cong \sum A_i$, then $rA \cong \sum rA_i$ and $A[r] \cong \sum A_i[r]$.
\item[(v)] If $f: A \to B$ is an isomorphism, then $f: A_t \cong B_t$ and $f: A(p) \cong B(p)$.
\end{enumerate}
