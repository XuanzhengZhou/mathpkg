Let $R$ be a ring and $\{A_i \mid i \in I\}$ a nonempty family of $R$-modules.
\begin{enumerate}
\item[(i)] $\prod_{i \in I} A_i$ is an $R$-module with $r\{a_i\} = \{ra_i\}$.
\item[(ii)] $\sum_{i \in I} A_i$ is a submodule of $\prod_{i \in I} A_i$.
\item[(iii)] For each $k \in I$, the canonical projection $\pi_k: \prod A_i \to A_k$ is an $R$-module epimorphism.
\item[(iv)] For each $k \in I$, the canonical injection $\iota_k: A_k \to \sum A_i$ is an $R$-module monomorphism.
\end{enumerate}
