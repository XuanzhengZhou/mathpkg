Let $A$ be a left module over an integral domain $R$ and $\mathcal{O}_a = \{r \in R \mid ra = 0\}$.
\begin{enumerate}
\item[(i)] $\mathcal{O}_a$ is an ideal of $R$.
\item[(ii)] $A_t = \{a \in A \mid \mathcal{O}_a \neq 0\}$ is the \emph{torsion submodule} of $A$.
\item[(iii)] $R/\mathcal{O}_a \cong Ra$.
\item[(iv)] If $R$ is a PID, $p \in R$ prime, and $p^i a = 0$, then $\mathcal{O}_a = (p^j)$ with $0 \leq j \leq i$.
\item[(v)] If $\mathcal{O}_a = (p^i)$, then $p^j a \neq 0$ for $0 \leq j < i$.
\end{enumerate}
