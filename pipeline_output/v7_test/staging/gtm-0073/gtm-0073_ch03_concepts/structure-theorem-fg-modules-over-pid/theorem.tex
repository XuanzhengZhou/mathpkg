Let $A$ be a finitely generated module over a PID $R$.
\begin{enumerate}
\item[(i)] $A$ is the direct sum of a free submodule $F$ of finite rank and cyclic torsion modules of orders $r_1,\ldots,r_t$ with $r_1 \mid r_2 \mid \cdots \mid r_t$ (\emph{invariant factors}). The rank of $F$ and the ideals $(r_i)$ are uniquely determined.
\item[(ii)] $A$ is the direct sum of a free submodule $E$ of finite rank and cyclic torsion modules of orders $p_1^{s_1},\ldots,p_k^{s_k}$ where $p_i$ are primes (\emph{elementary divisors}). The rank of $E$ and the ideals $(p_i^{s_i})$ are uniquely determined up to order.
\end{enumerate}
