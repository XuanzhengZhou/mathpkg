\begin{theorem}
For each $n \geq 3$, the dihedral group $D_n$ is a group of order $2n$ generated by elements $a$ and $b$ satisfying:
\begin{enumerate}
    \item[(i)] $a^n = (1)$; \quad $b^2 = (1)$;
    \item[(ii)] $bab^{-1} = a^{-1}$.
\end{enumerate}
Any group $G$ which is generated by elements $a,b \in G$ satisfying (i) and (ii) for some $n \geq 3$ (with $e \in G$ in place of (1)) is isomorphic to $D_n$.
\end{theorem}
