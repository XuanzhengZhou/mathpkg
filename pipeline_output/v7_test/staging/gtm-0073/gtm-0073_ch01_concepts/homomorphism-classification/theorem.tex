\begin{theorem}
Let $f: G \rightarrow H$ be a homomorphism of groups. Then:
\begin{enumerate}
    \item[(i)] $f$ is a monomorphism if and only if $\operatorname{Ker} f = \{e\}$;
    \item[(ii)] $f$ is an isomorphism if and only if there is a homomorphism $f^{-1}: H \rightarrow G$ such that $ff^{-1} = 1_H$ and $f^{-1}f = 1_G$.
\end{enumerate}
\end{theorem}
