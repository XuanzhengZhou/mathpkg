\begin{theorem}
Let $G = \langle a \rangle$ be a cyclic group.
\begin{enumerate}
    \item[(i)] If $G$ is infinite, then $a^k = e$ if and only if $k = 0$; and all powers $a^k$ are distinct.
    \item[(ii)] If $G$ is finite of order $m$, then $a^k = e$ if and only if $m \mid k$.
    \item[(iii)] If $G$ is finite of order $m$, then $|a^k| = m/(k,m)$ for each $k \in \mathbb{Z}$.
    \item[(iv)] $|a| = |\langle a \rangle|$.
\end{enumerate}
\end{theorem}
