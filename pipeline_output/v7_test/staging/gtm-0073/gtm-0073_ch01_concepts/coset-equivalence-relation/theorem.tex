\begin{theorem}
Let $H$ be a subgroup of a group $G$.
\begin{enumerate}
    \item[(i)] Right [resp.\ left] congruence modulo $H$ is an equivalence relation on $G$.
    \item[(ii)] The equivalence class of $a \in G$ under right [resp.\ left] congruence modulo $H$ is the set $Ha = \{ha \mid h \in H\}$ [resp.\ $aH = \{ah \mid h \in H\}$].
    \item[(iii)] $|Ha| = |H| = |aH|$ for all $a \in G$.
\end{enumerate}
\end{theorem}
