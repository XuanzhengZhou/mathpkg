\begin{theorem}
Let $K$ and $N$ be subgroups of a group $G$ with $N$ normal in $G$. Then
\begin{enumerate}
    \item[(i)] $N \cap K$ is a normal subgroup of $K$;
    \item[(ii)] $N$ is a normal subgroup of $N \vee K$;
    \item[(iii)] $NK = N \vee K = KN$;
    \item[(iv)] if $K$ is normal in $G$ and $K \cap N = \langle e \rangle$, then $nk = kn$ for all $k \in K$ and $n \in N$.
\end{enumerate}
\end{theorem}
