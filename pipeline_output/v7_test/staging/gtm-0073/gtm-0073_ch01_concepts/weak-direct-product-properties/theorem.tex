\begin{theorem}
If $\{G_i \mid i \in I\}$ is a family of groups, then
\begin{enumerate}
    \item[(i)] $\prod_{i \in I}^w G_i$ is a normal subgroup of $\prod_{i \in I} G_i$;
    \item[(ii)] for each $k \in I$, the map $\iota_k : G_k \rightarrow \prod_{i \in I}^w G_i$ given by $\iota_k(a) = \{a_i\}_{i \in I}$, where $a_i = e$ for $i \neq k$ and $a_k = a$, is a monomorphism of groups.
\end{enumerate}
\end{theorem}
