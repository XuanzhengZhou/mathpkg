\begin{theorem}
If $G$ is a group [resp.\ semigroup, monoid] and $a \in G$, then for all $m, n \in \mathbb{Z}$ [resp.\ $\mathbb{N}^*$, $\mathbb{N}$]:
\begin{enumerate}
    \item[(i)] $a^m a^n = a^{m+n}$ \quad (additive notation: $ma + na = (m + n)a$);
    \item[(ii)] $(a^m)^n = a^{mn}$ \quad (additive notation: $n(ma) = mna$).
\end{enumerate}
\end{theorem}
