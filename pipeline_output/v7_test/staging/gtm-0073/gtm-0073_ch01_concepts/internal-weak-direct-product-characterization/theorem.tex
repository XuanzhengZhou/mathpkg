\begin{theorem}
Let $\{N_i \mid i \in I\}$ be a family of normal subgroups of a group $G$ such that
\begin{enumerate}
    \item[(i)] $G = \langle \bigcup_{i \in I} N_i \rangle$;
    \item[(ii)] for each $k \in I$, $N_k \cap \langle \bigcup_{i \neq k} N_i \rangle = \langle e \rangle$.
\end{enumerate}
Then $G \cong \prod_{i \in I}^w N_i$.
\end{theorem}
