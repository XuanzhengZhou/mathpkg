\begin{definition}
A \textbf{semigroup} is a nonempty set $G$ together with a binary operation on $G$ which is
\begin{enumerate}
    \item[(i)] \textbf{associative}: $a(bc) = (ab)c$ for all $a, b, c \in G$;
\end{enumerate}
a \textbf{monoid} is a semigroup $G$ which contains a
\begin{enumerate}
    \item[(ii)] (two-sided) \textbf{identity element} $e \in G$ such that $ae = ea = a$ for all $a \in G$.
\end{enumerate}
A \textbf{group} is a monoid $G$ such that
\begin{enumerate}
    \item[(iii)] for every $a \in G$ there exists a (two-sided) \textbf{inverse element} $a^{-1} \in G$ such that $a^{-1}a = aa^{-1} = e$.
\end{enumerate}
A semigroup $G$ is said to be \textbf{abelian} or \textbf{commutative} if its binary operation is
\begin{enumerate}
    \item[(iv)] \textbf{commutative}: $ab = ba$ for all $a,b \in G$.
\end{enumerate}
\end{definition}
