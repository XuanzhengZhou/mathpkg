\begin{proposition}
Let $G$ be a semigroup. Then $G$ is a group if and only if the following conditions hold:
\begin{enumerate}
    \item[(i)] there exists an element $e \in G$ such that $ea = a$ for all $a \in G$ (left identity element);
    \item[(ii)] for each $a \in G$, there exists an element $a^{-1} \in G$ such that $a^{-1}a = e$ (left inverse).
\end{enumerate}
\end{proposition}
