\begin{theorem}
If $N$ is a subgroup of a group $G$, then the following conditions are equivalent.
\begin{enumerate}
    \item[(i)] Left and right congruence modulo $N$ coincide (that is, define the same equivalence relation on $G$);
    \item[(ii)] every left coset of $N$ in $G$ is also a right coset;
    \item[(iii)] $aN = Na$ for all $a \in G$;
    \item[(iv)] $aNa^{-1} \subset N$ for all $a \in G$;
    \item[(v)] $aNa^{-1} = N$ for all $a \in G$.
\end{enumerate}
\end{theorem}
