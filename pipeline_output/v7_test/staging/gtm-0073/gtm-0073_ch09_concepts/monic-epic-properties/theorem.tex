Let $f : B \to C$ and $g : C \to D$ be morphisms of a category $\mathcal{C}$.
\begin{enumerate}
\item[(i)] $f$ and $g$ monic $\implies$ $gf$ monic;
\item[(ii)] $gf$ monic $\implies$ $f$ monic;
\item[(iii)] $f$ and $g$ epic $\implies$ $gf$ epic;
\item[(iv)] $gf$ epic $\implies$ $g$ epic;
\item[(v)] $f$ is an equivalence $\implies$ $f$ is monic and epic.
\end{enumerate}
