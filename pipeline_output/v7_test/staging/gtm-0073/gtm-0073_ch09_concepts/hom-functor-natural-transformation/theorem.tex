Let $\mathcal{C}$ be a category, $T : \mathcal{C} \to \mathcal{S}$ a covariant functor, and $A$ an object of $\mathcal{C}$.
\begin{enumerate}
\item[(i)] If $\alpha : h_A \to T$ is a natural transformation from the covariant hom functor $h_A$ to $T$ and $u = \alpha_A(1_A) \in T(A)$, then for any object $C$ of $\mathcal{C}$ and $g \in \hom_{\mathcal{C}}(A,C)$,
\[
\alpha_C(g) = T(g)(u).
\]
\item[(ii)] If $u \in T(A)$ and for each object $C$ of $\mathcal{C}$, $\beta_C : \hom_{\mathcal{C}}(A,C) \to T(C)$ is the map defined by $g \mapsto T(g)(u)$, then $\beta : h_A \to T$ is a natural transformation such that $\beta_A(1_A) = u$.
\end{enumerate}
