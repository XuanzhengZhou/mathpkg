Let $R$ be a commutative ring with identity and $A,B \in \operatorname{Mat}_n R$.
\begin{enumerate}
\item Every alternating $R$-multilinear form $f$ on $\operatorname{Mat}_n R$ is a unique scalar multiple of the determinant function $d$: $f = f(I_n) \cdot d$.
\item $|AB| = |A||B|$.
\item $|A|$ is a unit in $R$ if and only if $A$ is invertible. In this case $|A^{-1}| = |A|^{-1}$.
\item If $B$ is similar to $A$, then $|B| = |A|$.
\item $|A^t| = |A|$.
\item If $A$ is triangular (upper or lower), then $|A| = a_{11}a_{22} \cdots a_{nn}$.
\end{enumerate}
