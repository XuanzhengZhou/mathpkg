Let $B$ and $C$ be $R$-modules and $f: B^n \to C$ an $R$-multilinear function.
\begin{itemize}
\item $f$ is \textbf{symmetric} if $f(b_{\sigma 1}, \ldots, b_{\sigma n}) = f(b_1, \ldots, b_n)$ for all $\sigma \in S_n$.
\item $f$ is \textbf{skew-symmetric} if $f(b_{\sigma 1}, \ldots, b_{\sigma n}) = (\operatorname{sgn} \sigma) f(b_1, \ldots, b_n)$ for all $\sigma \in S_n$.
\item $f$ is \textbf{alternating} if $f(b_1, \ldots, b_n) = 0$ whenever $b_i = b_j$ for some $i \neq j$.
\end{itemize}
