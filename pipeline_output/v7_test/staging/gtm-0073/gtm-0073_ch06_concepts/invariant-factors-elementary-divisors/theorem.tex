Let $\phi : E \to E$ be a linear transformation of a finite dimensional vector space $E$ over a field $K$.
\begin{enumerate}
\item There exist uniquely determined monic polynomials $q_1, \ldots, q_t \in K[x]$ such that $q_1 \mid q_2 \mid \cdots \mid q_t$ and $E$ is the direct sum of $\phi$-invariant subspaces $E = E_1 \oplus \cdots \oplus E_t$, where each $E_i$ is a nonzero cyclic $K[x]$-module of order $q_i$. The polynomial $q_t$ is the minimal polynomial $q_\phi$ of $\phi$.

\item There exist irreducible monic polynomials $p_1, \ldots, p_s \in K[x]$ and positive integers $m_{ij}$ ($1 \le i \le s$, $1 \le j \le k_i$) such that $E$ is the direct sum of cyclic $K[x]$-modules of orders $p_i^{m_{ij}}$. The family $\{p_i^{m_{ij}}\}$ is uniquely determined.

\item The polynomials $q_1, \ldots, q_t$ are called the \textbf{invariant factors} of $\phi$, and the polynomials $p_i^{m_{ij}}$ are called the \textbf{elementary divisors} of $\phi$.
\end{enumerate}
