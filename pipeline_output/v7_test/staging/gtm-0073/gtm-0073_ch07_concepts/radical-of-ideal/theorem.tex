The \textbf{radical} of an ideal $I$ in a commutative ring $R$ is
\[
\operatorname{Rad} I = \{ r \in R \mid r^n \in I \text{ for some } n > 0 \}.
\]

Let $I, I_1, I_2, \ldots, I_n$ be ideals of $R$. Then:
\begin{enumerate}
  \item $\operatorname{Rad}(\operatorname{Rad} I) = \operatorname{Rad} I$;
  \item $\operatorname{Rad}(I_1 I_2 \cdots I_n) = \operatorname{Rad}\bigl(\bigcap_{j=1}^{n} I_j\bigr) = \bigcap_{j=1}^{n} \operatorname{Rad} I_j$;
  \item $\operatorname{Rad}(I^m) = \operatorname{Rad} I$.
\end{enumerate}
