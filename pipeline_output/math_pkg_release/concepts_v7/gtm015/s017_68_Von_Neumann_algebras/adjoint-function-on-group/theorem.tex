\begin{definition}
For $f \in \mathcal{L}^1(G)$, the \text{adjoint} of $f$ is the function $\tilde{f}$ defined by
$$\tilde{f}(t) = \Delta(t) f(t^{-1})^* \quad (t \in G),$$
where $\Delta$ is the modular function of $G$. Then $\tilde{f} \in \mathcal{L}^1(G)$ with $\|\tilde{f}\|_1 = \|f\|_1$, and
$$\int \tilde{f}(t)\, dt = \left( \int f(t)\, dt \right)^*.$$
This induces an involution on $L^1(G)$ via $[f]^{\sim} = [\tilde{f}]$, making $L^1(G)$ a Banach $*$-algebra.
\end{definition}
