\begin{theorem}[Second Long Exact Sequence of Derived Functors]
Let $T' \xrightarrow{\mu} T \xrightarrow{\varepsilon} T''$ be a sequence of additive covariant functors $\mathfrak{M}_A \rightarrow \mathfrak{Ub}$ and natural transformations such that for every projective object $P$ in $\mathfrak{M}_A$ the sequence
\[
0 \rightarrow T' P \xrightarrow{\mu_P} T P \xrightarrow{\varepsilon_P} T'' P \rightarrow 0
\]
is exact. Then for every object $A$ in $\mathfrak{M}_A$ there exist connecting homomorphisms
\[
\omega_n: L_n T'' A \rightarrow L_{n-1} T' A, \quad n = 1, 2, \ldots
\]
such that the following sequence is exact:
\[
\cdots \rightarrow L_n T' A \xrightarrow{L_n \mu_A} L_n T A \xrightarrow{L_n \varepsilon_A} L_n T'' A \xrightarrow{\omega_n} L_{n-1} T' A \rightarrow \cdots
\]
\[
\cdots \rightarrow L_1 T'' A \xrightarrow{\omega_1} L_0 T' A \xrightarrow{L_0 \mu_A} L_0 T A \xrightarrow{L_0 \varepsilon_A} L_0 T'' A \rightarrow 0.
\]
\end{theorem}
