\textbf{Proposition 8.3.} Let $\{X^p\}$ be a filtration of $X$ with associated subquotients $X_q^p = X^p / X^q$ (for $q \leq p$).
\begin{enumerate}
\item[(i)] If the filtration is right complete, then
$$
\varprojlim_q (X_q^p, \sigma_q^p) = X^p.
$$
\item[(ii)] If the filtration is left complete, then
$$
\varinjlim_p (X_q^p, \varrho_q^p) = X_q.
$$
\end{enumerate}
