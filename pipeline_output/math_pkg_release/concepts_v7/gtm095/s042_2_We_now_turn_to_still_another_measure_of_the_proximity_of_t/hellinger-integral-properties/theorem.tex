Let $P$ and $\tilde{P}$ be probability measures on $(\Omega, \mathcal{F})$.

\begin{enumerate}
\item The Hellinger integral of order $\alpha \in (0, 1)$, defined by
$$H(\alpha; P, \tilde{P}) = E_Q\left[z^\alpha \tilde{z}^{1-\alpha}\right],$$
and consequently the Kakutani--Hellinger distance $\rho(P, \tilde{P})$, is independent of the choice of the dominating measure $Q$ (where $z = dP/dQ$, $\tilde{z} = d\tilde{P}/dQ$).

\item The function $\rho(P, \tilde{P})$ defined by
$$\rho^2(P, \tilde{P}) = \frac{1}{2} E_Q\left[\sqrt{z} - \sqrt{\tilde{z}}\right]^2$$
is a metric on the set of probability measures. That is, for all probability measures $P, \tilde{P}, P'$:
\begin{itemize}
\item $\rho(P, \tilde{P}) = 0 \iff P = \tilde{P}$,
\item $\rho(P, \tilde{P}) = \rho(\tilde{P}, P)$,
\item $\rho(P, P'') \leq \rho(P, P') + \rho(P', P'')$.
\end{itemize}
\end{enumerate}
