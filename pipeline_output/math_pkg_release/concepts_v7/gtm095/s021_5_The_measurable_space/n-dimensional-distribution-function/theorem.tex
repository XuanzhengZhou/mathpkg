\textbf{Definition 2.} An $n$-dimensional distribution function on $R^n$ is a function $F_n = F_n(x_1, \ldots, x_n)$ satisfying:

\begin{enumerate}
\item[(i)] The difference operator $\Delta_{a_i b_i} F_n(x_1, \ldots, x_n) = F_n(x_1, \ldots, b_i, \ldots, x_n) - F_n(x_1, \ldots, a_i, \ldots, x_n) \geq 0$ for all $a_i \leq b_i$, and the iterated difference
$$\Delta_{a_1 b_1} \cdots \Delta_{a_n b_n} F_n(x_1, \ldots, x_n) \geq 0;$$

\item[(ii)] $F_n(x_1, \ldots, x_n)$ is continuous on the right with respect to the variables collectively: if $x^{(k)} \downarrow x$, then $F_n(x^{(k)}) \downarrow F_n(x)$;

\item[(iii)] $F_n(+\infty, \ldots, +\infty) = 1$;

\item[(iv)] $\lim_{x \downarrow y} F_n(x_1, \ldots, x_n) = 0$ if at least one coordinate of $y$ is $-\infty$.
\end{enumerate}
