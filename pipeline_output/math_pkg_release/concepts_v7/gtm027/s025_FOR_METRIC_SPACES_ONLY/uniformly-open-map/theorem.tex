\begin{definition}[Uniformly Open Map]
A map $f$ of a uniform space $(X, \mathfrak{u})$ into a uniform space $(Y, \mathfrak{v})$ is \text{uniformly open} iff for each $U \in \mathfrak{u}$ there exists $V \in \mathfrak{v}$ such that
\[
f[U[x]] \supset V[f(x)]
\]
for every $x \in X$.
\end{definition}
