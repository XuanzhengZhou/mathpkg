\begin{definition}[Nowhere Dense Set]
A subset $A$ of a topological space $X$ is \text{nowhere dense in } $X$ iff the interior of the closure of $A$ is void; equivalently, $A$ is nowhere dense in $X$ iff the open set $X \setminus A^{-}$ is dense in $X$.
\end{definition}

\begin{proposition}
The finite union of nowhere dense sets is nowhere dense.
\end{proposition}
